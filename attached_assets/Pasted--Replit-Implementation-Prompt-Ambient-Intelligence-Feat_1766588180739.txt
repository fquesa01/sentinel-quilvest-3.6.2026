## Feature Request: Ambient Intelligence - Live Conversation Recording with AI Document Surfacing

I need you to build a new feature called "Ambient Intelligence" for our legal case management application. This feature allows users to live-record conversations while AI automatically identifies and surfaces relevant case documents based on what's being discussed.

### Overview

When a user starts an Ambient Intelligence session linked to a case, the app will:
1. Record audio from their microphone
2. Transcribe speech in real-time with speaker diarization (labeling who said what)
3. Analyze the transcript every 10-15 seconds to detect references to documents, emails, people, dates, or topics
4. Query our existing case RAG system to find matching documents
5. Display AI suggestion cards prompting the user to view relevant documents
6. Allow split-screen viewing of documents while transcription continues
7. Save the complete session record to the case when finished

### Part 5: Transcription Service

Create a new service file `services/transcription-service.ts` that handles real-time audio transcription.

**Option A: Using Deepgram (Recommended for speed)**
```typescript
import { createClient, LiveTranscriptionEvents } from '@deepgram/sdk';

interface TranscriptSegment {
  timestamp_ms: number;
  speaker_id: number;
  speaker_label: string;
  content: string;
  confidence: number;
  is_final: boolean;
}

export class TranscriptionService {
  private deepgram: any;
  private connection: any;
  private onTranscript: (segment: TranscriptSegment) => void;
  private startTime: number;

  constructor(apiKey: string) {
    this.deepgram = createClient(apiKey);
  }

  async start(onTranscript: (segment: TranscriptSegment) => void): Promise<void> {
    this.onTranscript = onTranscript;
    this.startTime = Date.now();

    this.connection = this.deepgram.listen.live({
      model: 'nova-2',
      language: 'en',
      smart_format: true,
      diarize: true, // Enable speaker identification
      punctuate: true,
      interim_results: true,
    });

    this.connection.on(LiveTranscriptionEvents.Transcript, (data: any) => {
      const transcript = data.channel.alternatives[0];
      if (transcript.transcript) {
        const segment: TranscriptSegment = {
          timestamp_ms: Date.now() - this.startTime,
          speaker_id: transcript.words?.[0]?.speaker || 0,
          speaker_label: `Speaker ${transcript.words?.[0]?.speaker || 1}`,
          content: transcript.transcript,
          confidence: transcript.confidence,
          is_final: data.is_final,
        };
        this.onTranscript(segment);
      }
    });

    this.connection.on(LiveTranscriptionEvents.Error, (error: any) => {
      console.error('Transcription error:', error);
    });
  }

  sendAudio(audioData: ArrayBuffer): void {
    if (this.connection) {
      this.connection.send(audioData);
    }
  }

  async stop(): Promise<void> {
    if (this.connection) {
      this.connection.finish();
    }
  }

  pause(): void {
    // Deepgram handles gaps in audio naturally
  }

  resume(): void {
    // Continue sending audio
  }
}
```

**Browser Audio Capture Hook:**
```typescript
// hooks/useAudioCapture.ts
import { useEffect, useRef, useState } from 'react';

export function useAudioCapture(onAudioData: (data: ArrayBuffer) => void) {
  const [isCapturing, setIsCapturing] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const mediaRecorderRef = useRef<MediaRecorder | null>(null);
  const streamRef = useRef<MediaStream | null>(null);

  const startCapture = async () => {
    try {
      const stream = await navigator.mediaDevices.getUserMedia({ 
        audio: {
          echoCancellation: true,
          noiseSuppression: true,
          sampleRate: 16000,
        } 
      });
      streamRef.current = stream;

      const mediaRecorder = new MediaRecorder(stream, {
        mimeType: 'audio/webm;codecs=opus',
      });

      mediaRecorder.ondataavailable = async (event) => {
        if (event.data.size > 0) {
          const arrayBuffer = await event.data.arrayBuffer();
          onAudioData(arrayBuffer);
        }
      };

      mediaRecorder.start(250); // Send audio chunks every 250ms
      mediaRecorderRef.current = mediaRecorder;
      setIsCapturing(true);
    } catch (err) {
      setError('Microphone access denied');
      console.error('Audio capture error:', err);
    }
  };

  const stopCapture = () => {
    if (mediaRecorderRef.current) {
      mediaRecorderRef.current.stop();
    }
    if (streamRef.current) {
      streamRef.current.getTracks().forEach(track => track.stop());
    }
    setIsCapturing(false);
  };

  return { isCapturing, error, startCapture, stopCapture };
}
```

```typescript
import { GoogleGenerativeAI } from '@google/generative-ai';

interface DetectedReference {

  trigger_quote: string;
  search_query: string;
  explanation: string;
  user_prompt: string;
  confidence: 'high' | 'medium' | 'low';
  timestamp: string;
}

export class ReferenceDetectionService {
  private genAI: GoogleGenerativeAI;
  private model: any;

  constructor(apiKey: string) {
    this.genAI = new GoogleGenerativeAI(apiKey);
    this.model = this.genAI.getGenerativeModel({ model: 'gemini-2.0-flash' });
  }

  async detectReferences(
    transcriptChunk: string,
    previousContext: string,
    caseContext: { name: string; description: string }
  ): Promise<DetectedReference[]> {
    const prompt = `
You are analyzing a live conversation transcript for a legal case management system.
The conversation is related to the following case: ${caseContext.name}
Case context: ${caseContext.description}

Analyze the following transcript segment and identify references that could be matched
to documents, emails, or data in our case management system.

TRANSCRIPT SEGMENT:
${transcriptChunk}

PREVIOUS CONTEXT (last 30 seconds):
${previousContext}

Identify any of the following:
1. DOCUMENT REFERENCES: Mentions of specific documents, reports, forms, contracts
2. EMAIL REFERENCES: Mentions of emails, correspondence, messages sent/received
3. PERSON REFERENCES: Names of people who may have sent or received communications in the case
4. DATE REFERENCES: Specific dates or time periods that could filter document searches
5. TOPIC CLUSTERS: Sustained discussion of a topic that may have related documents
6. FACTUAL ASSERTIONS: Statements presented as facts that could be verified against documents

For each reference found, respond with a JSON array. Each item should have:
{
  "type": "document|email|person|date|topic|assertion",
  "trigger_quote": "exact quote from transcript that triggered this",
  "search_query": "optimized search query for our document RAG system",
  "explanation": "1-2 sentence explanation of what was found",
  "user_prompt": "natural question to ask the user, e.g. 'You mentioned an email about expense policies. Would you like me to find it?'",
  "confidence": "high|medium|low",
  "timestamp": "timestamp from the transcript"
}

Rules:
- Only return high and medium confidence references to avoid noise
- If no significant references found, return an empty array []
- Focus on actionable references that can be searched in a document system
- The user_prompt should be conversational and helpful, not robotic
- Return ONLY valid JSON, no markdown or explanation

JSON Response:`;

    try {
      const result = await this.model.generateContent(prompt);
      const response = result.response.text();
      
      // Parse JSON from response
      const jsonMatch = response.match(/\[[\s\S]*\]/);
      if (jsonMatch) {
        return JSON.parse(jsonMatch[0]);
      }
      return [];
    } catch (error) {
      console.error('Reference detection error:', error);
```

### Part 7: Integration with Existing Case RAG

When a reference is detected, query the existing case document RAG system. Use the existing `file-search-service.ts` patterns:

```typescript
// In your session page or a new ambient-intelligence-service.ts

import { FileSearchService } from './file-search-service';
import { ReferenceDetectionService } from './reference-detection-service';

export class AmbientIntelligenceService {
  private fileSearchService: FileSearchService;
  private referenceDetector: ReferenceDetectionService;
  private transcriptBuffer: string[] = [];
  private analysisInterval: NodeJS.Timeout | null = null;

  constructor() {
    this.fileSearchService = new FileSearchService();
    this.referenceDetector = new ReferenceDetectionService(process.env.GEMINI_API_KEY!);
  }

  // Add new transcript segment to buffer
  addTranscript(segment: { content: string; timestamp_ms: number }) {
    this.transcriptBuffer.push(`[${this.formatTime(segment.timestamp_ms)}] ${segment.content}`);
    
    // Keep only last 60 seconds worth of transcript (rough estimate)
    if (this.transcriptBuffer.length > 20) {
      this.transcriptBuffer.shift();
    }
  }

  // Start periodic analysis
  startAnalysis(caseId: string, caseContext: { name: string; description: string }, onSuggestion: (suggestion: any) => void) {
    this.analysisInterval = setInterval(async () => {
      if (this.transcriptBuffer.length < 3) return; // Need some content to analyze

      const recentTranscript = this.transcriptBuffer.slice(-5).join('\n');
      const previousContext = this.transcriptBuffer.slice(-10, -5).join('\n');

      const references = await this.referenceDetector.detectReferences(
        recentTranscript,
        previousContext,
        caseContext
      );

      for (const ref of references) {
        if (ref.confidence === 'low') continue;

        // Search case documents using existing RAG
        const searchResults = await this.fileSearchService.searchCaseDocuments(
          caseId,
          ref.search_query,
          { maxResults: 5 }
        );

        if (searchResults.length > 0) {
          onSuggestion({
            ...ref,
            document_ids: searchResults.map(r => r.id),
            documents: searchResults,
          });
        }
      }
    }, 12000); // Analyze every 12 seconds
  }

  stopAnalysis() {
    if (this.analysisInterval) {
      clearInterval(this.analysisInterval);
    }
  }

  private formatTime(ms: number): string {
    const seconds = Math.floor(ms / 1000);
    const minutes = Math.floor(seconds / 60);
    const hours = Math.floor(minutes / 60);
    return `${hours.toString().padStart(2, '0')}:${(minutes % 60).toString().padStart(2, '0')}:${(seconds % 60).toString().padStart(2, '0')}`;
  }
}
```

```typescript
// hooks/useAmbientSession.ts
import { useState, useCallback, useEffect, useRef } from 'react';
import { TranscriptionService } from '@/services/transcription-service';
import { AmbientIntelligenceService } from '@/services/ambient-intelligence-service';

interface TranscriptSegment {
  id: string;
  timestamp_ms: number;
  speaker_label: string;
  content: string;
}

interface Suggestion {
  id: string;
  suggestion_type: string;
  trigger_quote: string;
  explanation: string;
  user_prompt: string;
  documents: any[];
  status: 'pending' | 'viewed' | 'dismissed';
  timestamp_ms: number;
}

export function useAmbientSession(sessionId: string, caseId: string, caseContext: { name: string; description: string }) {
  const [status, setStatus] = useState<'idle' | 'active' | 'paused' | 'ended'>('idle');
  const [transcripts, setTranscripts] = useState<TranscriptSegment[]>([]);
  const [suggestions, setSuggestions] = useState<Suggestion[]>([]);
  const [elapsedSeconds, setElapsedSeconds] = useState(0);
  const [currentDocument, setCurrentDocument] = useState<any | null>(null);
  const [viewMode, setViewMode] = useState<'transcript' | 'split' | 'document'>('transcript');

  const transcriptionService = useRef<TranscriptionService | null>(null);
  const ambientService = useRef<AmbientIntelligenceService | null>(null);
  const timerRef = useRef<NodeJS.Timeout | null>(null);

  const startSession = useCallback(async () => {
    transcriptionService.current = new TranscriptionService(process.env.NEXT_PUBLIC_DEEPGRAM_KEY!);
    ambientService.current = new AmbientIntelligenceService();

    // Handle new transcript segments
    await transcriptionService.current.start((segment) => {
      if (segment.is_final) {
        const newSegment = {
          id: crypto.randomUUID(),
          timestamp_ms: segment.timestamp_ms,
          speaker_label: segment.speaker_label,
          content: segment.content,
        };
        setTranscripts(prev => [...prev, newSegment]);
        ambientService.current?.addTranscript(segment);
        
        // Save to database
        saveTranscriptSegment(sessionId, newSegment);
      }
    });

    // Handle new suggestions
    ambientService.current.startAnalysis(caseId, caseContext, (suggestion) => {
      const newSuggestion = {
        ...suggestion,
        id: crypto.randomUUID(),
        status: 'pending' as const,
        timestamp_ms: Date.now(),
      };
      setSuggestions(prev => [newSuggestion, ...prev]);
      
      // Save to database
      saveSuggestion(sessionId, newSuggestion);
    });

    // Start timer
    timerRef.current = setInterval(() => {
      setElapsedSeconds(prev => prev + 1);
    }, 1000);

    setStatus('active');
  }, [sessionId, caseId, caseContext]);

  const pauseSession = useCallback(() => {
    transcriptionService.current?.pause();
    if (timerRef.current) clearInterval(timerRef.current);
    setStatus('paused');
  }, []);

  const resumeSession = useCallback(() => {
    transcriptionService.current?.resume();
    timerRef.current = setInterval(() => {
      setElapsedSeconds(prev => prev + 1);
    }, 1000);
    setStatus('active');
  }, []);

  const endSession = useCallback(async () => {
    transcriptionService.current?.stop();
    ambientService.current?.stopAnalysis();
    if (timerRef.current) clearInterval(timerRef.current);
    
    // Update session record in database
    await updateSessionStatus(sessionId, 'completed', elapsedSeconds);
    
    // Trigger post-processing (summary generation)
    await triggerPostProcessing(sessionId);
    
    setStatus('ended');
  }, [sessionId, elapsedSeconds]);

  const viewDocument = useCallback((document: any) => {
    setCurrentDocument(document);
    setViewMode('split');
  }, []);

  const expandDocument = useCallback(() => {
    setViewMode('document');
  }, []);

  const closeDocument = useCallback(() => {
    setCurrentDocument(null);
    setViewMode('transcript');
  }, []);

  const returnToTranscript = useCallback(() => {
    setViewMode('split');
  }, []);

  const dismissSuggestion = useCallback((suggestionId: string) => {
    setSuggestions(prev => 
      prev.map(s => s.id === suggestionId ? { ...s, status: 'dismissed' as const } : s)
    );
    updateSuggestionStatus(sessionId, suggestionId, 'dismissed');
  }, [sessionId]);

  const markSuggestionViewed = useCallback((suggestionId: string) => {
    setSuggestions(prev => 
      prev.map(s => s.id === suggestionId ? { ...s, status: 'viewed' as const } : s)
    );
    updateSuggestionStatus(sessionId, suggestionId, 'viewed');
  }, [sessionId]);

  // Cleanup on unmount
  useEffect(() => {
    return () => {
      transcriptionService.current?.stop();
      ambientService.current?.stopAnalysis();
      if (timerRef.current) clearInterval(timerRef.current);
    };
  }, []);

  return {
    status,
    transcripts,
    suggestions: suggestions.filter(s => s.status !== 'dismissed'),
    elapsedSeconds,
    currentDocument,
    viewMode,
    startSession,
    pauseSession,
    resumeSession,
    endSession,
    viewDocument,
    expandDocument,
    closeDocument,
    returnToTranscript,
    dismissSuggestion,
    markSuggestionViewed,
  };
}
```

### Part 10: Post-Session Summary Generation

After session ends, generate a summary using Gemini:

```typescript
async function generateSessionSummary(sessionId: string): Promise<void> {
  // Fetch all transcripts
  const transcripts = await getSessionTranscripts(sessionId);
  const suggestions = await getSessionSuggestions(sessionId);
  
  const fullTranscript = transcripts
    .map(t => `[${t.speaker_label}]: ${t.content}`)
    .join('\n');

  const prompt = `
Analyze this conversation transcript from a legal case management session and provide:

1. A concise summary (2-3 paragraphs) of what was discussed
2. Key moments that should be tagged for review (with timestamps if available)
3. List of all documents/evidence that were referenced or viewed

TRANSCRIPT:
${fullTranscript}

DOCUMENTS VIEWED DURING SESSION:
${suggestions.filter(s => s.status === 'viewed').map(s => s.explanation).join('\n')}

Respond in JSON format:
{
  "summary": "...",
  "key_moments": [
    {"description": "...", "type": "reference|assertion|discrepancy|decision"}
  ],
  "documents_referenced": [
    {"title": "...", "status": "viewed|mentioned"}
  ]
}`;

  const result = await model.generateContent(prompt);
  const summary = JSON.parse(result.response.text());
  
  await saveSessionSummary(sessionId, summary);
}
```
