function riskColor(risk) {
  if (risk === "high") return "bg-danger";
  if (risk === "medium") return "bg-warning";
  if (risk === "low") return "bg-success";
  return "bg-primary";
}

export default function NetworkGraph() {
  return (
    <div className="relative h-72">
      {/* Center node */}
      <div className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2">
        <div className="flex flex-col items-center">
          <div className="h-24 w-24 rounded-full bg-primary flex items-center justify-center text-white text-xl font-semibold shadow-sm">
            RW
          </div>
          <div className="mt-2 text-xs text-navy font-medium">
            Robert Williams
          </div>
          <div className="text-[11px] text-slate">1,333 messages</div>
        </div>
      </div>

      {/* Spokes & nodes */}
      {nodes.map((n) => {
        const rad = (n.angle * Math.PI) / 180;
        const x = Math.cos(rad) * n.distance;
        const y = Math.sin(rad) * n.distance;

        return (
          <div
            key={n.name}
            className="absolute top-1/2 left-1/2"
            style={{
              transform: `translate(${x}px, ${y}px)`
            }}
          >
            {/* Line back to center */}
            <svg className="absolute -z-10 -top-10 -left-10" width="120" height="120">
              <line
                x1="60"
                y1="60"
                x2="60"
                y2="10"
                stroke="#E0E6F0"
                strokeWidth="1.5"
              />
            </svg>
            <div className="flex flex-col items-center group">
              <div
                className={`h-9 w-9 rounded-full ${riskColor(
                  n.risk
                )} bg-opacity-80 flex items-center justify-center text-[11px] text-white shadow-sm`}
              >
                {n.name.split(" ")[0][0]}
              </div>
              <div className="mt-1 text-[11px] text-slate text-center max-w-[80px]">
                {n.name}
              </div>

}

ActivityHeatStrip.jsx
export default function ActivityHeatStrip({ data }) {
  function color(count, flagged) {
    if (flagged) return "bg-danger/80";
    if (count === 0) return "bg-lightgray";
    if (count < 40) return "bg-[#D4DFEF]";
    if (count < 80) return "bg-[#8AA2C5]";
    return "bg-primary";
  }

  return (
    <div className="space-y-2">
      <div className="flex flex-wrap gap-1">
        {data.map((d) => (
          <div
            key={d.id}
            className={`h-4 w-4 rounded-sm ${color(d.count, d.flagged)} cursor-default`}
            title={`${d.label} – ${d.count} messages${
              d.flagged ? " – flagged risk events" : ""
            }`}
          />
        ))}
      </div>
      <div className="text-[11px] text-slate">
        Darker squares indicate higher message volume. Red days include flagged risk events.
      </div>
    </div>
  );
}
