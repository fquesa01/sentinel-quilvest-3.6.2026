# Auto Search Term Generation Feature Specification
## Sentinel Counsel LLP - Discovery & Case Intelligence Module

---

## Overview

This feature automates the creation of Boolean search terms from legal documents to streamline document review. It consists of three interconnected capabilities:

1. **Discovery Search Terms**: Parse opposing counsel's Requests for Production (RFPs) to generate search terms per request
2. **Case Proving Search Terms**: Parse complaints to generate search terms per claim/cause of action
3. **Privilege Detection**: Auto-identify privileged communications and generate downloadable privilege logs

The workflow follows Sentinel's proven two-stage methodology: Boolean queries filter documents to relevant subsets, then focused AI analysis on pre-filtered content.

---

```sql
-- Search Term Sets (parent container for a set of generated terms)
CREATE TABLE search_term_sets (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  case_id UUID NOT NULL REFERENCES cases(id) ON DELETE CASCADE,
  
  -- Source Document Info
  source_type VARCHAR(30) NOT NULL CHECK (
    source_type IN ('rfp', 'complaint', 'subpoena', 'interrogatory', 'manual')
  ),
  source_document_id UUID REFERENCES case_pleadings(id), -- Link to uploaded pleading
  source_document_name TEXT,
  
  -- Metadata
  name TEXT NOT NULL, -- e.g., "Defendant's First RFPs" or "Complaint Claims"
  description TEXT,
  
  -- Processing Status
  generation_status VARCHAR(20) DEFAULT 'pending' CHECK (
    generation_status IN ('pending', 'processing', 'completed', 'failed')
  ),
  generation_error TEXT,
  
  -- Execution Stats
  total_requests INTEGER DEFAULT 0, -- Number of RFPs or claims
  last_executed_at TIMESTAMP WITH TIME ZONE,
  documents_searched INTEGER DEFAULT 0,
  documents_tagged INTEGER DEFAULT 0,
  
  -- Audit
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  created_by UUID REFERENCES users(id),
  updated_by UUID REFERENCES users(id)
);

CREATE INDEX idx_search_term_sets_case ON search_term_sets(case_id);
CREATE INDEX idx_search_term_sets_source ON search_term_sets(source_type);

-- Individual Requests/Claims with their Search Terms
CREATE TABLE search_term_items (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  search_term_set_id UUID NOT NULL REFERENCES search_term_sets(id) ON DELETE CASCADE,
  
  -- Request/Claim Info
  item_number INTEGER NOT NULL, -- RFP #1, Claim #2, etc.
  item_type VARCHAR(30) NOT NULL CHECK (
    item_type IN ('rfp_request', 'complaint_claim', 'privilege_category', 'custom')
  ),
  
  -- Full Text of Request/Claim
  full_text TEXT NOT NULL, -- Complete text of the RFP request or claim
  summary TEXT, -- AI-generated summary
  
  -- For Complaints: Cause of Action Details
  cause_of_action TEXT, -- e.g., "Breach of Contract", "Negligence"
  legal_elements JSONB, -- Elements that must be proven
  
  -- Generated Search Terms (editable by user)
  search_terms JSONB NOT NULL DEFAULT '[]', 
  -- Structure: [
  --   { 
  --     "id": "uuid",
  --     "term": "contract AND breach", 
  --     "type": "boolean", 
  --     "enabled": true,
  --     "ai_generated": true,
  --     "rationale": "Why this term was suggested"
  --   }
  -- ]
  
  -- Combined Boolean String (auto-generated from enabled terms)
  combined_boolean_string TEXT,
  
  -- Execution Status
  execution_status VARCHAR(20) DEFAULT 'pending' CHECK (
    execution_status IN ('pending', 'running', 'completed', 'failed')
  ),
  last_executed_at TIMESTAMP WITH TIME ZONE,
  documents_matched INTEGER DEFAULT 0,
  
  -- For Privilege Items
  is_privilege_category BOOLEAN DEFAULT FALSE,
  
  -- Audit
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_search_term_items_set ON search_term_items(search_term_set_id);
CREATE INDEX idx_search_term_items_type ON search_term_items(item_type);

-- Document Tags (results of search execution)
CREATE TABLE document_search_tags (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  case_id UUID NOT NULL REFERENCES cases(id) ON DELETE CASCADE,
  document_id UUID NOT NULL, -- References document in doc review
  
  -- Tag Source
  search_term_item_id UUID REFERENCES search_term_items(id) ON DELETE SET NULL,
  tag_source VARCHAR(30) NOT NULL CHECK (
    tag_source IN ('search_term', 'ai_review', 'manual', 'privilege_scan')
  ),
  
  -- Tag Details
  tag_name TEXT NOT NULL, -- e.g., "RFP-001", "Claim-Negligence", "Privileged"
  tag_category VARCHAR(50), -- e.g., "discovery_response", "case_proving", "privilege"
  tag_color VARCHAR(7) DEFAULT '#6B7280', -- Hex color for UI
  
  -- Match Details
  matched_terms TEXT[], -- Which specific terms matched
  match_snippets JSONB, -- Highlighted text snippets showing matches
  confidence_score DECIMAL(5,4), -- For AI-based tagging
  
  -- Review Status
  review_status VARCHAR(20) DEFAULT 'pending' CHECK (
    review_status IN ('pending', 'confirmed', 'rejected', 'needs_review')
  ),
  reviewed_by UUID REFERENCES users(id),
  reviewed_at TIMESTAMP WITH TIME ZONE,
  review_notes TEXT,
  
  -- Audit
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  
  UNIQUE(document_id, search_term_item_id) -- Prevent duplicate tags

CREATE INDEX idx_document_tags_case ON document_search_tags(case_id);
```

## AI Services

### RFP Parser Service

```typescript
// services/rfpParserService.ts
import { GoogleGenerativeAI } from '@google/generative-ai';

interface ParsedRFP {
  requestNumber: number;
  fullText: string;
  summary: string;
  searchTerms: SearchTerm[];
  combinedBoolean: string;
}

interface SearchTerm {
  id: string;
  term: string;
  type: 'boolean' | 'phrase' | 'proximity' | 'wildcard';
  enabled: boolean;
  aiGenerated: boolean;
  rationale: string;
}

export class RFPParserService {
  private genAI: GoogleGenerativeAI;
  
  constructor() {
    this.genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY!);
  }
  
  async parseRFPDocument(documentText: string): Promise<ParsedRFP[]> {
    const model = this.genAI.getGenerativeModel({ model: 'gemini-1.5-pro' });
    
    const prompt = `You are a legal discovery expert. Analyze this Request for Production document and extract each individual request.

For each request, generate Boolean search terms that would find responsive documents. Follow these guidelines:

## Boolean Search Best Practices:
1. Use AND to require all terms: "contract AND breach AND damages"
2. Use OR for synonyms/alternatives: "(email OR correspondence OR communication)"
3. Use quotation marks for exact phrases: "employment agreement"
4. Use proximity operators where supported: "termination W/10 cause"
5. Use wildcards for variations: "employ*" matches employ, employee, employment, employer
6. Consider common misspellings and abbreviations
7. Think about how the concepts would appear in actual business documents

## For Each RFP Request, Generate:
1. **Primary Terms**: Core concepts that MUST appear (use AND)
2. **Synonym Groups**: Alternative words for the same concept (use OR within parentheses)
3. **Contextual Terms**: Terms that indicate relevance when combined with primary terms
4. **Exclusion Terms**: Terms that might indicate non-responsive documents (use NOT sparingly)

## Example Output for "All documents relating to the termination of John Smith's employment":

Primary Search String:
(terminat* OR discharg* OR "let go" OR fired OR separation) AND (Smith OR "John Smith") AND (employ* OR job OR position OR "human resources" OR HR)

Alternative Searches:
1. "performance review" AND Smith - rationale: termination often preceded by performance issues
2. "exit interview" AND Smith - rationale: exit documentation
3. "severance" AND Smith - rationale: severance discussions indicate termination

Now analyze this document:

---
${documentText}
---

Return JSON array:
[
  {
    "requestNumber": 1,
    "fullText": "Complete text of the request",
    "summary": "Brief 1-2 sentence summary",
    "searchTerms": [
      {
        "id": "uuid-here",
        "term": "Boolean search string",
        "type": "boolean",
        "enabled": true,
        "aiGenerated": true,
        "rationale": "Why this search term is relevant"
      }
    ],
    "combinedBoolean": "All enabled terms combined with OR"
  }
]`;

    const result = await model.generateContent(prompt);
    const response = result.response.text();
    
    // Parse JSON from response
    const jsonMatch = response.match(/\[[\s\S]*\]/);
    if (!jsonMatch) {
      throw new Error('Failed to parse RFP response');
    }
    
    const parsed: ParsedRFP[] = JSON.parse(jsonMatch[0]);
    
    // Add UUIDs to search terms if not present
    return parsed.map(rfp => ({
      ...rfp,
      searchTerms: rfp.searchTerms.map(term => ({
        ...term,
        id: term.id || crypto.randomUUID()
      }))
    }));
  }
}
```

```typescript
// services/complaintParserService.ts
import { GoogleGenerativeAI } from '@google/generative-ai';

interface ParsedClaim {
  claimNumber: number;
  causeOfAction: string;
  fullText: string;
  summary: string;
  legalElements: LegalElement[];
  searchTerms: SearchTerm[];
  combinedBoolean: string;
}

interface LegalElement {
  element: string;
  description: string;
  searchTerms: SearchTerm[];
}

export class ComplaintParserService {
  private genAI: GoogleGenerativeAI;
  
  constructor() {
    this.genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY!);
  }
  
  async parseComplaint(documentText: string): Promise<ParsedClaim[]> {
    const model = this.genAI.getGenerativeModel({ model: 'gemini-1.5-pro' });
    
    const prompt = `You are a litigation support expert. Analyze this complaint and extract each cause of action/claim.

For each claim, you need to:
1. Identify the cause of action (e.g., "Breach of Contract", "Negligence", "Fraud")
2. Break down the legal elements that must be proven
3. Generate Boolean search terms to find documents that prove each element

## Legal Element Analysis:
For each cause of action, identify what must be proven. Example for Breach of Contract:
- Element 1: Existence of a valid contract
- Element 2: Plaintiff's performance or excuse for non-performance  
- Element 3: Defendant's breach
- Element 4: Damages resulting from breach

## Search Term Generation Strategy:
For proving a case, focus on:
1. **Documentary Evidence**: Contracts, agreements, communications showing obligations
2. **Performance Evidence**: Invoices, deliverables, correspondence showing work done
3. **Breach Evidence**: Complaints, failure notices, deviation from requirements
4. **Damages Evidence**: Financial records, lost revenue, mitigation efforts
5. **Timeline Evidence**: Dates, deadlines, milestone documents

## Boolean Search Best Practices:
- Use AND to require concepts: "contract AND breach"
- Use OR for synonyms: "(agreement OR contract OR understanding)"
- Use phrases: "material breach"
- Use wildcards: "perform*" for perform, performance, performed
- Use proximity: "deliver* W/15 fail*"

Now analyze this complaint:

---
${documentText}
---

Return JSON array:
[
  {
    "claimNumber": 1,
    "causeOfAction": "Breach of Contract",
    "fullText": "Complete text of the claim from the complaint",
    "summary": "Brief summary of what is alleged",
    "legalElements": [
      {
        "element": "Existence of Valid Contract",
        "description": "Plaintiff must prove a valid contract existed",
        "searchTerms": [
          {
            "id": "uuid",
            "term": "(contract OR agreement OR \"master services agreement\" OR MSA)",
            "type": "boolean",
            "enabled": true,
            "aiGenerated": true,
            "rationale": "Find the underlying contract documents"
          }
        ]
      }
    ],
    "searchTerms": [
      {
        "id": "uuid",
        "term": "Combined search for this entire claim",
        "type": "boolean",
        "enabled": true,
        "aiGenerated": true,
        "rationale": "Overall search for claim-related documents"
      }
    ],
    "combinedBoolean": "All enabled terms for this claim"
  }
]`;

    const result = await model.generateContent(prompt);
    const response = result.response.text();
    
    const jsonMatch = response.match(/\[[\s\S]*\]/);
    if (!jsonMatch) {
      throw new Error('Failed to parse complaint response');
    }
    
    return JSON.parse(jsonMatch[0]);
  }
```

### Boolean Search Executor Service

```typescript
// services/booleanSearchService.ts
import { db } from '../db';
import { sql } from 'drizzle-orm';

interface SearchResult {
  documentId: string;
  matchedTerms: string[];
  snippets: { term: string; context: string; position: number }[];
  score: number;
}

export class BooleanSearchService {
  
  // Parse Boolean string into PostgreSQL tsquery
  parseBooleanToTsquery(booleanString: string): string {
    let query = booleanString;
    
    // Handle quoted phrases - convert to tsquery phrase
    query = query.replace(/"([^"]+)"/g, (_, phrase) => {
      return phrase.split(/\s+/).join(' <-> ');
    });
    
    // Handle OR
    query = query.replace(/\bOR\b/gi, ' | ');
    
    // Handle AND
    query = query.replace(/\bAND\b/gi, ' & ');
    
    // Handle NOT
    query = query.replace(/\bNOT\b/gi, ' !');
    
    // Handle wildcards - PostgreSQL uses :* for prefix matching
    query = query.replace(/(\w+)\*/g, '$1:*');
    
    // Handle parentheses (already PostgreSQL compatible)
    
    // Handle proximity operators (W/n) - convert to distance operator
    query = query.replace(/(\S+)\s+W\/(\d+)\s+(\S+)/gi, (_, term1, distance, term2) => {
      return `${term1} <${distance}> ${term2}`;
    });
    
    return query;
  }
  
  // Execute search against document collection
  async executeSearch(
    caseId: string,
    booleanString: string,
    options: {
      documentSetId?: string; // Limit to specific document set
      limit?: number;
      offset?: number;
    } = {}
  ): Promise<{ results: SearchResult[]; totalCount: number }> {
    const tsquery = this.parseBooleanToTsquery(booleanString);
    
    // Build the search query
    const query = sql`
      SELECT 
        d.id as document_id,
        ts_headline('english', d.content, to_tsquery('english', ${tsquery}),
          'StartSel=<mark>, StopSel=</mark>, MaxWords=35, MinWords=15') as snippet,
        ts_rank(d.search_vector, to_tsquery('english', ${tsquery})) as score
      FROM case_documents d
      WHERE d.case_id = ${caseId}
        AND d.search_vector @@ to_tsquery('english', ${tsquery})
        ${options.documentSetId ? sql`AND d.document_set_id = ${options.documentSetId}` : sql``}
      ORDER BY score DESC
      LIMIT ${options.limit || 1000}
      OFFSET ${options.offset || 0}
    `;
    
    const results = await db.execute(query);
    
    // Get total count
    const countQuery = sql`
      SELECT COUNT(*) as total
      FROM case_documents d
      WHERE d.case_id = ${caseId}
        AND d.search_vector @@ to_tsquery('english', ${tsquery})
        ${options.documentSetId ? sql`AND d.document_set_id = ${options.documentSetId}` : sql``}
    `;
    
    const countResult = await db.execute(countQuery);
    
    return {
      results: results.rows.map((row: any) => ({
        documentId: row.document_id,
        matchedTerms: [booleanString], // In real impl, extract actual matched terms
        snippets: [{ term: booleanString, context: row.snippet, position: 0 }],
        score: row.score
      })),
      totalCount: parseInt(countResult.rows[0]?.total || '0')
    };
  }
  
  // Execute multiple search terms and tag documents
  async executeAndTagDocuments(
    caseId: string,
    searchTermItemId: string,
    searchTerms: SearchTerm[],
    tagInfo: { tagName: string; tagCategory: string; tagColor: string },
    options: { documentSetId?: string } = {}
  ): Promise<{ documentsTagged: number; documentsSearched: number }> {
    const enabledTerms = searchTerms.filter(t => t.enabled);
    
    if (enabledTerms.length === 0) {
      return { documentsTagged: 0, documentsSearched: 0 };
    }
    
    // Combine all enabled terms with OR for maximum recall
    const combinedBoolean = enabledTerms.map(t => `(${t.term})`).join(' OR ');
    
    const { results, totalCount } = await this.executeSearch(caseId, combinedBoolean, options);
    
    // Tag each matching document
    for (const result of results) {
      await db.insert(documentSearchTags).values({
        caseId,
        documentId: result.documentId,
        searchTermItemId,
        tagSource: 'search_term',
        tagName: tagInfo.tagName,
        tagCategory: tagInfo.tagCategory,
        tagColor: tagInfo.tagColor,
        matchedTerms: result.matchedTerms,
        matchSnippets: result.snippets,
        confidenceScore: result.score.toString(),
        reviewStatus: 'pending'
      }).onConflictDoUpdate({
        target: [documentSearchTags.documentId, documentSearchTags.searchTermItemId],
        set: {
          matchedTerms: result.matchedTerms,
          matchSnippets: result.snippets,
          confidenceScore: result.score.toString()
        }
      });
    }
    
    return {
      documentsTagged: results.length,
      documentsSearched: totalCount
    };
  }
}
```

```typescript
          left: { style: 'thin' },
          bottom: { style: 'thin' },
          right: { style: 'thin' }
        };
      });
    });
    
    // Send file
    res.setHeader('Content-Type', 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet');
    res.setHeader('Content-Disposition', `attachment; filename="privilege-log-${caseId}.xlsx"`);
    
    await workbook.xlsx.write(res);
    res.end();
    
  } catch (error) {
    console.error('Failed to export privilege log:', error);
    res.status(500).json({ success: false, error: 'Failed to export' });
  }
});

function formatPrivilegeType(type: string): string {
  const labels: Record<string, string> = {
    'attorney_client': 'Attorney-Client Privilege',
    'work_product': 'Work Product - Fact',
    'work_product_opinion': 'Work Product - Opinion',
    'common_interest': 'Common Interest',
    'joint_defense': 'Joint Defense',
    'deliberative_process': 'Deliberative Process',
    'other': 'Other Privilege'
  };
  return labels[type] || type;
}

export default router;
```

```tsx
// components/search-terms/SearchTermItemView.tsx
import React, { useState } from 'react';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Badge } from '@/components/ui/badge';
import { Input } from '@/components/ui/input';
import { Switch } from '@/components/ui/switch';
import { 
  Play, Edit2, Plus, Trash2, ChevronDown, ChevronUp, 
  FileSearch, CheckCircle, Loader2, AlertCircle 
} from 'lucide-react';
import { Collapsible, CollapsibleContent, CollapsibleTrigger } from '@/components/ui/collapsible';

interface SearchTerm {
  id: string;
  term: string;
  type: string;
  enabled: boolean;
  aiGenerated: boolean;
  rationale?: string;
}

interface SearchTermItemViewProps {
  item: {
    id: string;
    itemNumber: number;
    itemType: string;
    fullText: string;
    summary?: string;
    causeOfAction?: string;
    legalElements?: any[];
    searchTerms: SearchTerm[];
    combinedBooleanString?: string;
    executionStatus: string;
    documentsMatched?: number;
    isPrivilegeCategory?: boolean;
  };
  caseId: string;
  onUpdate: () => void;
}

export function SearchTermItemView({ item, caseId, onUpdate }: SearchTermItemViewProps) {
  const [isOpen, setIsOpen] = useState(false);
  const [searchTerms, setSearchTerms] = useState<SearchTerm[]>(item.searchTerms);
  const [isEditing, setIsEditing] = useState(false);
  const [isSaving, setIsSaving] = useState(false);
  const [isExecuting, setIsExecuting] = useState(false);
  const [newTerm, setNewTerm] = useState('');

  const itemLabel = item.isPrivilegeCategory 
    ? item.summary 
    : item.itemType === 'rfp_request' 
      ? `Request No. ${item.itemNumber}` 
      : `${item.causeOfAction || `Claim ${item.itemNumber}`}`;

  const toggleTerm = (termId: string) => {
    setSearchTerms(terms => 
      terms.map(t => t.id === termId ? { ...t, enabled: !t.enabled } : t)
    );
  };

  const updateTerm = (termId: string, newTermText: string) => {
    setSearchTerms(terms =>
      terms.map(t => t.id === termId ? { ...t, term: newTermText } : t)
    );
  };

  const deleteTerm = (termId: string) => {
    setSearchTerms(terms => terms.filter(t => t.id !== termId));
  };

  const addTerm = () => {
    if (!newTerm.trim()) return;
    setSearchTerms(terms => [
      ...terms,
      {
        id: crypto.randomUUID(),
        term: newTerm,
        type: 'boolean',
        enabled: true,
        aiGenerated: false
      }
    ]);
    setNewTerm('');
  };

  const handleSave = async () => {
    setIsSaving(true);
    try {
      const combinedBoolean = searchTerms
        .filter(t => t.enabled)
        .map(t => `(${t.term})`)
        .join(' OR ');

      await fetch(`/api/cases/${caseId}/search-term-sets/${item.id.split('-')[0]}/items/${item.id}`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ searchTerms, combinedBooleanString: combinedBoolean })
      });
      
      setIsEditing(false);
      onUpdate();
    } catch (error) {
      console.error('Failed to save:', error);
    } finally {
      setIsSaving(false);
    }
  };

  const handleExecute = async () => {
    setIsExecuting(true);
    try {
      await fetch(`/api/cases/${caseId}/search-term-sets/${item.id.split('-')[0]}/items/${item.id}/execute`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({})
      });
      onUpdate();
    } catch (error) {
      console.error('Failed to execute:', error);
    } finally {
      setIsExecuting(false);
    }
  };

  return (
    <Card className="bg-gray-800/50 border-gray-700">
      <Collapsible open={isOpen} onOpenChange={setIsOpen}>
        <CollapsibleTrigger asChild>
          <CardHeader className="cursor-pointer hover:bg-gray-800/80 transition-colors">
            <div className="flex items-center justify-between">
              <div className="flex items-center gap-3">
                {isOpen ? <ChevronUp className="w-5 h-5 text-gray-400" /> : <ChevronDown className="w-5 h-5 text-gray-400" />}
                <div>
                  <CardTitle className="text-white flex items-center gap-2">
                    {item.isPrivilegeCategory && <Badge variant="outline" className="bg-red-500/20 text-red-400 border-red-500/30">Privilege</Badge>}
                    {itemLabel}
                  </CardTitle>
                  {item.summary && !item.isPrivilegeCategory && (
                    <p className="text-sm text-gray-400 mt-1">{item.summary}</p>
                  )}
                </div>
              </div>
              <div className="flex items-center gap-2">
                {item.executionStatus === 'completed' && (
                  <Badge variant="outline" className="bg-green-500/20 text-green-400 border-green-500/30">
                    <CheckCircle className="w-3 h-3 mr-1" />
                    {item.documentsMatched} docs matched
                  </Badge>
                )}
                <Badge variant="outline" className="bg-gray-600/50 text-gray-300 border-gray-500">
                  {searchTerms.filter(t => t.enabled).length} active terms
                </Badge>
              </div>
            </div>
          </CardHeader>
        </CollapsibleTrigger>

        <CollapsibleContent>
          <CardContent className="border-t border-gray-700 pt-4 space-y-4">
            {/* Full Text of Request/Claim */}
            <div className="bg-gray-900/50 p-4 rounded-lg border border-gray-700">
              <h4 className="text-sm font-medium text-gray-400 mb-2">
                {item.itemType === 'rfp_request' ? 'Full Request Text' : 'Claim Text'}
              </h4>
              <p className="text-white text-sm whitespace-pre-wrap">{item.fullText}</p>
            </div>

            {/* Legal Elements (for complaints) */}
            {item.legalElements && item.legalElements.length > 0 && (
              <div className="bg-gray-900/50 p-4 rounded-lg border border-gray-700">
                <h4 className="text-sm font-medium text-gray-400 mb-2">Legal Elements to Prove</h4>
                <ul className="space-y-2">
                  {item.legalElements.map((el: any, idx: number) => (
                    <li key={idx} className="text-sm text-gray-300">
                      <span className="font-medium text-white">{idx + 1}. {el.element}:</span>{' '}
                      {el.description}
                    </li>
                  ))}
                </ul>
              </div>
            )}

            {/* Search Terms */}
            <div>
              <div className="flex items-center justify-between mb-3">
                <h4 className="text-sm font-medium text-gray-400">
                  Boolean Search Terms
                </h4>
                <div className="flex gap-2">
                  {isEditing ? (
                    <>
                      <Button 
                        size="sm" 
                        variant="outline"
                        onClick={() => {
                          setSearchTerms(item.searchTerms);
                          setIsEditing(false);
                        }}
                        className="border-gray-600"
                      >
                        Cancel
                      </Button>
                      <Button 
                        size="sm"
                        onClick={handleSave}
                        disabled={isSaving}
                        className="bg-green-600 hover:bg-green-700"
                      >
                        {isSaving ? <Loader2 className="w-4 h-4 animate-spin" /> : 'Save Changes'}

                    </>
                  ) : (
                    <Button 
                      size="sm" 
                      variant="outline"
                      onClick={() => setIsEditing(true)}
                      className="border-gray-600"
                    >
                      <Edit2 className="w-4 h-4 mr-2" />
                      Edit Terms
                    </Button>
                  )}
                </div>
              </div>

              <div className="space-y-2">
                {searchTerms.map((term) => (
                  <div 
                    key={term.id}
                    className={`flex items-start gap-3 p-3 rounded-lg border ${
                      term.enabled 
                        ? 'bg-gray-900/30 border-gray-600' 
                        : 'bg-gray-900/10 border-gray-700 opacity-60'
                    }`}
                  >
                    <Switch
                      checked={term.enabled}
                      onCheckedChange={() => toggleTerm(term.id)}
                      disabled={!isEditing}
                    />
                    <div className="flex-1 min-w-0">
                      {isEditing ? (
                        <Input
                          value={term.term}
                          onChange={(e) => updateTerm(term.id, e.target.value)}
                          className="bg-gray-800 border-gray-600 text-white font-mono text-sm"
                        />
                      ) : (
                        <code className="text-sm text-green-400 font-mono break-all">
                          {term.term}

                        </Button>

                {isEditing && (

                    <Button
                      size="sm"
                      onClick={addTerm}
                      disabled={!newTerm.trim()}
                      className="bg-orange-500 hover:bg-orange-600"
                    >
                      <Plus className="w-4 h-4" />
                    </Button>
                  </div>
                )}
              </div>
            </div>

            {/* Combined Boolean String Preview */}
            <div className="bg-gray-900/50 p-4 rounded-lg border border-gray-700">
              <h4 className="text-sm font-medium text-gray-400 mb-2">Combined Search Query</h4>
              <code className="text-xs text-blue-400 font-mono break-all">
                {searchTerms.filter(t => t.enabled).map(t => `(${t.term})`).join(' OR ') || 'No terms enabled'}
              </code>
            </div>

            {/* Execute Button */}
            <div className="flex justify-end pt-2">
              <Button
                onClick={handleExecute}
                disabled={isExecuting || searchTerms.filter(t => t.enabled).length === 0}
                className="bg-orange-500 hover:bg-orange-600"

                {isExecuting ? (

                    <Loader2 className="w-4 h-4 mr-2 animate-spin" />
                    Searching...
                  </>
                ) : (
                  <>
                    <Play className="w-4 h-4 mr-2" />
                    Run Search & Tag Documents
                  </>
                )}
              </Button>
            </div>
          </CardContent>
        </CollapsibleContent>
      </Collapsible>
    </Card>
  );
}
```

```tsx
                    </TableCell>
                    <TableCell className="text-gray-300 text-sm">
                      {entry.documentType || '-'}
                    </TableCell>
                    <TableCell className="text-white text-sm">
                      {entry.author || '-'}
                      {entry.authorTitle && (
                        <span className="text-gray-500 text-xs block">{entry.authorTitle}</span>
                      )}
                    </TableCell>
                    <TableCell className="text-gray-300 text-sm max-w-[200px] truncate">
                      {entry.recipients?.join('; ') || '-'}
                    </TableCell>
                    <TableCell>
                      <Badge 
                        variant="outline" 
                        className={privilegeTypeColors[entry.privilegeType] || privilegeTypeColors.other}
                      >
                        {privilegeTypeLabels[entry.privilegeType] || entry.privilegeType}
                      </Badge>
                    </TableCell>
                    <TableCell className="text-gray-300 text-sm max-w-[300px]">
                      {entry.privilegeDescription}
                    </TableCell>
                    <TableCell>
                      <Badge 
                        variant="outline"
                        className={
                          entry.logStatus === 'final' 
                            ? 'bg-green-500/20 text-green-400 border-green-500/30'
                            : entry.logStatus === 'reviewed'
                            ? 'bg-blue-500/20 text-blue-400 border-blue-500/30'
                            : 'bg-yellow-500/20 text-yellow-400 border-yellow-500/30'
                        }
                      >
                        {entry.logStatus}
                      </Badge>
                    </TableCell>
                  </TableRow>
                ))
              )}
            </TableBody>
          </Table>
        </div>
      </div>
    </div>
  );
}
```
