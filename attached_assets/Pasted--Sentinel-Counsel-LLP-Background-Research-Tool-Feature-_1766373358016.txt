# Sentinel Counsel LLP - Background Research Tool

## Feature Overview

AI-powered preliminary due diligence tool that performs comprehensive background research on target companies, individuals, and entities for business transactions, acquisitions, and investment decisions.

**Research Categories:**
1. **Corporate Structure** - Entity formation, ownership, subsidiaries, affiliates
2. **Leadership & Key Personnel** - Officers, directors, beneficial owners, key employees
3. **Litigation History** - Federal/state lawsuits, bankruptcies, judgments
4. **UCC Filings & Liens** - Secured transactions, tax liens, judgment liens
5. **Regulatory & Compliance** - Agency actions, investigations, sanctions
6. **News & Media** - Press coverage, controversies, reputation issues
7. **Financial Indicators** - SEC filings, credit indicators, financial health
8. **Red Flags & Risk Assessment** - Synthesized risk analysis

```typescript
// services/sec-edgar-api.service.ts

/**
 * SEC EDGAR API Service
 * 
 * The SEC provides free, public API access to all EDGAR filings.
 * No API key required - just need to provide a User-Agent header with contact info.
 * 
 * API Documentation: https://www.sec.gov/developer
 * Rate Limits: 10 requests per second
 */

export interface EdgarCompanyInfo {
  cik: string;

  exchanges: string[];

  formerNames?: Array<{ name: string; from: string; to: string }>;
}

export class SecEdgarApiService {
  private readonly BASE_URL = 'https://data.sec.gov';
  private readonly SEARCH_URL = 'https://efts.sec.gov/LATEST/search-index';
  private readonly BROWSE_URL = 'https://www.sec.gov/cgi-bin/browse-edgar';
  
  // SEC requires a User-Agent header with company name and contact email
  private readonly headers: Record<string, string>;
  
  constructor() {
    const userAgent = process.env.SEC_EDGAR_USER_AGENT || 'Sentinel Counsel LLP contact@sentinelcounsel.com';
    this.headers = {
      'User-Agent': userAgent,
      'Accept': 'application/json',
    };
  }

  /**
   * Search for companies by name
   * Returns matching companies with their CIK numbers
   */
  async searchCompany(companyName: string): Promise<EdgarSearchResult[]> {
    try {
      // Use the full-text search API
      const response = await fetch(
        `${this.SEARCH_URL}?q="${encodeURIComponent(companyName)}"&dateRange=custom&startdt=2020-01-01&enddt=${new Date().toISOString().split('T')[0]}&forms=10-K,10-Q,8-K`,
        { headers: this.headers }
      );
      
      if (!response.ok) {
        throw new Error(`SEC search failed: ${response.statusText}`);
      }
      
      const data = await response.json();
      
      // Extract unique companies from search results
      const companiesMap = new Map<string, EdgarSearchResult>();
      
      for (const hit of data.hits?.hits || []) {
        const cik = hit._source.ciks?.[0];
        if (cik && !companiesMap.has(cik)) {
          companiesMap.set(cik, {
            cik,
            companyName: hit._source.display_names?.[0] || hit._source.entity,
            filingType: hit._source.form,
            filedDate: hit._source.file_date,
            accessionNumber: hit._source.adsh,
          });
        }
      }
      
      return Array.from(companiesMap.values());
    } catch (error) {
      console.error('SEC company search failed:', error);
      return [];
    }
  }

  /**
   * Look up a company by ticker symbol
   */
  async getCompanyByTicker(ticker: string): Promise<EdgarCompanyInfo | null> {
    try {
      // SEC provides a ticker-to-CIK mapping file
      const tickerResponse = await fetch(
        'https://www.sec.gov/files/company_tickers.json',
        { headers: this.headers }
      );
      
      const tickerData = await tickerResponse.json();
      
      // Find the company by ticker
      for (const key in tickerData) {
        if (tickerData[key].ticker?.toUpperCase() === ticker.toUpperCase()) {
          const cik = String(tickerData[key].cik_str);
          return this.getCompanyInfo(cik);
        }
      }
      
      return null;
    } catch (error) {
      console.error('Ticker lookup failed:', error);
      return null;
    }
  }

  /**
   * Get detailed company information by CIK
   */
  async getCompanyInfo(cik: string): Promise<EdgarCompanyInfo | null> {
    try {
      // Pad CIK to 10 digits (SEC requirement)
      const paddedCik = cik.replace(/^0+/, '').padStart(10, '0');
      
      const response = await fetch(
        `${this.BASE_URL}/submissions/CIK${paddedCik}.json`,
        { headers: this.headers }
      );
      
      if (!response.ok) {
        if (response.status === 404) {
          return null; // Company not found (likely not public)
        }
        throw new Error(`Failed to get company info: ${response.statusText}`);
      }
      
      const data = await response.json();
      
      return {
        cik: data.cik,
        entityType: data.entityType,
        name: data.name,
        sic: data.sic,
        sicDescription: data.sicDescription,
        tickers: data.tickers || [],
        exchanges: data.exchanges || [],
        ein: data.ein,
        stateOfIncorporation: data.stateOfIncorporation,
        fiscalYearEnd: data.fiscalYearEnd,
        formerNames: data.formerNames,
      };
    } catch (error) {
      console.error('Get company info failed:', error);
      return null;
    }
  }

  /**
   * Get all filings for a company
   */
  async getCompanyFilings(cik: string, formTypes?: string[]): Promise<EdgarFiling[]> {
    try {
      const paddedCik = cik.replace(/^0+/, '').padStart(10, '0');
      
      const response = await fetch(
        `${this.BASE_URL}/submissions/CIK${paddedCik}.json`,
        { headers: this.headers }
      );
      
      if (!response.ok) {
        throw new Error(`Failed to get filings: ${response.statusText}`);
      }
      
      const data = await response.json();
      const recent = data.filings?.recent;
      
      if (!recent) return [];
      
      const filings: EdgarFiling[] = [];
      
      for (let i = 0; i < recent.form.length; i++) {
        const form = recent.form[i];
        
        // Filter by form type if specified
        if (formTypes && formTypes.length > 0 && !formTypes.includes(form)) {
          continue;
        }
        
        const accessionNumber = recent.accessionNumber[i];
        const primaryDocument = recent.primaryDocument[i];
        
        filings.push({
          accessionNumber,
          form,
          filingDate: recent.filingDate[i],
          reportDate: recent.reportDate[i],
          acceptanceDateTime: recent.acceptanceDateTime[i],
          primaryDocument,
          primaryDocDescription: recent.primaryDocDescription[i],
          size: recent.size[i],
          documentUrl: this.buildDocumentUrl(data.cik, accessionNumber, primaryDocument),
        });
      }
      
      return filings;
    } catch (error) {
      console.error('Get company filings failed:', error);
      return [];
    }
  }

  /**
   * Build the URL to access a specific filing document
   */
  private buildDocumentUrl(cik: string, accessionNumber: string, document: string): string {
    const accessionFormatted = accessionNumber.replace(/-/g, '');
    return `https://www.sec.gov/Archives/edgar/data/${cik}/${accessionFormatted}/${document}`;
  }

  /**
   * Get structured financial data (XBRL) for a company
   */
  async getCompanyFacts(cik: string): Promise<any> {
    try {
      const paddedCik = cik.replace(/^0+/, '').padStart(10, '0');
      
      const response = await fetch(
        `${this.BASE_URL}/api/xbrl/companyfacts/CIK${paddedCik}.json`,
        { headers: this.headers }
      );
      
      if (!response.ok) {
        return null;
      }
      
      return response.json();
    } catch (error) {
      console.error('Get company facts failed:', error);
      return null;
    }
  }

  /**
   * Extract key financial metrics from XBRL data
   */
  async getFinancialSummary(cik: string): Promise<FinancialSummary> {
    const facts = await this.getCompanyFacts(cik);
    
    if (!facts) {
      return {
        revenue: null,
        netIncome: null,
        totalAssets: null,
        totalLiabilities: null,
        stockholdersEquity: null,
        cashAndEquivalents: null,
        longTermDebt: null,
        earningsPerShare: null,
        periodEnd: null,
      };
    }
    
    const usgaap = facts.facts?.['us-gaap'] || {};
    
    // Helper to get most recent annual (10-K) value
    const getMostRecent = (concepts: string[], unit: string = 'USD'): { value: number | null; period: string | null } => {
      for (const concept of concepts) {
        const values = usgaap[concept]?.units?.[unit];
        if (values?.length) {
          // Get most recent 10-K value
          const annual = values
            .filter((v: any) => v.form === '10-K' && v.frame) // Has fiscal year frame
            .sort((a: any, b: any) => new Date(b.end).getTime() - new Date(a.end).getTime());
          
          if (annual.length > 0) {
            return { value: annual[0].val, period: annual[0].end };
          }
        }
      }
      return { value: null, period: null };
    };

    const revenue = getMostRecent([
      'Revenues',
      'RevenueFromContractWithCustomerExcludingAssessedTax',
      'SalesRevenueNet',
      'RevenueFromContractWithCustomerIncludingAssessedTax',
    ]);

    return {
      revenue: revenue.value,
      netIncome: getMostRecent(['NetIncomeLoss', 'ProfitLoss']).value,
      totalAssets: getMostRecent(['Assets']).value,
      totalLiabilities: getMostRecent(['Liabilities']).value,
      stockholdersEquity: getMostRecent(['StockholdersEquity', 'StockholdersEquityIncludingPortionAttributableToNoncontrollingInterest']).value,
      cashAndEquivalents: getMostRecent(['CashAndCashEquivalentsAtCarryingValue', 'Cash']).value,
      longTermDebt: getMostRecent(['LongTermDebt', 'LongTermDebtNoncurrent']).value,
      earningsPerShare: getMostRecent(['EarningsPerShareBasic'], 'USD/shares').value,
      periodEnd: revenue.period,
    };
  }

  /**
   * Get recent 8-K filings and identify material events
   */
  async getMaterialEvents(cik: string, limit: number = 20): Promise<MaterialEvent[]> {
    const eightKFilings = await this.getCompanyFilings(cik, ['8-K', '8-K/A']);
    
    // 8-K items that indicate significant events requiring review
    const significantItems: Record<string, string> = {

      '2.03': 'Creation of Direct Financial Obligation',
      '2.04': 'Triggering Events That Accelerate Obligation',
      '2.05': 'Costs Associated with Exit Activities',
      '2.06': 'Material Impairments',
      '3.01': 'Notice of Delisting',
      '3.02': 'Unregistered Sales of Equity Securities',
      '3.03': 'Material Modification to Rights of Securities',
      '4.01': 'Changes in Registrant\'s Certifying Accountant',
      '4.02': 'Non-Reliance on Previously Issued Financial Statements',
      '5.01': 'Changes in Control of Registrant',
      '5.02': 'Departure/Election of Directors or Officers',
      '5.03': 'Amendments to Articles of Incorporation or Bylaws',
      '5.05': 'Amendments to Code of Ethics',
      '5.06': 'Change in Shell Company Status',
      '5.07': 'Submission of Matters to Vote of Security Holders',
      '6.01': 'ABS Informational and Computational Material',
      '7.01': 'Regulation FD Disclosure',
      '8.01': 'Other Events',
      '9.01': 'Financial Statements and Exhibits',
    };

    // High-priority items that definitely need review
    const highPriorityItems = ['1.01', '1.02', '1.03', '1.05', '2.01', '2.04', '2.06', '3.01', '4.01', '4.02', '5.01', '5.02'];

    return eightKFilings.slice(0, limit).map(filing => ({
      date: filing.filingDate,
      form: filing.form,
      description: filing.primaryDocDescription || 'Form 8-K',
      url: filing.documentUrl,
      items: [], // Would need to parse the 8-K content to get specific items
      requiresReview: true, // Flag all 8-Ks for review
    }));
  }

  /**
   * Check if a company is publicly traded
   */
  async isPublicCompany(companyName: string): Promise<{ isPublic: boolean; cik?: string; info?: EdgarCompanyInfo }> {
    const searchResults = await this.searchCompany(companyName);
    
    if (searchResults.length === 0) {
      return { isPublic: false };
    }
    
    // Verify by getting company info
    const info = await this.getCompanyInfo(searchResults[0].cik);
    
    if (info) {
      return { isPublic: true, cik: info.cik, info };
    }
    
    return { isPublic: false };
  }

  /**
   * Get all SEC data for background research
   */
  async getFullSecProfile(companyName: string): Promise<SecProfileResult> {
    // Search for the company
    const searchResults = await this.searchCompany(companyName);
    
    if (searchResults.length === 0) {
      return {
        isPublic: false,
        found: false,
      };
    }
    
    const cik = searchResults[0].cik;
    
    // Get all data in parallel
    const [companyInfo, financials, tenKFilings, tenQFilings, eightKFilings] = await Promise.all([
      this.getCompanyInfo(cik),
      this.getFinancialSummary(cik),
      this.getCompanyFilings(cik, ['10-K']),
      this.getCompanyFilings(cik, ['10-Q']),
      this.getCompanyFilings(cik, ['8-K', '8-K/A']),
    ]);

    const materialEvents = await this.getMaterialEvents(cik);

    return {
      isPublic: true,
      found: true,
      cik,
      companyInfo,
      financials,
      filings: {
        annualReports: tenKFilings.slice(0, 5),
        quarterlyReports: tenQFilings.slice(0, 8),
        currentReports: eightKFilings.slice(0, 20),
      },
      materialEvents,
      secUrl: `https://www.sec.gov/cgi-bin/browse-edgar?action=getcompany&CIK=${cik}&type=&dateb=&owner=include&count=40`,
    };
  }
}
```

```typescript
export class UccApiService {
  private provider: string;
  private apiKey: string | null;
  private accountId: string | null;

  constructor() {
    this.provider = process.env.UCC_PROVIDER || 'manual';
    this.apiKey = process.env.UCC_PROVIDER_API_KEY || null;
    this.accountId = process.env.UCC_PROVIDER_ACCOUNT_ID || null;
  }

  /**
   * Search for UCC filings
   */
  async searchUccFilings(
    entityName: string,
    states: string[],
    options: {
      searchAsDebtor?: boolean;
      searchAsSecuredParty?: boolean;
      includeTerminated?: boolean;
    } = {}
  ): Promise<UccSearchResult> {
    const { searchAsDebtor = true, searchAsSecuredParty = false, includeTerminated = false } = options;

    // If we have a commercial provider configured, use it
    if (this.provider !== 'manual' && this.apiKey) {
      return this.searchViaProvider(entityName, states, { searchAsDebtor, searchAsSecuredParty, includeTerminated });
    }

    // Otherwise, return manual search instructions
    return this.getManualSearchInstructions(entityName, states);
  }

  /**
   * Search via commercial UCC provider
   */
  private async searchViaProvider(
    entityName: string,
    states: string[],
    options: { searchAsDebtor: boolean; searchAsSecuredParty: boolean; includeTerminated: boolean }
  ): Promise<UccSearchResult> {
    // Implementation depends on provider
    // This is a template - actual implementation would depend on the specific provider's API
    
    switch (this.provider.toLowerCase()) {
      case 'infotrack':
        return this.searchInfoTrack(entityName, states, options);
      case 'csc':
        return this.searchCSC(entityName, states, options);
      case 'cobalt':
        return this.searchCobalt(entityName, states, options);
      default:
        return this.getManualSearchInstructions(entityName, states);
    }
  }

  /**
   * InfoTrack UCC Search
   * API Documentation: https://www.infotrack.com/api/
   */
  private async searchInfoTrack(
    entityName: string,
    states: string[],
    options: { searchAsDebtor: boolean; searchAsSecuredParty: boolean; includeTerminated: boolean }
  ): Promise<UccSearchResult> {
    const filings: UccFiling[] = [];
    
    try {
      for (const state of states) {
        const response = await fetch('https://api.infotrack.com/v1/ucc/search', {
          method: 'POST',
          headers: {
            'Authorization': `Bearer ${this.apiKey}`,
            'Content-Type': 'application/json',
          },
          body: JSON.stringify({
            searchName: entityName,
            state: state,
            searchType: options.searchAsDebtor ? 'debtor' : 'secured_party',
            includeInactive: options.includeTerminated,
          }),
        });

        if (response.ok) {
          const data = await response.json();
          
          for (const result of data.results || []) {
            filings.push(this.mapInfoTrackResult(result, state));
          }
        }
      }

      return {
        automated: true,
        filings,
        searchedStates: states,
      };
    } catch (error) {
      console.error('InfoTrack search failed:', error);
      return this.getManualSearchInstructions(entityName, states);
    }
  }

  /**
   * Get manual search instructions when no API is available
   */
  private getManualSearchInstructions(entityName: string, states: string[]): UccSearchResult {
    const manualSearchUrls: Record<string, string> = {};
    
    for (const state of states) {
      if (STATE_UCC_PORTALS[state]) {
        manualSearchUrls[state] = STATE_UCC_PORTALS[state];
      }
    }

    // Suggest additional states to search
    const suggestedStates = this.getSuggestedSearchStates(states[0]);
    for (const state of suggestedStates) {
      if (!manualSearchUrls[state] && STATE_UCC_PORTALS[state]) {
        manualSearchUrls[state] = STATE_UCC_PORTALS[state];
      }
    }

    return {
      automated: false,
      filings: [],
      searchedStates: [],
      manualSearchRequired: true,
      manualSearchInstructions: `
Search for "${entityName}" as DEBTOR in the following state UCC databases:

${Object.entries(manualSearchUrls).map(([state, url]) => `• ${state}: ${url}`).join('\n')}

Tips for searching:
1. Search the exact entity name first
2. Then try variations (without Inc., LLC, etc.)
3. Check both organization and individual debtor searches
4. Note the filing numbers, dates, secured parties, and collateral descriptions
5. For "All Assets" filings, this typically indicates significant secured debt
      `.trim(),
      manualSearchUrls,
    };
  }

  /**
   * Suggest additional states to search based on primary state
   */
  private getSuggestedSearchStates(primaryState: string): string[] {
    const suggestions = new Set<string>([primaryState]);
    
    // Always search Delaware (common incorporation state)
    suggestions.add('DE');
    
    // If primary state is a major commercial state, add others
    const majorStates = ['CA', 'NY', 'TX', 'FL', 'IL', 'NV', 'DE'];
    if (majorStates.includes(primaryState)) {
      majorStates.forEach(s => suggestions.add(s));
    }
    
    return Array.from(suggestions);
  }

  /**
   * Map filing type strings to enum
   */
  private mapFilingType(type: string): UccFiling['filingType'] {
    const typeUpper = type?.toUpperCase() || '';
    if (typeUpper.includes('TERMINATION')) return 'UCC3_TERMINATION';
    if (typeUpper.includes('AMENDMENT')) return 'UCC3_AMENDMENT';
    if (typeUpper.includes('CONTINUATION')) return 'UCC3_CONTINUATION';
    if (typeUpper.includes('ASSIGNMENT')) return 'UCC3_ASSIGNMENT';
    if (typeUpper.includes('UCC3') || typeUpper.includes('UCC-3')) return 'UCC3_AMENDMENT';
    return 'UCC1';
  }

  /**
   * Map status strings to enum
   */
  private mapStatus(status: string): UccFiling['status'] {
    const statusLower = status?.toLowerCase() || '';
    if (statusLower.includes('terminated') || statusLower.includes('released')) return 'terminated';
    if (statusLower.includes('lapsed') || statusLower.includes('expired')) return 'lapsed';
    return 'active';
  }

  /**
   * Infer collateral type from description
   */
  private inferCollateralType(description: string): UccFiling['collateralType'] {
    if (!description) return 'other';
    
    const descLower = description.toLowerCase();
    
    if (descLower.includes('all assets') || 
        descLower.includes('all personal property') || 
        descLower.includes('all tangible and intangible')) {
      return 'all_assets';
    }
    if (descLower.includes('equipment') && !descLower.includes('inventory')) {
      return 'equipment';
    }
    if (descLower.includes('inventory')) {
      return 'inventory';
    }
    if (descLower.includes('accounts receivable') || descLower.includes('accounts')) {
      return 'accounts';
    }
    
    return 'specific';
  }

  /**
   * Analyze UCC filings for risk indicators
   */
  analyzeUccFilings(filings: UccFiling[]): UccAnalysis {
    const activeFilings = filings.filter(f => f.status === 'active');
    const allAssetsFilings = activeFilings.filter(f => f.collateralType === 'all_assets');
    
    // Group by secured party
    const bySecuredParty = new Map<string, UccFiling[]>();
    for (const filing of activeFilings) {
      const key = filing.securedPartyName.toLowerCase();
      if (!bySecuredParty.has(key)) {
        bySecuredParty.set(key, []);
      }
      bySecuredParty.get(key)!.push(filing);
    }

    // Identify known lender types
    const lenderTypes: string[] = [];
    for (const [name] of bySecuredParty) {
      if (name.includes('bank') || name.includes('capital') || name.includes('credit')) {
        lenderTypes.push('Traditional Lender');
      } else if (name.includes('factor') || name.includes('funding')) {
        lenderTypes.push('Factor/ABL');
      } else if (name.includes('leasing') || name.includes('lease')) {
        lenderTypes.push('Equipment Lessor');
      }
    }

    return {
      totalFilings: filings.length,
      activeFilings: activeFilings.length,
      lapsedFilings: filings.filter(f => f.status === 'lapsed').length,
      terminatedFilings: filings.filter(f => f.status === 'terminated').length,
      
      hasAllAssetsLien: allAssetsFilings.length > 0,
      allAssetsLienCount: allAssetsFilings.length,
      
      uniqueSecuredParties: bySecuredParty.size,
      securedParties: Array.from(bySecuredParty.keys()),
      lenderTypes: [...new Set(lenderTypes)],
      
      riskLevel: this.assessUccRisk(activeFilings, allAssetsFilings),
      riskNotes: this.generateUccRiskNotes(activeFilings, allAssetsFilings, bySecuredParty),
    };
  }

  private assessUccRisk(activeFilings: UccFiling[], allAssetsFilings: UccFiling[]): 'low' | 'medium' | 'high' {
    if (allAssetsFilings.length > 1) return 'high';
    if (allAssetsFilings.length === 1) return 'medium';
    if (activeFilings.length > 5) return 'medium';
    return 'low';
  }

  private generateUccRiskNotes(
    activeFilings: UccFiling[],
    allAssetsFilings: UccFiling[],
    bySecuredParty: Map<string, UccFiling[]>
  ): string[] {
    const notes: string[] = [];
    
    if (allAssetsFilings.length > 0) {
      notes.push(`${allAssetsFilings.length} "All Assets" filing(s) found - indicates significant secured debt that would take priority in any asset purchase`);
    }
    
    if (bySecuredParty.size > 3) {
      notes.push(`Multiple secured creditors (${bySecuredParty.size}) - may indicate complex capital structure`);
    }
    
    // Check for recent filings (within 90 days)
    const ninetyDaysAgo = new Date();
    ninetyDaysAgo.setDate(ninetyDaysAgo.getDate() - 90);
    const recentFilings = activeFilings.filter(f => new Date(f.filingDate) > ninetyDaysAgo);
    
    if (recentFilings.length > 0) {
      notes.push(`${recentFilings.length} filing(s) within last 90 days - recent secured financing activity`);
    }
    
    return notes;
  }
}

  hasAllAssetsLien: boolean;
  allAssetsLienCount: number;
  uniqueSecuredParties: number;
  securedParties: string[];
```

```typescript
  /**
   * Build URL with optional API key
   */
  private buildUrl(path: string, params: Record<string, string> = {}): string {
    const url = new URL(`${this.BASE_URL}${path}`);
    
    if (this.apiKey) {
      url.searchParams.set('api_token', this.apiKey);
    }
    
    for (const [key, value] of Object.entries(params)) {
      url.searchParams.set(key, value);
    }
    
    return url.toString();
  }

  async searchCompanies(

      companyType?: string;

      perPage?: number;

      q: name,
      ...(options.jurisdiction && { jurisdiction_code: options.jurisdiction }),

      ...(options.status && { status: options.status }),
      ...(options.companyType && { company_type: options.companyType }),

      const response = await fetch(this.buildUrl('/companies/search', params));
      
      if (!response.ok) {
        throw new Error(`OpenCorporates search failed: ${response.statusText}`);
      }
      
      const data = await response.json();

        companies: data.results?.companies?.map((c: any) => this.mapCompany(c.company)) || [],
        totalCount: data.total_count || 0,
        page: data.page || 1,
        perPage: data.per_page || 30,
      };
    } catch (error) {

      company: {
        ...bestMatch,
        officers,
        filings,
      },
      alternativeMatches: searchResults.companies.slice(1, 5),
    };
```

```typescript
// services/background-research.service.ts

import { SecEdgarApiService, SecProfileResult } from './sec-edgar-api.service';
import { UccApiService, UccSearchResult, UccAnalysis } from './ucc-api.service';
import { OpenCorporatesApiService, CorporateProfileResult } from './opencorporates-api.service';
import { OpenAI } from 'openai';
import { db } from '@/db';
import { eq } from 'drizzle-orm';
import {
  backgroundResearch,
  researchSecFilings,
  researchCorporateStructure,
  researchLeadership,
  researchLitigation,
  researchUccFilings,
  researchUccSearchTasks,
  researchRegulatoryActions,
  researchNewsMedia,
  researchFinancialIndicators,
  researchRedFlags,
} from '@/db/schema';

export interface ResearchScope {
  corporate: boolean;
  leadership: boolean;
  litigation: boolean;
  ucc: boolean;
  regulatory: boolean;
  news: boolean;
  financial: boolean;
  sec: boolean;
}

export class BackgroundResearchService {
  private secApi: SecEdgarApiService;
  private uccApi: UccApiService;
  private openCorpApi: OpenCorporatesApiService;
  private openai: OpenAI;

  constructor() {
    this.secApi = new SecEdgarApiService();
    this.uccApi = new UccApiService();
    this.openCorpApi = new OpenCorporatesApiService();
    this.openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
  }

  /**
   * Run comprehensive background research
   */
  async runResearch(researchId: number): Promise<void> {
    const research = await db.query.backgroundResearch.findFirst({
      where: eq(backgroundResearch.id, researchId),
    });

    if (!research) throw new Error('Research not found');

    // Update status
    await db.update(backgroundResearch)
      .set({ status: 'in_progress', startedAt: new Date() })
      .where(eq(backgroundResearch.id, researchId));

    try {
      const scope = research.researchScope as ResearchScope;

      // Run research in parallel where possible
      const results = await Promise.allSettled([
        // API-based research
        scope.sec && this.searchSecFilings(research),
        scope.corporate && this.searchCorporateStructure(research),
        scope.ucc && this.searchUccFilings(research),
        
        // Web search-based research
        scope.leadership && this.searchLeadership(research),
        scope.litigation && this.searchLitigation(research),
        scope.regulatory && this.searchRegulatoryActions(research),
        scope.news && this.searchNewsMedia(research),
        scope.financial && this.enrichFinancialIndicators(research),
      ]);

      // Log any errors
      results.forEach((result, index) => {
        if (result.status === 'rejected') {
          console.error(`Research task ${index} failed:`, result.reason);
        }
      });

      // Analyze all findings and generate red flags
      await this.analyzeAndGenerateRedFlags(researchId);

      // Generate executive summary
      await this.generateExecutiveSummary(researchId);

      // Calculate risk score
      await this.calculateRiskScore(researchId);

      // Update status to completed
      await db.update(backgroundResearch)
        .set({ status: 'completed', completedAt: new Date() })
        .where(eq(backgroundResearch.id, researchId));

    } catch (error) {
      console.error('Research failed:', error);
      await db.update(backgroundResearch)
        .set({ status: 'failed' })
        .where(eq(backgroundResearch.id, researchId));
      throw error;
    }
  }

  /**
   * Search SEC EDGAR for public company filings
   */
  private async searchSecFilings(research: any): Promise<void> {
    console.log(`Searching SEC EDGAR for: ${research.subjectName}`);
    
    const secProfile = await this.secApi.getFullSecProfile(research.subjectName);
    
    // Store SEC findings
    await db.insert(researchSecFilings).values({
      researchId: research.id,
      isPublicCompany: secProfile.isPublic,
      cik: secProfile.cik,
      ticker: secProfile.companyInfo?.tickers?.[0],
      exchange: secProfile.companyInfo?.exchanges?.[0],
      secCompanyName: secProfile.companyInfo?.name,
      sic: secProfile.companyInfo?.sic,
      sicDescription: secProfile.companyInfo?.sicDescription,
      stateOfIncorporation: secProfile.companyInfo?.stateOfIncorporation,
      ein: secProfile.companyInfo?.ein,
      fiscalYearEnd: secProfile.companyInfo?.fiscalYearEnd,
      latestRevenue: secProfile.financials?.revenue,
      latestNetIncome: secProfile.financials?.netIncome,
      totalAssets: secProfile.financials?.totalAssets,
      totalLiabilities: secProfile.financials?.totalLiabilities,
      stockholdersEquity: secProfile.financials?.stockholdersEquity,
      cashAndEquivalents: secProfile.financials?.cashAndEquivalents,
      longTermDebt: secProfile.financials?.longTermDebt,
      financialPeriodEnd: secProfile.financials?.periodEnd,
      recentFilings: secProfile.filings,
      materialEvents: secProfile.materialEvents,
      secProfileUrl: secProfile.secUrl,
    });

    // Generate red flags for SEC findings
    if (secProfile.isPublic) {
      // Check for concerning 8-K events
      if (secProfile.materialEvents && secProfile.materialEvents.length > 0) {
        await db.insert(researchRedFlags).values({
          researchId: research.id,
          category: 'sec',
          severity: 'medium',
          title: `${secProfile.materialEvents.length} Recent 8-K Filing(s) Require Review`,
          description: `The company has filed ${secProfile.materialEvents.length} Form 8-K current reports in recent months. These filings may disclose material events that could affect the transaction.`,
          evidence: JSON.stringify(secProfile.materialEvents.slice(0, 5)),
          recommendation: 'Review each 8-K filing to understand the material events disclosed.',
        });
      }

      // Check financial health
      if (secProfile.financials) {
        const { netIncome, totalLiabilities, totalAssets, revenue } = secProfile.financials;
        
        // Net loss
        if (netIncome && netIncome < 0) {
          await db.insert(researchRedFlags).values({
            researchId: research.id,
            category: 'financial',
            severity: Math.abs(netIncome) > 10000000 ? 'high' : 'medium',
            title: 'Company Reported Net Loss',
            description: `The company reported a net loss of $${Math.abs(netIncome).toLocaleString()} in the most recent fiscal year.`,
            recommendation: 'Review the company\'s financial statements and management discussion for explanations.',
          });
        }
        
        // High debt ratio
        if (totalLiabilities && totalAssets && totalLiabilities / totalAssets > 0.8) {
          await db.insert(researchRedFlags).values({
            researchId: research.id,
            category: 'financial',
            severity: 'medium',
            title: 'High Debt-to-Asset Ratio',
            description: `The company has a debt-to-asset ratio of ${((totalLiabilities / totalAssets) * 100).toFixed(1)}%, indicating significant leverage.`,
            recommendation: 'Review debt covenants and ability to service debt obligations.',
          });
        }
      }
    }
  }

  /**
   * Search for corporate structure using OpenCorporates
   */
  private async searchCorporateStructure(research: any): Promise<void> {
    console.log(`Searching corporate records for: ${research.subjectName}`);
    
    const corpProfile = await this.openCorpApi.getFullCorporateProfile(
      research.subjectName,
      research.stateOfFormation
    );

    if (corpProfile.found && corpProfile.company) {
      const company = corpProfile.company;
      
      // Store corporate structure
      await db.insert(researchCorporateStructure).values({
        researchId: research.id,
        legalName: company.name,
        entityType: company.companyType,
        formationState: company.jurisdictionCode?.replace('us_', '').toUpperCase(),
        formationDate: company.incorporationDate,
        status: company.status,
        principalAddress: company.registeredAddress,
        fileNumber: company.companyNumber,
        sourceUrls: [company.registryUrl],
        sourceType: 'opencorporates',
      });

      // Store officers/leadership
      if (company.officers) {
        for (const officer of company.officers) {
          await db.insert(researchLeadership).values({
            researchId: research.id,
            personName: officer.name,
            title: officer.position,
            role: this.mapOfficerRole(officer.position),
            startDate: officer.startDate,
            endDate: officer.endDate,
            isCurrent: officer.status === 'active',
            sourceType: 'opencorporates',
          });
        }
      }

      // Check for inactive status
      if (company.status && !['active', 'good standing'].includes(company.status.toLowerCase())) {
        await db.insert(researchRedFlags).values({
          researchId: research.id,
          category: 'corporate',
          severity: 'high',
          title: `Entity Status: ${company.status}`,
          description: `The company's status is listed as "${company.status}" in corporate records. This may indicate dissolution, suspension, or other issues.`,
          recommendation: 'Verify the entity\'s current standing with the Secretary of State before proceeding.',
        });
      }
    }
  }

  /**
   * Search UCC filings
   */
  private async searchUccFilings(research: any): Promise<void> {
    console.log(`Searching UCC filings for: ${research.subjectName}`);
    
    // Determine which states to search
    const statesToSearch = this.getStatesToSearch(
      research.stateOfFormation,
      research.state
    );

    const uccResult = await this.uccApi.searchUccFilings(
      research.subjectName,
      statesToSearch,
      { searchAsDebtor: true, searchAsSecuredParty: false }
    );

    if (uccResult.automated && uccResult.filings.length > 0) {
      // Store UCC filings
      for (const filing of uccResult.filings) {
        await db.insert(researchUccFilings).values({
          researchId: research.id,
          filingNumber: filing.filingNumber,
          filingType: filing.filingType.toLowerCase(),
          filingState: filing.filingState,
          filingDate: filing.filingDate,
          lapseDate: filing.lapseDate,
          status: filing.status,
          subjectRole: 'debtor',
          securedPartyName: filing.securedPartyName,
          securedPartyAddress: filing.securedPartyAddress,
          debtorName: filing.debtorName,
          debtorAddress: filing.debtorAddress,
          collateralDescription: filing.collateralDescription,
          collateralType: filing.collateralType,
          sourceUrl: filing.sourceUrl,
          sourceType: process.env.UCC_PROVIDER || 'automated',
        });
      }

      // Analyze UCC profile
      const analysis = this.uccApi.analyzeUccFilings(uccResult.filings);
      
      // Generate red flags based on analysis
      if (analysis.hasAllAssetsLien) {
        await db.insert(researchRedFlags).values({
          researchId: research.id,
          category: 'ucc',
          severity: analysis.allAssetsLienCount > 1 ? 'high' : 'medium',
          title: `"All Assets" UCC Filing(s) Found`,
          description: `${analysis.allAssetsLienCount} UCC filing(s) covering "all assets" were found. This indicates significant secured debt that would take priority over unsecured creditors in any asset purchase or bankruptcy.`,
          evidence: JSON.stringify({
            totalActive: analysis.activeFilings,
            allAssetsCount: analysis.allAssetsLienCount,
            securedParties: analysis.securedParties,
          }),
          recommendation: 'Obtain payoff letters from secured creditors and understand priority of liens before closing.',
        });
      }

      for (const note of analysis.riskNotes) {
        await db.insert(researchRedFlags).values({
          researchId: research.id,
          category: 'ucc',
          severity: 'low',
          title: 'UCC Filing Note',
          description: note,
        });
      }
    } else if (uccResult.manualSearchRequired) {
      // Create manual search tasks
      for (const [state, url] of Object.entries(uccResult.manualSearchUrls || {})) {
        await db.insert(researchUccSearchTasks).values({
          researchId: research.id,
          state,
          searchUrl: url,
          searchInstructions: `Search for "${research.subjectName}" as debtor in ${state} UCC records`,
          status: 'pending',
        });
      }
    }
  }

  private async searchLitigation(research: any): Promise<void> {
    console.log(`Searching litigation for: ${research.subjectName}`);
    
    // Use AI-powered web search to find litigation

  /**
   * Calculate overall risk score
   */
  private async calculateRiskScore(researchId: number): Promise<void> {
    const [litigation, regulatory, redFlags, liens, uccFilings] = await Promise.all([
      db.query.researchLitigation.findMany({ where: eq(researchLitigation.researchId, researchId) }),
      db.query.researchRegulatoryActions.findMany({ where: eq(researchRegulatoryActions.researchId, researchId) }),
      db.query.researchRedFlags.findMany({ where: eq(researchRedFlags.researchId, researchId) }),
      db.query.researchLiens.findMany({ where: eq(researchLiens.researchId, researchId) }),
      db.query.researchUccFilings.findMany({ where: eq(researchUccFilings.researchId, researchId) }),
    ]);

    let score = 0;

    // Red flags scoring
    score += redFlags.filter(f => f.severity === 'critical').length * 30;
    score += redFlags.filter(f => f.severity === 'high').length * 20;
    score += redFlags.filter(f => f.severity === 'medium').length * 10;
    score += redFlags.filter(f => f.severity === 'low').length * 3;

    // Litigation scoring
    score += litigation.filter(l => l.status === 'pending' && l.subjectRole === 'defendant').length * 15;
    score += litigation.filter(l => l.subjectRole === 'defendant').length * 5;

    // Regulatory scoring
    score += regulatory.filter(r => r.status === 'pending').length * 25;
    score += regulatory.filter(r => r.penaltyAmount && Number(r.penaltyAmount) > 100000).length * 15;

    // Lien scoring
    score += liens.filter(l => l.lienType === 'federal_tax' && l.status === 'active').length * 30;
    score += liens.filter(l => l.lienType === 'state_tax' && l.status === 'active').length * 20;
    score += liens.filter(l => l.lienType === 'judgment' && l.status === 'active').length * 15;

    // UCC scoring
    const activeUcc = uccFilings.filter(u => u.status === 'active');
    const allAssetsUcc = activeUcc.filter(u => u.collateralType === 'all_assets');
    score += allAssetsUcc.length * 10;
    if (activeUcc.length > 5) score += 10;

    // Cap at 100
    score = Math.min(score, 100);

    // Determine risk level
    let riskLevel: string;
    let recommendation: string;
    
    if (score <= 20) {
      riskLevel = 'low';
      recommendation = 'proceed';
    } else if (score <= 40) {
      riskLevel = 'medium';
      recommendation = 'proceed_with_caution';
    } else if (score <= 70) {
      riskLevel = 'high';
      recommendation = 'additional_diligence';
    } else {
      riskLevel = 'critical';
      recommendation = 'do_not_proceed';
    }

    await db.update(backgroundResearch)
      .set({
        riskScore: score,
        overallRiskLevel: riskLevel,
        recommendation,
        redFlagsCount: redFlags.length,
      })
      .where(eq(backgroundResearch.id, researchId));
  }

  private getStatesToSearch(stateOfFormation?: string, operatingState?: string): string[] {
    const states = new Set<string>();
    
    // Always search Delaware
    states.add('DE');
    
    if (stateOfFormation) states.add(stateOfFormation);
    if (operatingState) states.add(operatingState);

    const majorStates = ['CA', 'NY', 'TX', 'FL', 'IL'];

      majorStates.forEach(s => states.add(s));
    }
    
    return Array.from(states);
  }

    if (posLower.includes('secretary')) return 'secretary';
    if (posLower.includes('treasurer')) return 'treasurer';
    if (posLower.includes('agent')) return 'registered_agent';

  }
```
