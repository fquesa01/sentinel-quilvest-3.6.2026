// Type badge for allegations

          </div>

        {/* Right: Focus Issues + Results */}
        <div className="w-[440px] flex flex-col bg-slate-850">
          {/* Header */}
          <div className="h-11 px-3 flex items-center justify-between border-b border-slate-700 bg-slate-800 shrink-0">
            <div className="flex items-center gap-2">
              <Target className="w-4 h-4 text-amber-400" />
              <span className="text-sm font-medium text-slate-200">Focus Issues</span>
              <span className="px-1.5 py-0.5 bg-slate-700 rounded text-[10px] text-slate-400">
                {activeAllegations.length} active
              </span>
            </div>
          </div>

          {/* Results grouped by allegation */}
          <div className="flex-1 overflow-y-auto p-2 space-y-2">
            {activeAllegations.map(allegation => {
              const results = getResultsForAllegation(allegation.id);
              const isExpanded = expandedAllegations.includes(allegation.id);
              const supports = results.filter(r => r.position === 'supports').length;
              const hurts = results.filter(r => r.position === 'hurts').length;
              
              return (
                <div key={allegation.id} className="rounded-lg border border-slate-700/50 overflow-hidden">
                  {/* Allegation Header */}
                  <button
                    onClick={() => toggleExpand(allegation.id)}
                    className="w-full px-3 py-2 flex items-center gap-2 text-left bg-slate-800/50 hover:bg-slate-800 transition-colors"
                  >
                    {isExpanded ? (
                      <ChevronDown className="w-4 h-4 text-slate-400 shrink-0" />
                    ) : (
                      <ChevronRight className="w-4 h-4 text-slate-400 shrink-0" />
                    )}
                    <span className="text-[10px] font-mono text-slate-500">{allegation.paragraph}</span>
                    <span className="flex-1 text-xs font-medium text-slate-200 truncate">
                      {allegation.text.slice(0, 50)}...
                    </span>
                    <div className="flex items-center gap-2 shrink-0">
                      {supports > 0 && (
                        <span className="flex items-center gap-0.5 text-[10px] text-green-400">
                          <ThumbsUp className="w-3 h-3" />{supports}
                        </span>
                      )}
                      {hurts > 0 && (
                        <span className="flex items-center gap-0.5 text-[10px] text-red-400">
                          <ThumbsDown className="w-3 h-3" />{hurts}
                        </span>
                      )}
                    </div>
                  </button>
                  
                  {/* Results */}
                  {isExpanded && (
                    <div className="border-t border-slate-700/30 p-2 space-y-2">
                      {results.length > 0 ? (
                        results.map(result => (
                          <div 
                            key={result.id}
                            className={`p-2.5 rounded-lg border cursor-pointer transition-all hover:border-blue-500/50 ${
                              result.position === 'hurts' 
                                ? 'bg-red-500/5 border-red-500/20' 
                                : 'bg-green-500/5 border-green-500/20'
                            }`}
                          >
                            <div className="flex items-start justify-between gap-2 mb-1.5">
                              <div className="flex-1 min-w-0">
                                <div className="flex items-center gap-2 mb-0.5">
                                  <span className="text-xs font-medium text-slate-200">{result.source}</span>
                                </div>
                                <span className="text-[10px] text-slate-500">{result.time}</span>
                              </div>
                              <PositionBadge position={result.position} />
                            </div>
                            <p className="text-xs text-slate-300 mb-2">{result.text}</p>
                            <div className="flex items-start gap-1.5">
                              <ArrowRight className={`w-3 h-3 mt-0.5 shrink-0 ${
                                result.position === 'hurts' ? 'text-red-400' : 'text-green-400'
                              }`} />
                              <span className={`text-[10px] ${
                                result.position === 'hurts' ? 'text-red-400' : 'text-green-400'
                              }`}>
                                {result.explanation}
                              </span>
                            </div>
                          </div>
                        ))
                      ) : (
                        <div className="py-4 text-center text-xs text-slate-500">
                          No evidence found yet for this allegation
                        </div>
                      )}
                    </div>
                  )}
                </div>
              );
            })}
          </div>

          {/* Footer Status */}
          <div className="px-3 py-2 border-t border-slate-700/50 bg-slate-800/50">
            <div className="flex items-center justify-between text-[10px]">
              <div className="flex items-center gap-2 text-slate-400">
                <div className="w-2 h-2 bg-green-500 rounded-full animate-pulse" />
                <span>Monitoring {activeAllegations.length} allegations</span>
              </div>
              <span className="text-slate-500">Last analysis: 2s ago</span>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}
