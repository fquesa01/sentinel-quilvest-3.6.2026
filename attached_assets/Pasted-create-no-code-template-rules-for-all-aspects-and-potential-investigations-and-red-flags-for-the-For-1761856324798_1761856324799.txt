rules:
  - code: FAST_PAY
    category: Payments
    description: "Invoice paid earlier than contractual terms, suggesting undue influence or favoritism."
    trigger: "Payment date is >50% earlier than vendor payment terms or >30 days faster than historical average."
    examples: ["Invoice paid in 2 days when terms are Net 60"]
    score: 25
    investigation: "Flag for 'unusual payment speed' review. Auto-create case for Accounts Payable audit."

  - code: ROUND_DOLLAR
    category: Payments
    description: "Round-dollar or threshold-hugging amounts may indicate split invoicing or structuring."
    trigger: "Amount divisible by 100 or just below approval limit."
    examples: ["$9,950 where approval limit = $10,000"]
    score: 20
    investigation: "Check for invoice splitting or approval circumvention."

  - code: DUPLICATE_PAYMENT
    category: Payments
    description: "Duplicate invoices or payments to same vendor within short interval."
    trigger: "Same vendor, same amount, same invoice number within 7 days."
    score: 40
    investigation: "Review for duplicate payment fraud or kickbacks."

  - code: OFFSHORE_PAYMENT
    category: Payments
    description: "Payments routed through offshore or sanctioned jurisdictions."
    trigger: "Beneficiary bank located in high-risk/offshore jurisdiction list."
    examples: ["Wire to Cayman Islands, Cyprus, or Seychelles"]
    score: 50
    investigation: "Verify beneficial owner and business justification."
