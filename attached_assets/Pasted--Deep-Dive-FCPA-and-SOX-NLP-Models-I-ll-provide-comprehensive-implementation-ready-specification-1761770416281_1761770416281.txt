## II. ADVANCED FCPA NLP MODEL ARCHITECTURE

### A. Multi-Stage Detection Pipeline

```python
import torch
import torch.nn as nn
from transformers import AutoModel, AutoTokenizer
import numpy as np

class AdvancedFCPAModel(nn.Module):
    """
    Multi-stage FCPA detection with interpretable sub-components
    
    Architecture:
    1. Context Encoder (DeBERTa-v3)
    2. Foreign Official Detector
    3. Payment Intent Classifier
    4. Corruption Intent Analyzer
    5. Third-Party Risk Assessor
    6. Accounting Violation Detector
    7. Final Risk Aggregator
    """
    
    def __init__(self, config):
        super(AdvancedFCPAModel, self).__init__()
        
        # Base encoder
        self.encoder = AutoModel.from_pretrained("microsoft/deberta-v3-large")
        hidden_size = self.encoder.config.hidden_size
        
        # Stage 1: Foreign Official Detection
        self.foreign_official_detector = ForeignOfficialDetector(hidden_size)
        
        # Stage 2: Payment Intent Classification
        self.payment_intent_classifier = PaymentIntentClassifier(hidden_size)
        
        # Stage 3: Corruption Intent Analysis
        self.corruption_intent_analyzer = CorruptionIntentAnalyzer(hidden_size)
        
        # Stage 4: Third-Party Risk Assessment
        self.third_party_assessor = ThirdPartyRiskAssessor(hidden_size)
        
        # Stage 5: Accounting Violation Detection
        self.accounting_detector = AccountingViolationDetector(hidden_size)
        
        # Stage 6: Final Risk Aggregation
        self.risk_aggregator = FCPARiskAggregator(hidden_size)
        
    def forward(self, input_ids, attention_mask, entity_features, metadata):
        """
        Multi-stage forward pass
        
        Args:
            input_ids: Tokenized text
            attention_mask: Attention mask
            entity_features: Dict containing NER results
            metadata: Communication metadata (sender, recipients, etc.)
        """
        # Encode text
        encoder_outputs = self.encoder(
            input_ids=input_ids,
            attention_mask=attention_mask,
            output_hidden_states=True
        )
        
        sequence_output = encoder_outputs.last_hidden_state  # [batch, seq_len, hidden]
        pooled_output = encoder_outputs.pooler_output        # [batch, hidden]
        hidden_states = encoder_outputs.hidden_states        # All layers
        
        # Stage 1: Detect foreign official involvement
        foreign_official_output = self.foreign_official_detector(
            sequence_output=sequence_output,
            pooled_output=pooled_output,
            entity_features=entity_features
        )
        
        # Stage 2: Classify payment intent
        payment_intent_output = self.payment_intent_classifier(
            sequence_output=sequence_output,
            pooled_output=pooled_output,
            foreign_official_context=foreign_official_output
        )
        
        # Stage 3: Analyze corruption intent
        corruption_intent_output = self.corruption_intent_analyzer(
            sequence_output=sequence_output,
            pooled_output=pooled_output,
            payment_context=payment_intent_output
        )
        
        # Stage 4: Assess third-party risk
        third_party_output = self.third_party_assessor(
            pooled_output=pooled_output,
            entity_features=entity_features,
            metadata=metadata
        )
        
        # Stage 5: Detect accounting violations
        accounting_output = self.accounting_detector(
            sequence_output=sequence_output,
            pooled_output=pooled_output
        )
        
        # Stage 6: Aggregate all signals into final risk assessment
        final_risk = self.risk_aggregator(
            foreign_official=foreign_official_output,
            payment_intent=payment_intent_output,
            corruption_intent=corruption_intent_output,
            third_party=third_party_output,
            accounting=accounting_output,
            pooled_output=pooled_output
        )
        
        return {
            'final_risk_score': final_risk['risk_score'],
            'violation_type': final_risk['violation_type'],
            'confidence': final_risk['confidence'],
            
            # Interpretable sub-scores
            'foreign_official_score': foreign_official_output['score'],
            'foreign_official_detected': foreign_official_output['detected'],
            'payment_intent_score': payment_intent_output['score'],
            'corruption_intent_score': corruption_intent_output['score'],
            'third_party_risk_score': third_party_output['score'],
            'accounting_violation_score': accounting_output['score'],
            
            # Attention weights for explainability
            'attention_weights': final_risk['attention_weights'],
            'key_phrases': final_risk['key_phrases']
        }


class ForeignOfficialDetector(nn.Module):
    """
    Specialized sub-module to detect foreign official involvement
    
    This is CRITICAL for FCPA - must accurately identify foreign officials
    """
    
    def __init__(self, hidden_size):
        super(ForeignOfficialDetector, self).__init__()
        
        # Binary classifier: Is foreign official mentioned?
        self.binary_classifier = nn.Sequential(
            nn.Linear(hidden_size, 512),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(512, 2)  # Yes/No
        )
        
        # Confidence scorer
        self.confidence_head = nn.Sequential(
            nn.Linear(hidden_size, 256),
            nn.ReLU(),
            nn.Linear(256, 1),
            nn.Sigmoid()
        )
        
        # Official type classifier
        self.official_type = nn.Sequential(
            nn.Linear(hidden_size, 256),
            nn.ReLU(),
            nn.Linear(256, 7)  # Types: government, SOE, party, candidate, etc.
        )
        
        # Span detector (which tokens refer to the official?)
        self.span_start = nn.Linear(hidden_size, 1)
        self.span_end = nn.Linear(hidden_size, 1)
        
    def forward(self, sequence_output, pooled_output, entity_features):
        """
        Detect and characterize foreign official involvement
        """
        # Binary detection
        binary_logits = self.binary_classifier(pooled_output)
        detection_prob = torch.softmax(binary_logits, dim=-1)[:, 1]  # Prob of "Yes"
        
        # Confidence
        confidence = self.confidence_head(pooled_output)
        
        # Official type
        type_logits = self.official_type(pooled_output)
        
        # Span detection (where in text is the official mentioned?)
        start_logits = self.span_start(sequence_output).squeeze(-1)
        end_logits = self.span_end(sequence_output).squeeze(-1)
        
        # Extract entity features (from NER)
        has_foreign_official_entity = (
            entity_features.get('num_foreign_officials', 0) > 0
        )
        
        # Combine model prediction with entity detection
        # If NER found foreign official entity, boost detection confidence
        if has_foreign_official_entity:
            detection_prob = torch.maximum(
                detection_prob,
                torch.tensor([0.7], device=detection_prob.device)
            )
        
        return {
            'score': detection_prob,
            'detected': detection_prob > 0.5,
            'confidence': confidence,
            'official_type_logits': type_logits,
            'span_start_logits': start_logits,
            'span_end_logits': end_logits,
            'entities': entity_features.get('FOREIGN_OFFICIAL', [])
        }


class PaymentIntentClassifier(nn.Module):
    """
    Classifies the intent of any payment discussion
    
    Categories:
    - No payment discussed
    - Legitimate business payment (invoice, salary, etc.)
    - Gift/hospitality (may be legitimate if reasonable)
    - Suspicious payment (potential bribe)
    - Clear bribery language
    """
    
    def __init__(self, hidden_size):
        super(PaymentIntentClassifier, self).__init__()
        
        # Multi-class classifier
        self.classifier = nn.Sequential(
            nn.Linear(hidden_size * 2, 512),  # *2 for concatenation
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(512, 256),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(256, 5)  # 5 payment intent categories
        )
        
        # Payment amount extractor
        self.amount_attention = nn.MultiheadAttention(
            embed_dim=hidden_size,
            num_heads=8
        )
        
    def forward(self, sequence_output, pooled_output, foreign_official_context):
        """
        Classify payment intent, considering foreign official context
        """
        # Apply attention to focus on payment-related tokens
        attended_output, _ = self.amount_attention(
            sequence_output.transpose(0, 1),
            sequence_output.transpose(0, 1),
            sequence_output.transpose(0, 1)
        )
        attended_pooled = attended_output.mean(dim=0)
        
        # Concatenate with foreign official context
        foreign_official_score = foreign_official_context['score'].unsqueeze(1)
        combined = torch.cat([pooled_output, attended_pooled], dim=1)
        
        # Classify
        intent_logits = self.classifier(combined)
        
        # Calculate risk score based on intent + foreign official presence
        intent_probs = torch.softmax(intent_logits, dim=-1)
        
        # Risk multiplier if foreign official present
        risk_multiplier = 1.0 + (2.0 * foreign_official_score.squeeze())
        
        # Weighted risk score
        risk_weights = torch.tensor(
            [0.0, 0.2, 0.5, 0.8, 1.0],  # Weights for each intent category
            device=intent_logits.device
        )
        base_risk = torch.sum(intent_probs * risk_weights, dim=1)
        adjusted_risk = torch.clamp(base_risk * risk_multiplier, 0.0, 1.0)
        
        return {
            'score': adjusted_risk,
            'intent_logits': intent_logits,
            'intent_probs': intent_probs,
            'categories': [
                'no_payment',
                'legitimate_business',
                'gift_hospitality',
                'suspicious',
                'clear_bribery'
            ]
        }


class CorruptionIntentAnalyzer(nn.Module):
    """
    Analyzes whether payment intent is "corrupt" per FCPA definition
    
    FCPA requires "corrupt" intent - willfully giving something of value
    to improperly influence official action
    """
    
    def __init__(self, hidden_size):
        super(CorruptionIntentAnalyzer, self).__init__()
        
        # Corruption intent signals
        self.intent_encoder = nn.Sequential(
            nn.Linear(hidden_size, 512),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(512, 256)
        )
        
        # Quid pro quo detector (payment in exchange for action)
        self.quid_pro_quo_detector = nn.Sequential(
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, 1),
            nn.Sigmoid()
        )
        
        # Business purpose detector (is there legitimate business justification?)
        self.business_purpose_detector = nn.Sequential(
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, 1),
            nn.Sigmoid()
        )
        
        # Willful blindness detector (conscious avoidance of knowledge)
        self.willful_blindness_detector = nn.Sequential(
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, 1),
            nn.Sigmoid()
        )
        
    def forward(self, sequence_output, pooled_output, payment_context):
        """
        Analyze corruption intent
        """
        # Encode
        intent_features = self.intent_encoder(pooled_output)
        
        # Detect quid pro quo
        quid_pro_quo_score = self.quid_pro_quo_detector(intent_features)
        
        # Detect legitimate business purpose
        business_purpose_score = self.business_purpose_detector(intent_features)
        
        # Detect willful blindness
        willful_blindness_score = self.willful_blindness_detector(intent_features)
        
        # Aggregate corruption intent
        # High corruption if: high quid pro quo AND low business purpose
        # OR high willful blindness
        corruption_score = torch.maximum(
            quid_pro_quo_score * (1 - business_purpose_score),
            willful_blindness_score
        )
        
        return {
            'score': corruption_score,
            'quid_pro_quo': quid_pro_quo_score,
            'business_purpose': business_purpose_score,
            'willful_blindness': willful_blindness_score
        }


class ThirdPartyRiskAssessor(nn.Module):
    """
    Assesses risk of third-party intermediary bribery scheme
    
    Many FCPA violations occur through consultants, agents, distributors
    """
    
    def __init__(self, hidden_size):
        super(ThirdPartyRiskAssessor, self).__init__()
        
        # Feature combiner
        self.feature_combiner = nn.Sequential(
            nn.Linear(hidden_size + 64, 256),  # +64 for metadata features
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, 128)
        )
        
        # Red flag detectors
        self.excessive_fee_detector = nn.Linear(128, 1)
        self.relationship_detector = nn.Linear(128, 1)
        self.qualification_detector = nn.Linear(128, 1)
        self.transparency_detector = nn.Linear(128, 1)
        
        self.sigmoid = nn.Sigmoid()
        
    def forward(self, pooled_output, entity_features, metadata):
        """
        Assess third-party risk
        """
        # Extract third-party metadata features
        tp_features = self.extract_third_party_features(entity_features, metadata)
        tp_features_tensor = torch.tensor(
            tp_features,
            dtype=torch.float32,
            device=pooled_output.device
        ).unsqueeze(0)
        
        # Combine
        combined = torch.cat([pooled_output, tp_features_tensor], dim=1)
        features = self.feature_combiner(combined)
        
        # Detect red flags
        excessive_fee = self.sigmoid(self.excessive_fee_detector(features))
        relationship = self.sigmoid(self.relationship_detector(features))
        qualification = self.sigmoid(self.qualification_detector(features))
        transparency = self.sigmoid(self.transparency_detector(features))
        
        # Aggregate risk
        risk_score = torch.mean(torch.stack([
            excessive_fee,
            relationship,
            1 - qualification,  # Low qualification = high risk
            1 - transparency     # Low transparency = high risk
        ]), dim=0)
        
        return {
            'score': risk_score,
            'red_flags': {
                'excessive_fee': excessive_fee,
                'government_relationship': relationship,
                'lacks_qualifications': 1 - qualification,
                'lacks_transparency': 1 - transparency
            }
        }
    
    def extract_third_party_features(self, entity_features, metadata):
        """
        Extract structured features about third parties
        """
        features = [0.0] * 64  # Fixed size feature vector
        
        # Feature 0-9: Organization mentions
        orgs = entity_features.get('ORGANIZATION', [])
        features[0] = min(len(orgs), 10) / 10.0
        
        # Feature 10-19: Financial amounts
        amounts = entity_features.get('MONETARY_AMOUNT', [])
        if amounts:
            max_amount = max([self.parse_amount(a['text']) for a in amounts])
            features[10] = min(max_amount / 1000000, 10) / 10.0  # Scale to 0-1
        
        # Feature 20-29: Geographic risk
        countries = entity_features.get('COUNTRY', [])
        high_risk_countries = ['china', 'russia', 'venezuela', 'nigeria']
        for country in countries:
            if country['text'].lower() in high_risk_countries:
                features[20] = 1.0
                break
        
        # Features 30-63: Reserved for additional metadata
        
        return features
    
    def parse_amount(self, amount_str):
        """Parse dollar amount from string"""
        import re
        match = re.search(r'[\d,]+', amount_str)
        if match:
            return float(match.group().replace(',', ''))
        return 0


class AccountingViolationDetector(nn.Module):
    """
    Detects FCPA accounting provisions violations:
    - Books and records (inaccurate recording)
    - Internal controls (circumvention)
    """
    
    def __init__(self, hidden_size):
        super(AccountingViolationDetector, self).__init__()
        
        # Books and records violation
        self.books_records_head = nn.Sequential(
            nn.Linear(hidden_size, 256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, 4)  # No violation, Minor, Moderate, Severe
        )
        
        # Internal controls violation
        self.internal_controls_head = nn.Sequential(
            nn.Linear(hidden_size, 256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, 4)  # No violation, Minor, Moderate, Severe
        )
        
    def forward(self, sequence_output, pooled_output):
        """
        Detect accounting violations
        """
        books_logits = self.books_records_head(pooled_output)
        controls_logits = self.internal_controls_head(pooled_output)
        
        books_probs = torch.softmax(books_logits, dim=-1)
        controls_probs = torch.softmax(controls_logits, dim=-1)
        
        # Risk weights
        weights = torch.tensor([0.0, 0.3, 0.6, 1.0], device=pooled_output.device)
        
        books_risk = torch.sum(books_probs * weights, dim=1)
        controls_risk = torch.sum(controls_probs * weights, dim=1)
        
        # Overall accounting risk (max of two)
        overall_risk = torch.maximum(books_risk, controls_risk)
        
        return {
            'score': overall_risk,
            'books_records_risk': books_risk,
            'internal_controls_risk': controls_risk,
            'books_logits': books_logits,
            'controls_logits': controls_logits
        }


class FCPARiskAggregator(nn.Module):
    """
    Aggregates all sub-component scores into final risk assessment
    """
    
    def __init__(self, hidden_size):
        super(FCPARiskAggregator, self).__init__()
        
        # Learned aggregation weights
        self.aggregator = nn.Sequential(
            nn.Linear(5 + hidden_size, 256),  # 5 sub-scores + pooled
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, 1),
            nn.Sigmoid()
        )
        
        # Violation type classifier
        self.violation_classifier = nn.Sequential(
            nn.Linear(5 + hidden_size, 256),
            nn.ReLU(),
            nn.Linear(256, 4)  # Anti-bribery, Books/Records, Internal Controls, Multiple
        )
        
        # Attention for explainability
        self.attention = nn.Linear(hidden_size, 1)
        
    def forward(self, foreign_official, payment_intent, corruption_intent,
                third_party, accounting, pooled_output):
        """
        Aggregate all signals
        """
        # Stack sub-scores
        sub_scores = torch.stack([
            foreign_official['score'],
            payment_intent['score'],
            corruption_intent['score'],
            third_party['score'],
            accounting['score']
        ], dim=1)
        
        # Concatenate with pooled output
        combined = torch.cat([sub_scores, pooled_output], dim=1)
        
        # Final risk score
        risk_score = self.aggregator(combined)
        
        # Violation type
        violation_logits = self.violation_classifier(combined)
        violation_probs = torch.softmax(violation_logits, dim=-1)
        
        # Confidence (based on consistency of sub-scores)
        confidence = 1.0 - torch.std(sub_scores, dim=1, keepdim=True)
        
        return {
            'risk_score': risk_score,
            'violation_type': violation_probs,
            'confidence': confidence,
            'sub_scores': sub_scores,
            'attention_weights': None,  # Would compute from sequence
            'key_phrases': []  # Would extract from attention
        }
```

## IV. FCPA DETECTION IN PRODUCTION

### A. Real-Time Detection Pipeline

```python
class FCPADetectionPipeline:
    """
    Production pipeline for FCPA violation detection
    """
    
    def __init__(self):
        self.model = self.load_model()
        self.preprocessor = CommunicationPreprocessor()
        self.feature_extractor = FCPAFeatureExtractor()
        self.ner_model = ComplianceNER()
        self.knowledge_base = FCPAKnowledgeBase()
        
    def detect_violation(self, communication):
        """
        End-to-end detection pipeline
        """
        # Step 1: Preprocess
        processed = self.preprocessor.preprocess(communication)
        
        # Step 2: Extract entities
        entities = self.ner_model.extract_entities(processed['text'])
        
        # Step 3: Extract FCPA-specific features
        features = self.feature_extractor.extract_features(
            processed['text'],
            entities
        )
        
        # Step 4: Run model inference
        with torch.no_grad():
            model_output = self.model(
                input_ids=processed['input_ids'],
                attention_mask=processed['attention_mask'],
                entity_features=entities,
                metadata=communication['metadata']
            )
        
        # Step 5: Apply rule-based validators
        rule_based_score = self.apply_rules(processed['text'], features, entities)
        
        # Step 6: Combine model and rules (ensemble)
        final_score = self.ensemble_scores(
            model_score=model_output['final_risk_score'],
            rule_score=rule_based_score
        )
        
        # Step 7: Generate explanation
        explanation = self.generate_explanation(
            communication=communication,
            model_output=model_output,
            features=features,
            entities=entities
        )
        
        # Step 8: Determine action
        action = self.determine_action(final_score, explanation)
        
        return {
            'risk_score': final_score,
            'severity': self.score_to_severity(final_score),
            'violation_type': model_output['violation_type'],
            'confidence': model_output['confidence'],
            'explanation': explanation,
            'recommended_action': action,
            'regulatory_citations': self.get_relevant_citations(explanation),
            'similar_cases': self.find_similar_cases(communication),
            
            # Detailed sub-scores for attorney review
            'detailed_assessment': {
                'foreign_official_detected': model_output['foreign_official_detected'],
                'foreign_official_score': model_output['foreign_official_score'],
                'payment_intent_score': model_output['payment_intent_score'],
                'corruption_intent_score': model_output['corruption_intent_score'],
                'third_party_risk_score': model_output['third_party_risk_score'],
                'accounting_violation_score': model_output['accounting_violation_score']
            },
            
            # Evidence for investigation
            'evidence': {
                'flagged_phrases': self.extract_key_phrases(communication, model_output),
                'entities': entities,
                'features': features
            }
        }
    
    def apply_rules(self, text, features, entities):
        """
        Rule-based validation (deterministic)
        
        These serve as a safety net and handle clear-cut cases
        """
        score = 0
        
        # Rule 1: Foreign official + payment = HIGH RISK
        if features['num_foreign_officials'] > 0 and features['has_monetary_amount']:
            score += 40
        
        # Rule 2: Third-party + government relationship + high fee = HIGH RISK
        if (features['third_party_red_flags'] >= 2 and 
            features['has_government_references']):
            score += 30
        
        # Rule 3: Explicit bribery keywords
        if features['has_bribery_keywords'] > 0:
            score += 25
        
        # Rule 4: High-risk jurisdiction
        if features['mentions_high_risk_country']:
            score += 15
        
        # Rule 5: Secrecy indicators
        if features['secrecy_indicators'] > 1:
            score += 10
        
        # Rule 6: Accounting red flags
        if any(phrase in text.lower() for phrase in [
            'off the books', 'don\'t record', 'cash payment',
            'no receipt', 'disguise as'
        ]):
            score += 20
        
        return min(score, 100)
    
    def ensemble_scores(self, model_score, rule_score):
        """
        Combine model and rule scores
        
        Use max (most conservative) - if either is high, flag it
        """
        model_val = model_score.item() * 100
        
        # Take maximum of the two
        final = max(model_val, rule_score)
        
        # Boost if both agree (high confidence)
        if model_val > 60 and rule_score > 60:
            final = min(final * 1.1, 100)
        
        return final
    
    def generate_explanation(self, communication, model_output, features, entities):
        """
        Generate human-readable explanation of detection
        
        This is CRITICAL for attorneys to understand and validate
        """
        explanation = []
        
        # Foreign official involvement
        if model_output['foreign_official_detected']:
            officials = entities.get('FOREIGN_OFFICIAL', [])
            if officials:
                explanation.append(
                    f"Communication involves potential foreign official: "
                    f"{', '.join([o['text'] for o in officials[:3]])}"
                )
            else:
                explanation.append(
                    "Language suggests involvement of foreign government official"
                )
        
        # Payment discussion
        if model_output['payment_intent_score'] > 0.5:
            amounts = entities.get('MONETARY_AMOUNT', [])
            if amounts:
                explanation.append(
                    f"Payment discussion detected: "
                    f"{', '.join([a['text'] for a in amounts[:3]])}"
                )
            else:
                explanation.append("Payment or value transfer discussed")
        
        # Corruption intent
        if model_output['corruption_intent_score'] > 0.6:
            if features['quid_pro_quo_signals'] > 0:
                explanation.append(
                    "Language suggests quid pro quo arrangement "
                    "(payment in exchange for official action)"
                )
            if features['willful_blindness_signals'] > 0:
                explanation.append(
                    "Potential willful blindness - conscious avoidance of knowledge"
                )
        
        # Third-party risk
        if model_output['third_party_risk_score'] > 0.5:
            red_flags = [
                flag for flag, value in 
                model_output['detailed_assessment']['third_party_red_flags'].items()
                if value > 0.5
            ]
            if red_flags:
                explanation.append(
                    f"Third-party intermediary red flags: {', '.join(red_flags)}"
                )
        
        # Accounting violations
        if model_output['accounting_violation_score'] > 0.4:
            explanation.append(
                "Potential books and records / internal controls violation"
            )
            if any(phrase in communication['text'].lower() for phrase in [
                'record as', 'book as', 'disguise'
            ]):
                explanation.append(
                    "Language suggests improper or inaccurate recording of transaction"
                )
        
        # Geographic risk
        if features['mentions_high_risk_country']:
            countries = ', '.join(features['high_risk_countries_mentioned'][:3])
            explanation.append(
                f"Transaction involves high-risk jurisdiction: {countries}"
            )
        
        # Combine into paragraph
        if explanation:
            full_explanation = "FCPA Risk Factors:\n" + "\n".join(
                f"• {item}" for item in explanation
            )
        else:
            full_explanation = "No significant FCPA risk factors detected."
        
        return full_explanation
    
    def determine_action(self, risk_score, explanation):
        """
        Determine recommended action based on risk assessment
        """
        if risk_score >= 75:
            return {
                'action': 'IMMEDIATE_LEGAL_REVIEW',
                'priority': 'CRITICAL',
                'timeline': 'Within 2 hours',
                'notify': ['law_firm_partner', 'chief_compliance_officer', 'general_counsel'],
                'next_steps': [
                    'Freeze any pending payments',
                    'Preserve all related documents',
                    'Schedule employee interview within 24 hours',
                    'Assess need for voluntary disclosure to DOJ'
                ]
            }
        
        elif risk_score >= 50:
            return {
                'action': 'INVESTIGATE',
                'priority': 'HIGH',
                'timeline': 'Within 24 hours',
                'notify': ['compliance_officer', 'law_firm_associate'],
                'next_steps': [
                    'Review complete email thread',
                    'Gather related documents',
                    'Interview employee',
                    'Assess third-party relationships'
                ]
            }
        
        elif risk_score >= 25:
            return {
                'action': 'REVIEW_AND_DOCUMENT',
                'priority': 'MEDIUM',
                'timeline': 'Within 3 days',
                'notify': ['compliance_officer'],
                'next_steps': [
                    'Review communication with compliance officer',
                    'Document business justification',
                    'Provide guidance to employee if needed'
                ]
            }
        
        else:
            return {
                'action': 'TRAINING_OPPORTUNITY',
                'priority': 'LOW',
                'timeline': 'Within 2 weeks',
                'notify': ['compliance_officer'],
                'next_steps': [
                    'Consider for training refresher',
                    'Log for compliance program assessment'
                ]
            }
    
    def get_relevant_citations(self, explanation):
        """
        Retrieve relevant statutory and regulatory citations
        """
        citations = []
        
        if 'foreign official' in explanation.lower() or 'payment' in explanation.lower():
            citations.append({
                'citation': '15 USC §78dd-1',
                'title': 'FCPA Anti-Bribery Provisions (Issuers)',
                'summary': 'Prohibits payments to foreign officials to obtain or retain business',

            })
        
        if 'books and records' in explanation.lower() or 'accounting' in explanation.lower():
            citations.append({
                'citation': '15 USC §78m(b)(2)(A)',

            })
        
        if 'internal controls' in explanation.lower():
            citations.append({
                'citation': '15 USC §78m(b)(2)(B)',

                'summary': 'Requires system of internal accounting controls',
                'url': 'https://www.sec.gov/spotlight/fcpa/fcpa-resource-guide.pdf'
            })
        
        return citations
```

```python
        self.financial_fraud_detector = FinancialFraudDetector(hidden_size)
        self.internal_control_detector = InternalControlDetector(hidden_size)
        self.certification_fraud_detector = CertificationFraudDetector(hidden_size)
        self.whistleblower_retaliation_detector = WhistleblowerRetaliationDetector(hidden_size)
        
        # Context aggregator
        self.context_aggregator = SOXContextAggregator(hidden_size)
        
    def forward(self, input_ids, attention_mask, financial_context, employee_context):
        """
        Forward pass with financial and employee context
        
        Args:
            financial_context: Dict with quarter-end proximity, earnings date, etc.
            employee_context: Dict with sender role, department, reporting relationships
        """
        # Encode
        outputs = self.encoder(input_ids=input_ids, attention_mask=attention_mask)
        pooled = outputs.pooler_output
        sequence = outputs.last_hidden_state
        
        # Detect different violation types
        financial_fraud = self.financial_fraud_detector(
            pooled_output=pooled,
            sequence_output=sequence,
            financial_context=financial_context
        )
        
        internal_control = self.internal_control_detector(
            pooled_output=pooled,
            sequence_output=sequence,
            employee_context=employee_context
        )
        
        certification_fraud = self.certification_fraud_detector(
            pooled_output=pooled,
            employee_context=employee_context
        )
        
        whistleblower = self.whistleblower_retaliation_detector(
            pooled_output=pooled,
            sequence_output=sequence,
            employee_context=employee_context
        )
        
        # Aggregate
        aggregated = self.context_aggregator(
            financial_fraud=financial_fraud,
            internal_control=internal_control,
            certification_fraud=certification_fraud,
            whistleblower=whistleblower,
            pooled=pooled
        )
        
        return aggregated


class FinancialFraudDetector(nn.Module):
    """
    Detects financial reporting fraud schemes
    """
    
    def __init__(self, hidden_size):
        super(FinancialFraudDetector, self).__init__()
        
        # Revenue recognition manipulation
        self.revenue_fraud_head = nn.Sequential(
            nn.Linear(hidden_size + 32, 256),  # +32 for financial context features
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, 5)  # Types: None, Timing, Fictitious, Channel Stuffing, etc.
        )
        
        # Cookie jar reserves / improper estimates
        self.reserves_manipulation_head = nn.Sequential(
            nn.Linear(hidden_size + 32, 256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, 4)  # None, Minor, Moderate, Material
        )
        
        # Liability concealment
        self.liability_concealment_head = nn.Sequential(
            nn.Linear(hidden_size + 32, 256),
            nn.ReLU(),
            nn.Linear(256, 3)  # No, Possible, Probable
        )
        
        # Pressure/incentive detector (fraud triangle)
        self.pressure_detector = nn.Sequential(
            nn.Linear(hidden_size, 128),
            nn.ReLU(),
            nn.Linear(128, 1),
            nn.Sigmoid()
        )
        
    def forward(self, pooled_output, sequence_output, financial_context):
        """
        Detect financial fraud schemes
        """
        # Encode financial context (quarter-end, earnings miss risk, etc.)
        fin_context_tensor = self.encode_financial_context(financial_context)
        fin_context_tensor = torch.tensor(
            fin_context_tensor,
            dtype=torch.float32,
            device=pooled_output.device
        ).unsqueeze(0)
        
        # Concatenate
        combined = torch.cat([pooled_output, fin_context_tensor], dim=1)
        
        # Detect fraud types
        revenue_logits = self.revenue_fraud_head(combined)
        reserves_logits = self.reserves_manipulation_head(combined)
        liability_logits = self.liability_concealment_head(combined)
        
        # Detect pressure/incentive
        pressure_score = self.pressure_detector(pooled_output)
        
        # Calculate overall fraud risk
        revenue_probs = torch.softmax(revenue_logits, dim=-1)
        reserves_probs = torch.softmax(reserves_logits, dim=-1)
        liability_probs = torch.softmax(liability_logits, dim=-1)
        
        # Weights for each fraud type
        revenue_risk = torch.sum(
            revenue_probs * torch.tensor([0.0, 0.3, 0.6, 0.8, 1.0], device=pooled_output.device),
            dim=1
        )
        reserves_risk = torch.sum(
            reserves_probs * torch.tensor([0.0, 0.3, 0.6, 1.0], device=pooled_output.device),
            dim=1
        )
        liability_risk = torch.sum(
            liability_probs * torch.tensor([0.0, 0.5, 1.0], device=pooled_output.device),
            dim=1
        )
        
        # Overall financial fraud risk
        overall_risk = torch.maximum(
            torch.maximum(revenue_risk, reserves_risk),
            liability_risk
        )
        
        # Amplify if pressure detected
        overall_risk = overall_risk * (1 + pressure_score.squeeze())
        overall_risk = torch.clamp(overall_risk, 0.0, 1.0)
        
        return {
            'score': overall_risk,
            'revenue_fraud_risk': revenue_risk,
            'reserves_manipulation_risk': reserves_risk,
            'liability_concealment_risk': liability_risk,
            'pressure_score': pressure_score,
            'revenue_logits': revenue_logits,
            'reserves_logits': reserves_logits,
            'liability_logits': liability_logits
        }
    
    def encode_financial_context(self, financial_context):
        """
        Encode financial context into feature vector
        """
        features = [0.0] * 32
        
        # Feature 0: Days to quarter-end (normalized)
        days_to_qe = financial_context.get('days_to_quarter_end', 100)
        features[0] = max(0, 1 - (days_to_qe / 30))  # Higher if closer to QE
        
        # Feature 1: Days to earnings announcement
        days_to_earnings = financial_context.get('days_to_earnings', 100)
        features[1] = max(0, 1 - (days_to_earnings / 30))
        
        # Feature 2: At risk of missing earnings
        features[2] = 1.0 if financial_context.get('earnings_miss_risk', False) else 0.0
        
        # Feature 3: Historical misses
        historical_misses = financial_context.get('historical_earnings_misses', 0)
        features[3] = min(historical_misses / 4, 1.0)
        
        # Feature 4-7: Reserved for additional financial metrics
        
        return features


class InternalControlDetector(nn.Module):
    """
    Detects internal control circumvention or deficiencies
    
    This is core to SOX Section 404 compliance
    """
    
    def __init__(self, hidden_size):
        super(InternalControlDetector, self).__init__()
        
        # Control override detector
        self.override_detector = nn.Sequential(
            nn.Linear(hidden_size, 256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, 1),
            nn.Sigmoid()
        )
        
        # Segregation of duties violation
        self.sod_violation_detector = nn.Sequential(
            nn.Linear(hidden_size + 16, 256),  # +16 for employee context
            nn.ReLU(),
            nn.Linear(256, 1),
            nn.Sigmoid()
        )
        
        # Authorization bypass
        self.authorization_bypass_detector = nn.Sequential(
            nn.Linear(hidden_size, 256),
            nn.ReLU(),
            nn.Linear(256, 1),
            nn.Sigmoid()
        )
        
        # Severity classifier (deficiency severity)
        self.severity_classifier = nn.Sequential(
            nn.Linear(hidden_size, 256),
            nn.ReLU(),
            nn.Linear(256, 4)  # No deficiency, Deficiency, Significant, Material Weakness
        )
        
    def forward(self, pooled_output, sequence_output, employee_context):
        """
        Detect internal control issues
        """
        # Detect control override
        override_score = self.override_detector(pooled_output)
        
        # Encode employee context
        emp_context_tensor = self.encode_employee_context(employee_context)
        emp_context_tensor = torch.tensor(
            emp_context_tensor,
            dtype=torch.float32,
            device=pooled_output.device
        ).unsqueeze(0)
        
        # Detect SOD violation
        sod_combined = torch.cat([pooled_output, emp_context_tensor], dim=1)
        sod_score = self.sod_violation_detector(sod_combined)
        
        # Detect authorization bypass
        auth_bypass_score = self.authorization_bypass_detector(pooled_output)
        
        # Classify severity
        severity_logits = self.severity_classifier(pooled_output)
        severity_probs = torch.softmax(severity_logits, dim=-1)
        
        # Overall control risk
        control_risk = torch.maximum(
            torch.maximum(override_score, sod_score),
            auth_bypass_score
        )
        
        return {
            'score': control_risk,
            'override_detected': override_score > 0.5,
            'override_score': override_score,
            'sod_violation_score': sod_score,
            'authorization_bypass_score': auth_bypass_score,
            'severity_logits': severity_logits,
            'severity_probs': severity_probs
        }
    
    def encode_employee_context(self, employee_context):
        """
        Encode employee role and relationships
        """
        features = [0.0] * 16
        
        # Feature 0: Is C-suite (CEO/CFO)
        role = employee_context.get('role', '').lower()
        features[0] = 1.0 if any(exec_role in role for exec_role in ['ceo', 'cfo', 'chief']) else 0.0
        
        # Feature 1: Is in finance/accounting
        dept = employee_context.get('department', '').lower()
        features[1] = 1.0 if any(fin_dept in dept for fin_dept in ['finance', 'accounting', 'controller']) else 0.0
        
        # Feature 2: Has GL access
        features[2] = 1.0 if employee_context.get('has_gl_access', False) else 0.0
        
        # Feature 3: Has approval authority
        features[3] = 1.0 if employee_context.get('has_approval_authority', False) else 0.0
        
        # Feature 4-15: Reserved for additional context
        
        return features


class CertificationFraudDetector(nn.Module):
    """
    Detects false certification by CEO/CFO (SOX 302/906)
    
    Looks for executives knowingly certifying false financials
    or concealing control deficiencies
    """
    
    def __init__(self, hidden_size):
        super(CertificationFraudDetector, self).__init__()
        
        # False statement detector
        self.false_statement_detector = nn.Sequential(
            nn.Linear(hidden_size, 256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, 1),
            nn.Sigmoid()
        )
        
        # Knowledge of fraud detector
        self.knowledge_of_fraud_detector = nn.Sequential(
            nn.Linear(hidden_size, 256),
            nn.ReLU(),
            nn.Linear(256, 1),
            nn.Sigmoid()
        )
        
        # Control deficiency concealment
        self.concealment_detector = nn.Sequential(
            nn.Linear(hidden_size, 256),
            nn.ReLU(),
            nn.Linear(256, 1),
            nn.Sigmoid()
        )
        
    def forward(self, pooled_output, employee_context):
        """
        Detect certification fraud
        
        Only relevant if sender is CEO/CFO or senior finance
        """
        # Check if sender is relevant (CEO/CFO/Controller)
        role = employee_context.get('role', '').lower()
        is_relevant_role = any(
            exec_role in role 
            for exec_role in ['ceo', 'cfo', 'chief financial', 'controller']
        )
        
        if not is_relevant_role:
            # Not relevant - return zero risk
            return {
                'score': torch.tensor([0.0], device=pooled_output.device),
                'is_relevant': False
            }
        
        # Detect issues
        false_statement_score = self.false_statement_detector(pooled_output)
        knowledge_score = self.knowledge_of_fraud_detector(pooled_output)
        concealment_score = self.concealment_detector(pooled_output)
        
        # Overall certification fraud risk
        cert_fraud_risk = torch.maximum(
            torch.maximum(false_statement_score, knowledge_score),
            concealment_score
        )
        
        return {
            'score': cert_fraud_risk,
            'is_relevant': True,
            'false_statement_score': false_statement_score,
            'knowledge_of_fraud_score': knowledge_score,
            'concealment_score': concealment_score
        }


class WhistleblowerRetaliationDetector(nn.Module):
    """
    Detects potential whistleblower retaliation (SOX 806)
    
    Looks for adverse actions taken against employees who report violations
    """
    
    def __init__(self, hidden_size):
        super(WhistleblowerRetaliationDetector, self).__init__()
        
        # Adverse action detector
        self.adverse_action_detector = nn.Sequential(
            nn.Linear(hidden_size, 256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, 7)  # Types: Termination, Demotion, Suspension, etc.
        )
        
        # Retaliation intent detector
        self.intent_detector = nn.Sequential(
            nn.Linear(hidden_size, 256),
            nn.ReLU(),
            nn.Linear(256, 1),
            nn.Sigmoid()
        )
        
        # Causation detector (action linked to whistleblowing)
        self.causation_detector = nn.Sequential(
            nn.Linear(hidden_size, 256),
            nn.ReLU(),
            nn.Linear(256, 1),
            nn.Sigmoid()
        )
        
    def forward(self, pooled_output, sequence_output, employee_context):
        """
        Detect whistleblower retaliation
        """
        # Detect adverse action type
        adverse_action_logits = self.adverse_action_detector(pooled_output)
        adverse_action_probs = torch.softmax(adverse_action_logits, dim=-1)
        
        # Check if any adverse action present
        # (Index 0 is "no action", indices 1-6 are various adverse actions)
        adverse_action_present = 1 - adverse_action_probs[:, 0]
        
        # Detect retaliation intent
        intent_score = self.intent_detector(pooled_output)
        
        # Detect causation (action because of whistleblowing)
        causation_score = self.causation_detector(pooled_output)
        
        # Overall retaliation risk
        # Requires: (1) Adverse action + (2) Intent + (3) Causation
        retaliation_risk = adverse_action_present * intent_score.squeeze() * causation_score.squeeze()
        
        return {
            'score': retaliation_risk,
            'adverse_action_present': adverse_action_present > 0.5,
            'adverse_action_probs': adverse_action_probs,
            'intent_score': intent_score,
            'causation_score': causation_score
        }


class SOXContextAggregator(nn.Module):
    """
    Aggregates all SOX violation signals
    """
    
    def __init__(self, hidden_size):
        super(SOXContextAggregator, self).__init__()
        
        # Aggregator
        self.aggregator = nn.Sequential(
            nn.Linear(4 + hidden_size, 256),  # 4 sub-scores + pooled
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, 1),
            nn.Sigmoid()
        )
        
        # Violation type classifier
        self.violation_classifier = nn.Sequential(
            nn.Linear(4 + hidden_size, 256),
            nn.ReLU(),
            nn.Linear(256, 5)  # Financial Fraud, Internal Controls, Certification, Whistleblower, Multiple
        )
        
    def forward(self, financial_fraud, internal_control, certification_fraud, whistleblower, pooled):
        """
        Aggregate all SOX signals
        """
        # Stack sub-scores
        sub_scores = torch.stack([
            financial_fraud['score'],
            internal_control['score'],
            certification_fraud['score'],
            whistleblower['score']
        ], dim=1)
        
        # Concatenate
        combined = torch.cat([sub_scores, pooled], dim=1)
        
        # Final risk
        risk_score = self.aggregator(combined)
        
        # Violation type
        violation_logits = self.violation_classifier(combined)
        violation_probs = torch.softmax(violation_logits, dim=-1)
        
        # Confidence
        confidence = 1.0 - torch.std(sub_scores, dim=1, keepdim=True)
        
        return {
            'final_risk_score': risk_score,
            'violation_type_probs': violation_probs,
            'confidence': confidence,
            
            # Detailed sub-scores
            'financial_fraud_score': financial_fraud['score'],
            'internal_control_score': internal_control['score'],
            'certification_fraud_score': certification_fraud['score'],
            'whistleblower_retaliation_score': whistleblower['score'],
            
            # Detailed outputs for each detector
            'financial_fraud_details': financial_fraud,
            'internal_control_details': internal_control,
            'certification_fraud_details': certification_fraud,
            'whistleblower_details': whistleblower
        }
```
