# Comprehensive Technical Architecture for AI-Powered eDiscovery & Compliance Platform

## I. SYSTEM ARCHITECTURE OVERVIEW

### High-Level Architecture Pattern
```
┌─────────────────────────────────────────────────────────────┐
│                     PRESENTATION LAYER                       │
│  Web UI | Mobile Apps | API Gateway | Desktop Client        │
└─────────────────────────────────────────────────────────────┘
                              ↓
┌─────────────────────────────────────────────────────────────┐
│                    APPLICATION LAYER                         │
│  Microservices | Event Bus | Workflow Engine | Cache        │
└─────────────────────────────────────────────────────────────┘
                              ↓
┌─────────────────────────────────────────────────────────────┐
│                      DATA LAYER                              │
│  RDBMS | Document Store | Vector DB | Object Storage        │
└─────────────────────────────────────────────────────────────┘
                              ↓
┌─────────────────────────────────────────────────────────────┐
│                  INFRASTRUCTURE LAYER                        │
│  Kubernetes | Load Balancers | CDN | Monitoring             │
└─────────────────────────────────────────────────────────────┘
```

### Architectural Principles
- **Microservices Architecture**: Independently deployable, scalable services
- **Event-Driven Design**: Asynchronous processing for heavy operations
- **API-First**: All functionality accessible via RESTful/GraphQL APIs
- **Cloud-Native**: Container-based, horizontally scalable
- **Zero-Trust Security**: Authentication/authorization at every layer
- **Polyglot Persistence**: Right database for each use case

**b) Technology-Assisted Review (TAR):**

**TAR 1.0 (Traditional Predictive Coding):**
```python
from sklearn.linear_model import LogisticRegression
from sklearn.feature_extraction.text import TfidfVectorizer

class TAR1Pipeline:
    def __init__(self):
        self.vectorizer = TfidfVectorizer(max_features=10000)
        self.model = LogisticRegression()
        self.training_set = []
    
    def add_training_example(self, document: str, label: bool):
        self.training_set.append((document, label))
    
    def train(self):
        texts, labels = zip(*self.training_set)
        X = self.vectorizer.fit_transform(texts)
        self.model.fit(X, labels)
    
    def predict_batch(self, documents: List[str]) -> List[float]:
        X = self.vectorizer.transform(documents)
        return self.model.predict_proba(X)[:, 1]  # Probability of relevant
```

**TAR 2.0 (Continuous Active Learning):**
```python
from transformers import AutoModelForSequenceClassification, AutoTokenizer
import torch

class TAR2Pipeline:
    def __init__(self, model_name='bert-base-uncased'):
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForSequenceClassification.from_pretrained(
            model_name, num_labels=2
        )
        self.device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        self.model.to(self.device)
    
    def continuous_train(self, document: str, label: int):
        # Update model with each new labeled document
        inputs = self.tokenizer(document, return_tensors='pt', truncation=True, 
                                max_length=512, padding=True)
        inputs = {k: v.to(self.device) for k, v in inputs.items()}
        
        outputs = self.model(**inputs, labels=torch.tensor([label]).to(self.device))
        loss = outputs.loss
        loss.backward()
        # Optimizer step (managed externally)
        
    def rank_for_review(self, documents: List[str]) -> List[Tuple[int, float]]:
        # Use uncertainty sampling to select next documents for review
        predictions = self.predict_batch(documents)
        # Return documents closest to decision boundary
        uncertainties = [abs(0.5 - p) for p in predictions]
        ranked = sorted(enumerate(uncertainties), key=lambda x: x[1])
        return ranked[:100]  # Top 100 most uncertain
```

```python
# Using spaCy with custom entities
import spacy
from spacy.training import Example

class CustomNER:
    def __init__(self):
        self.nlp = spacy.load('en_core_web_trf')  # Transformer-based
        # Add custom entity types
        self.add_custom_entities(['TRANSACTION_ID', 'INTERNAL_PROJECT', 
                                  'COMPETITOR', 'CONFIDENTIAL_DESIGNATION'])
    
    def extract_entities(self, text: str) -> List[Dict]:
        doc = self.nlp(text)
        entities = []
        for ent in doc.ents:
            entities.append({
                'text': ent.text,
                'type': ent.label_,
                'start': ent.start_char,
                'end': ent.end_char,
                'confidence': self._calculate_confidence(ent)
            })
        return entities
    
    def train_on_annotations(self, examples: List[Tuple[str, Dict]]):
        # Incremental training on annotated examples
        optimizer = self.nlp.resume_training()
        for text, annotations in examples:
            doc = self.nlp.make_doc(text)
            example = Example.from_dict(doc, annotations)
            self.nlp.update([example], sgd=optimizer)
```

**e) Compliance Rule Engine:**
```python
class ComplianceRuleEngine:
    def __init__(self):
        self.rules = self._load_rules()
        self.ml_models = self._load_ml_detectors()
    
    def evaluate(self, document: Dict) -> List[Violation]:
        violations = []
        
        # Pattern-based rules (fast)
        for rule in self.rules:
            if rule.matches(document):
                violations.append(Violation(
                    rule_id=rule.id,
                    severity=rule.severity,
                    confidence=1.0,
                    type='rule_based'
                ))
        
        # ML-based detection (slower, more nuanced)
        for detector in self.ml_models:
            result = detector.predict(document)
            if result['violation_probability'] > 0.7:
                violations.append(Violation(
                    rule_id=detector.rule_id,
                    severity=detector.severity,
                    confidence=result['violation_probability'],
                    type='ml_based',
                    explanation=result.get('explanation')
                ))
        
        return self._deduplicate_violations(violations)
    
    def _load_rules(self):
        # Load from database

            BooleanRule(

                pattern='(payment|money|cash) AND (government|official|minister) AND (foreign|overseas)',
                severity='HIGH',

            ),
```

```python
# Temporal workflow definition
from temporalio import workflow, activity
from datetime import timedelta

@workflow.defn
class DocumentIngestionWorkflow:
    @workflow.run
    async def run(self, file_path: str, matter_id: str) -> str:
        # Step 1: Upload to object storage
        object_key = await workflow.execute_activity(
            upload_to_s3,
            args=[file_path],
            start_to_close_timeout=timedelta(minutes=10)
        )
        
        # Step 2: Extract text
        extracted_text = await workflow.execute_activity(
            extract_text,
            args=[object_key],
            start_to_close_timeout=timedelta(minutes=30)
        )
        
        # Step 3: Run OCR if needed
        if extracted_text.needs_ocr:
            ocr_text = await workflow.execute_activity(
                perform_ocr,
                args=[object_key],
                start_to_close_timeout=timedelta(hours=2)
            )
            extracted_text.content += ocr_text
        
        # Step 4: Extract entities
        entities = await workflow.execute_activity(
            extract_entities,
            args=[extracted_text.content],
            start_to_close_timeout=timedelta(minutes=5)
        )
        
        # Step 5: Generate embeddings
        embeddings = await workflow.execute_activity(
            generate_embeddings,
            args=[extracted_text.content],
            start_to_close_timeout=timedelta(minutes=5)
        )
        
        # Step 6: Index in Elasticsearch
        doc_id = await workflow.execute_activity(
            index_document,
            args=[{
                'matter_id': matter_id,
                'object_key': object_key,
```

```javascript
    } else if (priority === 'HIGH') {
```

#### 9. **Compliance Monitoring Service**
**Technology**:

**Real-Time Processing Pipeline:**
```go
// Kafka consumer for real-time email monitoring
package main

import (
    "github.com/confluentinc/confluent-kafka-go/kafka"
    "context"
)

type EmailMonitor struct {
    consumer      *kafka.Consumer
    ruleEngine    *RuleEngine
    alertService  *AlertService
}

func (m *EmailMonitor) Start(ctx context.Context) {
    for {
        select {
        case <-ctx.Done():
            return
        default:
            msg, err := m.consumer.ReadMessage(100 * time.Millisecond)
            if err != nil {
                continue
            }
            
            // Parse email
            email := ParseEmail(msg.Value)
            
            // Evaluate rules in parallel
            violations := m.ruleEngine.EvaluateAll(email)
            
            if len(violations) > 0 {
                // Determine if should block
                shouldBlock := m.shouldBlockEmail(violations)
                
                if shouldBlock {
                    // Prevent email from sending
                    m.blockEmail(email.ID)
                }
                
                // Create alerts
                for _, violation := range violations {
                    m.alertService.CreateAlert(Alert{
                        DocumentID: email.ID,
                        Violation:  violation,
                        Timestamp:  time.Now(),
                        Blocked:    shouldBlock,
                    })
                }
            }
            
            // Commit offset
            m.consumer.CommitMessage(msg)
        }
    }
}

func (m *EmailMonitor) shouldBlockEmail(violations []Violation) bool {
    for _, v := range violations {
        if v.Severity == "CRITICAL" && v.Confidence > 0.9 {
            return true
        }
    }
    return false
}
```

**Rule Engine Implementation:**
```go
type Rule struct {
    ID          string
    Name        string
    Description string
    Severity    string
    Pattern     CompiledPattern
    MLModel     *MLModel
}

type RuleEngine struct {
    rules      []Rule
    mlService  *MLService
}

func (re *RuleEngine) EvaluateAll(doc Document) []Violation {
    violations := make([]Violation, 0)
    
    // Evaluate pattern-based rules (fast path)
    for _, rule := range re.rules {
        if rule.Pattern != nil {
            if matches := rule.Pattern.Match(doc.Content); matches {
                violations = append(violations, Violation{
                    RuleID:     rule.ID,
                    Severity:   rule.Severity,
                    Confidence: 1.0,
                    Type:       "pattern",
                })
            }
        }
    }
    
    // Evaluate ML-based rules (slower, more accurate)
    for _, rule := range re.rules {
        if rule.MLModel != nil {
            prediction := re.mlService.Predict(rule.MLModel, doc)
            if prediction.Probability > rule.Threshold {
                violations = append(violations, Violation{
                    RuleID:      rule.ID,
                    Severity:    rule.Severity,
                    Confidence:  prediction.Probability,
                    Type:        "ml",
                    Explanation: prediction.Explanation,
                })
            }
        }
    }
    
    return violations
}
```

**Encryption:**
```python
from cryptography.fernet import Fernet
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2

# Encrypt sensitive data at application level
class EncryptionService:
    def __init__(self, master_key: bytes):
        kdf = PBKDF2(
            algorithm=hashes.SHA256(),
            length=32,
            salt=b'static_salt',  # Should be per-tenant
            iterations=100000,
        )
        key = base64.urlsafe_b64encode(kdf.derive(master_key))
        self.cipher = Fernet(key)
    
    def encrypt(self, data: str) -> str:
        return self.cipher.encrypt(data.encode()).decode()
    
    def decrypt(self, encrypted_data: str) -> str:
        return self.cipher.decrypt(encrypted_data.encode()).decode()

# Use for PII, sensitive documents
encryption_service = EncryptionService(os.getenv('MASTER_KEY'))
encrypted_ssn = encryption_service.encrypt(user.ssn)
```

**Automated Failover:**
```python
# Health check monitor
class FailoverManager:
    def __init__(self):
        self.primary_healthy = True
        self.route53 = boto3.client('route53')
    
    async def monitor_health(self):
        while True:
            primary_status = await self.check_region_health('us-east-1')
            
            if not primary_status and self.primary_healthy:
                # Primary failed, initiate failover
                logger.critical("Primary region unhealthy, failing over")
                await self.failover_to_dr()
                self.primary_healthy = False
            
            elif primary_status and not self.primary_healthy:
                # Primary recovered
                logger.info("Primary region recovered")
                self.primary_healthy = True
            
            await asyncio.sleep(30)
    
    async def failover_to_dr(self):
        # Update Route53 to point to DR region
        self.route53.change_resource_record_sets(
            HostedZoneId='Z123456',
            ChangeBatch={
                'Changes': [{
                    'Action': 'UPSERT',
                    'ResourceRecordSet': {
                        'Name': 'ediscovery.company.com',
                        'Type': 'A',
                        'AliasTarget': {
                            'HostedZoneId': 'Z789012',  # DR region
                            'DNSName': 'dr-lb.us-west-2.elb.amazonaws.com',
                            'EvaluateTargetHealth': True
                        }
                    }
                }]
            }
        )
        
        # Promote DR PostgreSQL replica to primary
        await self.promote_database_replica()
        
        # Send alerts
        await self.send_alert("CRITICAL: Failover to DR region complete")
```
