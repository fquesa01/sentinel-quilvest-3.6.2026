# Sentinel Counsel LLP - Intelligent Document Processing & Generation

## Feature Overview

This system provides two core capabilities:

1. **Auto-Checklist Satisfaction**: When documents are uploaded to a deal, AI automatically classifies them and checks off matching checklist items
1. **Deal Document Generation**: When deal terms are uploaded (LOI, Term Sheet, or structured input), AI generates transaction documents based on the firm’s past documents and templates

-----

## PART 1: AUTO-CHECKLIST DOCUMENT MATCHING

### Database Schema

```typescript
// Document classification results
export const documentClassifications = pgTable('document_classifications', {
  id: serial('id').primaryKey(),
  documentId: integer('document_id').references(() => dataRoomFiles.id).notNull(),
  dealId: integer('deal_id').references(() => deals.id).notNull(),
  
  // Classification results
  primaryClassification: varchar('primary_classification', { length: 100 }),
  secondaryClassifications: jsonb('secondary_classifications'), // Array of possible matches
  confidence: real('confidence'), // 0-1 confidence score
  
  // Matched checklist item
  matchedChecklistItemId: integer('matched_checklist_item_id').references(() => dealChecklistItems.id),
  matchConfidence: real('match_confidence'),
  matchMethod: varchar('match_method', { length: 50 }), // 'ai_classification', 'keyword', 'filename', 'manual'
  
  // Extraction results
  extractedMetadata: jsonb('extracted_metadata'), // Key data points extracted from doc
  extractedParties: jsonb('extracted_parties'), // Parties identified in document
  extractedDates: jsonb('extracted_dates'), // Important dates found
  extractedAmounts: jsonb('extracted_amounts'), // Dollar amounts found
  
  // Verification
  isVerified: boolean('is_verified').default(false),
  verifiedBy: integer('verified_by').references(() => users.id),
  verifiedAt: timestamp('verified_at'),
  verificationNotes: text('verification_notes'),
  
  // If classification was wrong
  wasReclassified: boolean('was_reclassified').default(false),
  originalClassification: varchar('original_classification', { length: 100 }),
  reclassifiedBy: integer('reclassified_by').references(() => users.id),
  
  classifiedAt: timestamp('classified_at').defaultNow(),
  createdAt: timestamp('created_at').defaultNow(),
});

// Document type definitions for classification
export const documentTypeDefinitions = pgTable('document_type_definitions', {
  id: serial('id').primaryKey(),
  
  // Type identification
  typeCode: varchar('type_code', { length: 100 }).notNull().unique(),
  typeName: varchar('type_name', { length: 255 }).notNull(),
  category: varchar('category', { length: 100 }), // 'contract', 'disclosure', 'title', 'financial', etc.
  transactionType: varchar('transaction_type', { length: 100 }), // 'real_estate', 'acquisition', 'loan', 'all'
  
  // Classification hints
  keywords: jsonb('keywords'), // Keywords that indicate this doc type
  filenamePatterns: jsonb('filename_patterns'), // Regex patterns for filenames
  contentPatterns: jsonb('content_patterns'), // Text patterns in document content
  requiredElements: jsonb('required_elements'), // Elements that MUST be present
  
  // Extraction rules
  extractionSchema: jsonb('extraction_schema'), // What data to extract from this doc type
  
  // Related checklist items (for matching)
  defaultChecklistItemSlugs: jsonb('default_checklist_item_slugs'),
  
  // Learning
  exampleDocumentIds: jsonb('example_document_ids'), // IDs of confirmed examples
  
  isActive: boolean('is_active').default(true),
  createdAt: timestamp('created_at').defaultNow(),
  updatedAt: timestamp('updated_at').defaultNow(),
});

// Classification learning/feedback
export const classificationFeedback = pgTable('classification_feedback', {
  id: serial('id').primaryKey(),
  documentId: integer('document_id').references(() => dataRoomFiles.id).notNull(),
  
  aiClassification: varchar('ai_classification', { length: 100 }),
  correctClassification: varchar('correct_classification', { length: 100 }),
  wasCorrect: boolean('was_correct'),
  
  feedbackBy: integer('feedback_by').references(() => users.id),
  feedbackAt: timestamp('feedback_at').defaultNow(),
  notes: text('notes'),
});
```

### Auto-Matching Service

```typescript
// services/document-classifier.ts

export class DocumentClassificationService {
  private documentTypes: DocumentTypeDefinition[];
  private openai: OpenAI;

  async classifyAndMatch(
    documentId: number,
    dealId: number,
    content: string | Buffer,
    filename: string
  ): Promise<ClassificationResult> {
    
    // Step 1: Quick filename-based pre-classification
    const filenameHints = this.getFilenameHints(filename);
    
    // Step 2: AI Classification
    const aiClassification = await this.classifyWithAI(content, filename, filenameHints);
    
    // Step 3: Find matching checklist items
    const checklistMatches = await this.findChecklistMatches(
      dealId,
      aiClassification.classification.primaryType
    );
    
    // Step 4: Auto-assign if high confidence
    if (aiClassification.classification.confidence >= 0.85 && checklistMatches.length > 0) {
      const bestMatch = checklistMatches[0];
      
      // Link document to checklist item
      await this.linkDocumentToChecklistItem(documentId, bestMatch.id);
      
      // Update checklist item status if document requirement met
      await this.updateChecklistItemStatus(bestMatch.id, documentId);
    }
    
    // Step 5: Store classification results
    await this.storeClassification(documentId, dealId, aiClassification, checklistMatches);
    
    return {
      classification: aiClassification,
      checklistMatches,
      autoAssigned: aiClassification.classification.confidence >= 0.85,
    };
  }

  private async findChecklistMatches(dealId: number, documentType: string): Promise<ChecklistItem[]> {
    // Get the deal's active checklist
    const checklist = await db.query.dealChecklists.findFirst({
      where: eq(dealChecklists.dealId, dealId),
      with: {
        items: {
          with: {
            templateItem: true,
          },
        },
      },
    });

    if (!checklist) return [];

    // Find the document type definition
    const docTypeDef = this.documentTypes.find(dt => dt.typeCode === documentType);
    if (!docTypeDef) return [];

    // Match against checklist items
    const matches: ChecklistItem[] = [];
    
    for (const item of checklist.items) {
      // Check by slug match
      if (docTypeDef.defaultChecklistItemSlugs?.includes(item.templateItem.slug)) {
        matches.push(item);
        continue;
      }
      
      // Check by keyword match in item name
      const itemNameLower = item.templateItem.name.toLowerCase();
      const matchesKeyword = docTypeDef.keywords.some(kw => 
        itemNameLower.includes(kw.toLowerCase())
      );
      if (matchesKeyword) {
        matches.push(item);
      }
    }

    return matches;
  }

  private async updateChecklistItemStatus(
    checklistItemId: number,
    documentId: number
  ): Promise<void> {
    const item = await db.query.dealChecklistItems.findFirst({
      where: eq(dealChecklistItems.id, checklistItemId),
      with: {
        templateItem: true,
        documents: true,
      },
    });

    if (!item) return;

    const requiredDocs = item.templateItem.requiredDocumentCount || 1;
    const currentDocs = item.documents.length + 1; // +1 for the new doc

    if (currentDocs >= requiredDocs) {
      // Mark as complete
      await db.update(dealChecklistItems)
        .set({
          status: 'complete',
          satisfactionMethod: 'document',
          satisfiedAt: new Date(),
        })
        .where(eq(dealChecklistItems.id, checklistItemId));
    } else if (item.status === 'pending') {
      // Mark as in progress
      await db.update(dealChecklistItems)
        .set({ status: 'in_progress' })
        .where(eq(dealChecklistItems.id, checklistItemId));
    }
  }
}
```

```typescript
    const dealTerms = await this.getDealTerms(dealId);
    if (!dealTerms || !dealTerms.isComplete) {
      throw new Error('Deal terms are incomplete. Please fill in required fields.');
    }
    
    // Step 2: Find best template
    const template = await this.findBestTemplate(documentType, dealTerms, options);
    if (!template) {
      throw new Error(`No template found for document type: ${documentType}`);
    }
    
    // Step 3: Generate with AI
    const generatedContent = await this.generateWithAI(
      documentType,
      dealTerms,
      template,
      options
    );
    
    // Step 4: Create document file
    const documentFile = await this.createDocumentFile(
      generatedContent,
      documentType,
      dealTerms,
      options.format || 'docx'
    );
    
    // Step 5: Store generation record
    const generatedDoc = await db.insert(generatedDocuments).values({
      dealId,
      documentType,
      documentName: this.generateDocumentName(documentType, dealTerms),
      templateId: template.id,
      dealTermsId: dealTerms.id,
      termsSnapshotJson: dealTerms, // Snapshot for version tracking
      generatedFileId: documentFile.id,
      status: 'draft',
    }).returning();
    
    // Step 6: Update template usage
    await db.update(documentTemplates)
      .set({
        usageCount: sql`${documentTemplates.usageCount} + 1`,
        lastUsedAt: new Date(),
      })
      .where(eq(documentTemplates.id, template.id));
    
    return generatedDoc[0];
  }

  private async findBestTemplate(
    documentType: string,
    dealTerms: DealTerms,
    options: GenerationOptions
  ): Promise<DocumentTemplate | null> {
    // Query templates matching criteria
    const candidates = await db.query.documentTemplates.findMany({
      where: and(
        eq(documentTemplates.documentType, documentType),
        eq(documentTemplates.isActive, true),
        or(
          eq(documentTemplates.propertyType, dealTerms.propertyType),
          eq(documentTemplates.propertyType, 'all')
        ),
        or(
          eq(documentTemplates.jurisdiction, dealTerms.propertyState),
          eq(documentTemplates.jurisdiction, 'all')
        )
      ),
      orderBy: [
        desc(documentTemplates.isDefault),
        desc(documentTemplates.rating),
        desc(documentTemplates.usageCount),
      ],
    });

    if (candidates.length === 0) return null;
    
    // If user specified a template, use that
    if (options.templateId) {
      return candidates.find(t => t.id === options.templateId) || candidates[0];
    }
    
    // Otherwise return highest ranked
    return candidates[0];
  }

  private async generateWithAI(
    documentType: string,
    dealTerms: DealTerms,
    template: DocumentTemplate,
    options: GenerationOptions
  ): Promise<GeneratedContent> {
    
    const guidance = DOCUMENT_GUIDANCE[documentType] || '';
    
    const prompt = DOCUMENT_GENERATION_PROMPT
      .replace('{documentType}', documentType)
      .replace('{dealTermsJson}', JSON.stringify(dealTerms, null, 2))
      .replace('{templateContent}', template.templateContent)
      .replace('{outputFormat}', options.format || 'docx')
      .replace('{documentSpecificGuidance}', guidance);

    const response = await this.openai.chat.completions.create({
      model: 'gpt-4-turbo-preview',
      messages: [
        { role: 'system', content: 'You are an expert real estate attorney drafting transaction documents.' },
        { role: 'user', content: prompt },
      ],
      temperature: 0.3, // Lower temperature for more consistent legal drafting
    });

    return this.parseGenerationResponse(response.choices[0].message.content);
  }

  private generateDocumentName(documentType: string, dealTerms: DealTerms): string {
    const typeNames: Record<string, string> = {
      psa: 'Purchase and Sale Agreement',
      amendment: 'Amendment to PSA',
      assignment: 'Assignment of Purchase Agreement',
      deed: 'Deed',
      bill_of_sale: 'Bill of Sale',
      closing_certificate: 'Closing Certificate',
      settlement_statement: 'Settlement Statement',
    };
    
    const typeName = typeNames[documentType] || documentType;
    const propertyName = dealTerms.propertyName || dealTerms.propertyAddress?.split(',')[0] || 'Property';
    
    return `${typeName} - ${propertyName}`;
  }
}
```
