# Sentinel Counsel LLP - Deal Templates Feature

## Feature Overview

Deal Templates provide pre-built, comprehensive checklists for common transaction types. Each template contains categories, checklist items, and document requirements that can be:
- **Manually satisfied** - User marks item complete
- **Auto-satisfied** - System detects when required document is uploaded
- **Partially satisfied** - Some but not all required documents present

Templates can be applied to deals, customized per-deal, and tracked through completion.

---

## DATABASE SCHEMA

```typescript
// Deal Templates Schema

// Master template definitions
export const dealTemplates = pgTable('deal_templates', {
  id: serial('id').primaryKey(),
  
  // Template info
  name: varchar('name', { length: 255 }).notNull(),
  slug: varchar('slug', { length: 100 }).notNull().unique(),
  description: text('description'),
  transactionType: varchar('transaction_type', { length: 100 }).notNull(),
  // 'real_estate', 'acquisition', 'loan', 'general_corporate', 'joint_venture', 'financing'
  
  // Template metadata
  version: integer('version').default(1),
  isActive: boolean('is_active').default(true),
  isDefault: boolean('is_default').default(false), // Default template for this transaction type
  
  // Customization
  isSystemTemplate: boolean('is_system_template').default(false), // Built-in vs user-created
  createdBy: integer('created_by').references(() => users.id),
  organizationId: integer('organization_id'), // For org-specific templates
  
  // Statistics
  usageCount: integer('usage_count').default(0),
  
  createdAt: timestamp('created_at').defaultNow(),
  updatedAt: timestamp('updated_at').defaultNow(),
});

// Template categories (sections of the checklist)
export const templateCategories = pgTable('template_categories', {
  id: serial('id').primaryKey(),
  templateId: integer('template_id').references(() => dealTemplates.id).notNull(),
  
  name: varchar('name', { length: 255 }).notNull(),
  description: text('description'),
  sortOrder: integer('sort_order').default(0),
  
  // Display options
  icon: varchar('icon', { length: 50 }), // Lucide icon name
  color: varchar('color', { length: 20 }),
  isCollapsible: boolean('is_collapsible').default(true),
  defaultExpanded: boolean('default_expanded').default(true),
  
  createdAt: timestamp('created_at').defaultNow(),
});

// Individual checklist items within categories
export const templateItems = pgTable('template_items', {
  id: serial('id').primaryKey(),
  categoryId: integer('category_id').references(() => templateCategories.id).notNull(),
  templateId: integer('template_id').references(() => dealTemplates.id).notNull(),
  
  // Item details
  name: varchar('name', { length: 500 }).notNull(),
  description: text('description'),
  guidance: text('guidance'), // Help text for the user
  sortOrder: integer('sort_order').default(0),
  
  // Item configuration
  itemType: varchar('item_type', { length: 50 }).default('checklist'),
  // 'checklist' - simple yes/no
  // 'document' - requires document upload
  // 'document_set' - requires multiple documents
  // 'approval' - requires sign-off
  // 'date' - requires date entry
  // 'text' - requires text input
  
  // Document matching (for auto-satisfaction)
  documentKeywords: jsonb('document_keywords'), // Keywords to match uploaded docs
  documentTypes: jsonb('document_types'), // Expected file types
  requiredDocumentCount: integer('required_document_count').default(1),
  
  // Requirements
  isRequired: boolean('is_required').default(false),
  isCritical: boolean('is_critical').default(false), // Blocks closing if not complete
  
  // Dependencies
  dependsOnItemId: integer('depends_on_item_id'), // Can't start until this is done
  
  // Responsibility
  defaultAssigneeRole: varchar('default_assignee_role', { length: 100 }),
  // 'buyer_counsel', 'seller_counsel', 'buyer', 'seller', 'title_company', 'lender', etc.
  
  // Timing
  typicalDaysFromStart: integer('typical_days_from_start'),
  typicalDaysBeforeClose: integer('typical_days_before_close'),
  
  createdAt: timestamp('created_at').defaultNow(),
});

// When a template is applied to a deal, create instance records
export const dealChecklists = pgTable('deal_checklists', {
  id: serial('id').primaryKey(),
  dealId: integer('deal_id').references(() => deals.id).notNull(),
  templateId: integer('template_id').references(() => dealTemplates.id).notNull(),
  
  // Status tracking
  totalItems: integer('total_items').default(0),
  completedItems: integer('completed_items').default(0),
  percentComplete: real('percent_complete').default(0),
  
  // Key dates (set when template applied)
  effectiveDate: date('effective_date'),
  targetCloseDate: date('target_close_date'),
  
  appliedBy: integer('applied_by').references(() => users.id),
  appliedAt: timestamp('applied_at').defaultNow(),
});

// Individual checklist item instances for a deal
export const dealChecklistItems = pgTable('deal_checklist_items', {
  id: serial('id').primaryKey(),
  dealChecklistId: integer('deal_checklist_id').references(() => dealChecklists.id).notNull(),
  templateItemId: integer('template_item_id').references(() => templateItems.id).notNull(),
  dealId: integer('deal_id').references(() => deals.id).notNull(),
  
  // Instance-specific overrides
  customName: varchar('custom_name', { length: 500 }),
  customDescription: text('custom_description'),
  
  // Status
  status: varchar('status', { length: 50 }).default('pending'),
  // 'pending', 'in_progress', 'waiting', 'complete', 'na', 'waived'
  
  // Satisfaction
  satisfactionMethod: varchar('satisfaction_method', { length: 50 }),
  // 'manual', 'document', 'auto', 'waived'
  satisfiedAt: timestamp('satisfied_at'),
  satisfiedBy: integer('satisfied_by').references(() => users.id),
  
  // Assignment
  assignedTo: integer('assigned_to').references(() => users.id),
  assignedRole: varchar('assigned_role', { length: 100 }),
  
  // Dates
  dueDate: date('due_date'),
  completedDate: date('completed_date'),
  
  // Notes & Comments
  notes: text('notes'),
  internalNotes: text('internal_notes'), // Not visible to external parties
  
  // For date/text item types
  dateValue: date('date_value'),
  textValue: text('text_value'),
  
  // Flags
  isBlocked: boolean('is_blocked').default(false),
  blockedReason: text('blocked_reason'),
  needsAttention: boolean('needs_attention').default(false),
  
  createdAt: timestamp('created_at').defaultNow(),
  updatedAt: timestamp('updated_at').defaultNow(),
});

// Documents linked to checklist items
export const checklistItemDocuments = pgTable('checklist_item_documents', {
  id: serial('id').primaryKey(),
  checklistItemId: integer('checklist_item_id').references(() => dealChecklistItems.id).notNull(),
  documentId: integer('document_id').references(() => dataRoomFiles.id),
  
  // Or external reference
  externalUrl: text('external_url'),
  externalName: varchar('external_name', { length: 255 }),
  
  // Document status
  documentStatus: varchar('document_status', { length: 50 }).default('uploaded'),
  // 'uploaded', 'under_review', 'approved', 'rejected', 'superseded'
  
  // Review
  reviewedBy: integer('reviewed_by').references(() => users.id),
  reviewedAt: timestamp('reviewed_at'),
  reviewNotes: text('review_notes'),
  
  // Version tracking
  versionNumber: integer('version_number').default(1),
  isCurrentVersion: boolean('is_current_version').default(true),
  
  uploadedBy: integer('uploaded_by').references(() => users.id),
  uploadedAt: timestamp('uploaded_at').defaultNow(),
});

// Comments/activity on checklist items
export const checklistItemComments = pgTable('checklist_item_comments', {
  id: serial('id').primaryKey(),
  checklistItemId: integer('checklist_item_id').references(() => dealChecklistItems.id).notNull(),
  
  commentText: text('comment_text').notNull(),
  commentType: varchar('comment_type', { length: 50 }).default('comment'),
  // 'comment', 'status_change', 'assignment', 'document_upload', 'system'
  
  isInternal: boolean('is_internal').default(false), // Only visible to internal team
  
  createdBy: integer('created_by').references(() => users.id),
  createdAt: timestamp('created_at').defaultNow(),
});
```

```tsx
function TemplateCard({ template }: { template: Template }) {
```

## DOCUMENT AUTO-MATCHING LOGIC

```typescript
// services/document-matching.ts

export async function matchDocumentsToChecklist(dealId: number, checklistId: number) {
  // Get all documents in the deal's data room
  const documents = await getDataRoomDocuments(dealId);
  
  // Get all checklist items that accept documents
  const checklistItems = await getChecklistItemsWithDocumentRequirements(checklistId);
  
  const matches: Match[] = [];
  
  for (const doc of documents) {
    const docText = doc.fileName.toLowerCase() + ' ' + (doc.extractedText?.toLowerCase() || '');
    
    for (const item of checklistItems) {
      if (!item.documentKeywords || item.documentKeywords.length === 0) continue;
      
      // Check if document matches any keywords
      const matchedKeywords = item.documentKeywords.filter(keyword => 
        docText.includes(keyword.toLowerCase())
      );
      
      if (matchedKeywords.length > 0) {
        // Calculate confidence score based on keyword matches
        const confidence = matchedKeywords.length / item.documentKeywords.length;
        
        matches.push({
          documentId: doc.id,
          checklistItemId: item.id,
          matchedKeywords,
          confidence,
        });
      }
    }
  }
  
  // Sort by confidence and apply matches
  const sortedMatches = matches.sort((a, b) => b.confidence - a.confidence);
  
  // Apply top matches (avoiding duplicate assignments)
  const assignedDocs = new Set<number>();
  const assignedItems = new Set<number>();
  
  for (const match of sortedMatches) {
    // Skip if document already assigned or item already satisfied
    if (assignedDocs.has(match.documentId)) continue;
    
    // Check if item needs more documents
    const itemDocs = await getChecklistItemDocuments(match.checklistItemId);
    const item = checklistItems.find(i => i.id === match.checklistItemId);
    
    if (itemDocs.length >= (item?.requiredDocumentCount || 1)) continue;
    
    // Create the link
    await linkDocumentToChecklistItem(match.checklistItemId, match.documentId, {
      matchedKeywords: match.matchedKeywords,
      confidence: match.confidence,
      autoMatched: true,
    });
    
    assignedDocs.add(match.documentId);
    
    // Update item status if requirements met
    const updatedDocs = await getChecklistItemDocuments(match.checklistItemId);
    if (updatedDocs.length >= (item?.requiredDocumentCount || 1)) {
      await updateChecklistItem(match.checklistItemId, {
        status: 'complete',
        satisfactionMethod: 'auto',
        satisfiedAt: new Date(),
      });
    }
  }
  
  return matches;
}
```
