2. Risk Level → Color Mapping
Define a mapping function in the UI layer:
type RiskLevel = "critical" | "medium" | "cleared" | "neutral";

function getRiskColor(riskLevel: RiskLevel) {
  switch (riskLevel) {
    case "critical":
      return { barClass: "bg-red-600", label: "Critical" };      // RED
    case "medium":
      return { barClass: "bg-yellow-400", label: "Medium Risk" }; // YELLOW
    case "cleared":
      return { barClass: "bg-green-500", label: "Cleared" };      // GREEN
    default:
      return { barClass: "bg-gray-400", label: "Neutral" };       // Gray bar or neutral
  }
}
All other text and background UI remain grayscale (white, black, gray) so the risk colors stand out.

4. TimelineEventRow (Row + Expanded Panel)
Each event row:
•	Left: Vertical color bar (3–4px wide) indicating risk level
•	Main row: B&W info, collapsed by default
•	Expanded panel below for details
4.1 Collapsed Row Design (B&W + color bar)
Visual:
•	Color bar <div className="w-1 h-full {riskColor.barClass}"></div>
•	Main row content:
o	Date
o	Type (monochrome icon + text)
o	Title
o	Short summary snippet (muted gray)
o	Right side: risk label text + caret icon for expand/collapse
Example JSX:
function TimelineEventRow({ event }: { event: TimelineEvent }) {
  const [expanded, setExpanded] = useState(false);
  const riskColor = getRiskColor(event.risk_level);

  return (
    <div className="flex">
      {/* Color bar */}
      <div className={`w-1 ${riskColor.barClass}`} />

      {/* Main content */}
      <div className="flex-1">
        <button
          className="w-full flex items-center justify-between px-4 py-3 bg-white hover:bg-gray-50"
          onClick={() => setExpanded(!expanded)}
        >
          <div className="flex items-center gap-4 text-sm">
            <div className="w-28 text-gray-700">
              {formatDate(event.event_date)}
            </div>
            <div className="w-24 text-xs uppercase tracking-wide text-gray-500">
              {formatEventType(event.event_type)}
            </div>
            <div className="flex flex-col">
              <div className="font-medium text-gray-900">
                {event.title}
              </div>
              <div className="text-xs text-gray-500 truncate">
                {shortSummarySnippet(event.summary)}
              </div>
            </div>
          </div>
          <div className="flex items-center gap-4">
            <span className="text-xs text-gray-600 italic">
              {riskColor.label}
            </span>
            <ChevronDownIcon
              className={`h-4 w-4 text-gray-500 transform transition-transform ${
                expanded ? "rotate-180" : ""
              }`}
            />
          </div>
        </button>

        {expanded && (
          <TimelineEventDetails event={event} />
        )}
      </div>
    </div>
  );
}
