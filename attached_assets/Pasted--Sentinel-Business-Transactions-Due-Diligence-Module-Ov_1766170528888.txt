# Sentinel Business Transactions & Due Diligence Module

## Overview
This module extends Sentinel to handle business transaction due diligence including M&A, investments, debt transactions, joint ventures, and other business deals. It integrates with existing Sentinel infrastructure (document review, communications analysis, AI capabilities) while adding transaction-specific workflows.

## DATABASE SCHEMA

### Core Tables

```sql
-- =====================================================
-- DEALS / TRANSACTIONS
-- =====================================================

CREATE TABLE deals (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    deal_number VARCHAR(50) UNIQUE NOT NULL, -- e.g., "DEAL-2025-0001"
    title VARCHAR(500) NOT NULL,
    deal_type VARCHAR(100) NOT NULL, -- 'ma_asset', 'ma_stock', 'merger', 'investment', 'debt', 'jv', 'real_estate', 'franchise'
    sub_type VARCHAR(100), -- More specific categorization
    status VARCHAR(50) DEFAULT 'active', -- 'pipeline', 'active', 'on_hold', 'closed', 'terminated'
    priority VARCHAR(20) DEFAULT 'medium', -- 'critical', 'high', 'medium', 'low'
    
    -- Deal Details
    deal_value DECIMAL(20, 2),
    deal_currency VARCHAR(10) DEFAULT 'USD',
    deal_structure TEXT, -- JSON with structure details
    description TEXT,
    
    -- Key Dates
    loi_date DATE,
    signing_target_date DATE,
    closing_target_date DATE,
    actual_signing_date DATE,
    actual_closing_date DATE,
    exclusivity_expiration DATE,
    
    -- Parties (JSON arrays)
    buyer_parties JSONB DEFAULT '[]',
    seller_parties JSONB DEFAULT '[]',
    target_entities JSONB DEFAULT '[]',
    advisors JSONB DEFAULT '[]', -- {role, firm, contacts}
    
    -- Linked Resources
    case_id UUID REFERENCES cases(id), -- Link to existing Sentinel case if needed
    data_room_id UUID,
    
    -- Risk & Scoring
    overall_risk_score INTEGER, -- 1-100
    risk_summary JSONB DEFAULT '{}',
    
    -- Metadata
    created_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    closed_at TIMESTAMP,
    
    -- Settings
    settings JSONB DEFAULT '{}'
);

CREATE INDEX idx_deals_status ON deals(status);
CREATE INDEX idx_deals_type ON deals(deal_type);
CREATE INDEX idx_deals_created ON deals(created_at DESC);

-- =====================================================
-- DEAL TEAM / PARTICIPANTS
-- =====================================================

CREATE TABLE deal_participants (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    deal_id UUID REFERENCES deals(id) ON DELETE CASCADE,
    user_id UUID REFERENCES users(id),
    
    role VARCHAR(100) NOT NULL, -- 'lead', 'reviewer', 'advisor', 'observer', 'counterparty'
    team VARCHAR(100), -- 'buyer', 'seller', 'lender', 'external_counsel'
    permissions JSONB DEFAULT '{}', -- granular permissions
    workstreams TEXT[], -- assigned workstreams
    
    added_at TIMESTAMP DEFAULT NOW(),
    added_by UUID REFERENCES users(id),
    is_active BOOLEAN DEFAULT true
);

CREATE INDEX idx_deal_participants_deal ON deal_participants(deal_id);
CREATE INDEX idx_deal_participants_user ON deal_participants(user_id);

-- =====================================================
-- DEAL MILESTONES & TIMELINE
-- =====================================================

CREATE TABLE deal_milestones (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    title VARCHAR(500) NOT NULL,
    description TEXT,
    milestone_type VARCHAR(100), -- 'signing', 'closing', 'regulatory', 'financing', 'custom'
    
    target_date DATE,
    actual_date DATE,
    status VARCHAR(50) DEFAULT 'pending', -- 'pending', 'in_progress', 'completed', 'delayed', 'cancelled'
    
    dependencies UUID[], -- other milestone IDs
    assignee_id UUID REFERENCES users(id),
    
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    completed_at TIMESTAMP
);

CREATE INDEX idx_deal_milestones_deal ON deal_milestones(deal_id);

-- =====================================================
-- DUE DILIGENCE REQUEST LISTS (DRL)
-- =====================================================

CREATE TABLE drl_templates (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(500) NOT NULL,

    industry VARCHAR(100),
    description TEXT,

    is_active BOOLEAN DEFAULT true,
    
    created_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()

CREATE TABLE deal_request_lists (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    deal_id UUID REFERENCES deals(id) ON DELETE CASCADE,
    
    name VARCHAR(500) NOT NULL,
    version INTEGER DEFAULT 1,
    status VARCHAR(50) DEFAULT 'draft', -- 'draft', 'sent', 'in_progress', 'complete'
    
    template_id UUID REFERENCES drl_templates(id),
    
    sent_to JSONB DEFAULT '[]', -- external parties
    sent_at TIMESTAMP,
    due_date DATE,
    
    created_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE drl_items (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    request_list_id UUID REFERENCES deal_request_lists(id) ON DELETE CASCADE,
    deal_id UUID REFERENCES deals(id) ON DELETE CASCADE,
    
    item_number VARCHAR(50), -- e.g., "1.1", "1.2.3"
    category VARCHAR(200) NOT NULL,
    subcategory VARCHAR(200),
    
    request_text TEXT NOT NULL,
    notes TEXT,
    
    priority VARCHAR(20) DEFAULT 'medium', -- 'critical', 'high', 'medium', 'low'
    status VARCHAR(50) DEFAULT 'requested', -- 'requested', 'received', 'partial', 'under_review', 'flagged', 'cleared', 'na', 'does_not_exist'
    
    assignee_id UUID REFERENCES users(id),
    reviewer_id UUID REFERENCES users(id),
    
    due_date DATE,
    received_date DATE,
    cleared_date DATE,
    
    -- Linked documents
    document_ids UUID[] DEFAULT '{}',
    
    -- Flags
    has_issues BOOLEAN DEFAULT false,
    issue_count INTEGER DEFAULT 0,
    
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_drl_items_list ON drl_items(request_list_id);
CREATE INDEX idx_drl_items_deal ON drl_items(deal_id);
CREATE INDEX idx_drl_items_status ON drl_items(status);
CREATE INDEX idx_drl_items_category ON drl_items(category);

-- =====================================================

CREATE TABLE data_rooms (

    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    folder_id UUID REFERENCES data_room_folders(id),
    
    -- Document info
    document_id UUID, -- Reference to main documents table
    file_name VARCHAR(500) NOT NULL,
    file_type VARCHAR(100),
    file_size BIGINT,
    
    -- Data room specific
    dr_index VARCHAR(50), -- Index number in data room
    version INTEGER DEFAULT 1,
    
    -- AI Analysis
    ai_summary TEXT,
    ai_key_terms JSONB DEFAULT '{}',
    ai_risk_flags JSONB DEFAULT '[]',
    ai_category_suggestion VARCHAR(200),
    
    -- Status

    uploaded_by UUID REFERENCES users(id),
    uploaded_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_dr_docs_room ON data_room_documents(data_room_id);
CREATE INDEX idx_dr_docs_folder ON data_room_documents(folder_id);

CREATE TABLE data_room_access (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    data_room_id UUID REFERENCES data_rooms(id) ON DELETE CASCADE,
    
    -- Access grant
    user_id UUID REFERENCES users(id),
    email VARCHAR(500), -- for external users
    access_type VARCHAR(50), -- 'full', 'limited', 'view_only'
    
    -- Permissions
    can_download BOOLEAN DEFAULT false,
    can_print BOOLEAN DEFAULT false,
    can_copy BOOLEAN DEFAULT false,
    folder_access UUID[], -- specific folders, null = all
    
    -- Access window

    access_expires_at TIMESTAMP,
    
    granted_by UUID REFERENCES users(id),
    is_active BOOLEAN DEFAULT true
);

CREATE TABLE data_room_activity (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    data_room_id UUID REFERENCES data_rooms(id) ON DELETE CASCADE,
    document_id UUID REFERENCES data_room_documents(id),
    user_id UUID REFERENCES users(id),
    
    action VARCHAR(100) NOT NULL, -- 'view', 'download', 'print', 'upload', 'delete'

    created_at TIMESTAMP DEFAULT NOW()

CREATE INDEX idx_dr_activity_room ON data_room_activity(data_room_id);
CREATE INDEX idx_dr_activity_doc ON data_room_activity(document_id);
CREATE INDEX idx_dr_activity_user ON data_room_activity(user_id);

-- =====================================================
-- CONTRACT ANALYSIS
-- =====================================================

CREATE TABLE contract_analyses (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    deal_id UUID REFERENCES deals(id) ON DELETE CASCADE,
    document_id UUID, -- Reference to document
    data_room_doc_id UUID REFERENCES data_room_documents(id),
    
    contract_type VARCHAR(200), -- 'customer_agreement', 'supplier_agreement', 'employment', 'lease', 'license', 'loan', 'insurance'
    contract_name VARCHAR(500),
    
    -- Parties
    parties JSONB DEFAULT '[]', -- [{name, role, entity_type}]
    
    -- Key Terms (extracted)

    expiration_date DATE,
    term_description TEXT,
    auto_renewal BOOLEAN,
    renewal_terms TEXT,

    payment_terms TEXT,

    assignment_provisions JSONB, -- {assignable, consent_required, restrictions}

    mfn_clause JSONB,

    liability_cap JSONB,
    insurance_requirements JSONB,

    -- Risk Assessment

    risk_factors JSONB DEFAULT '[]',
    flags JSONB DEFAULT '[]',
    
    -- AI Analysis
    ai_summary TEXT,
    ai_analysis JSONB DEFAULT '{}',
    ai_confidence_score DECIMAL(5, 2),
    
    -- Review
    review_status VARCHAR(50) DEFAULT 'pending', -- 'pending', 'in_progress', 'reviewed', 'flagged'
    reviewed_by UUID REFERENCES users(id),
    reviewed_at TIMESTAMP,
    review_notes TEXT,
    
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_contract_analysis_deal ON contract_analyses(deal_id);
CREATE INDEX idx_contract_analysis_type ON contract_analyses(contract_type);
CREATE INDEX idx_contract_analysis_risk ON contract_analyses(risk_score DESC);

CREATE TABLE deal_issues (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    deal_id UUID REFERENCES deals(id) ON DELETE CASCADE,
    issue_number VARCHAR(50), -- e.g., "ISS-001"
    
    title VARCHAR(500) NOT NULL,
    description TEXT,
    
    category VARCHAR(200), -- 'legal', 'financial', 'operational', 'regulatory', 'reputational', 'integration', 'tax', 'environmental', 'ip', 'hr', 'it'
    subcategory VARCHAR(200),
    
    severity VARCHAR(20) NOT NULL, -- 'critical', 'high', 'medium', 'low', 'info'
    status VARCHAR(50) DEFAULT 'open', -- 'open', 'investigating', 'pending_resolution', 'resolved', 'accepted', 'closed'
    
    -- Impact Assessment
    is_deal_breaker BOOLEAN DEFAULT false,
    financial_impact DECIMAL(20, 2),
    financial_impact_notes TEXT,
    
    -- Mitigation
    mitigation_strategy TEXT,
    recommended_action VARCHAR(100), -- 'price_adjustment', 'indemnity', 'escrow', 'holdback', 'rep_warranty', 'condition_precedent', 'accept_risk', 'terminate'
    
    -- Links
    source_document_ids UUID[] DEFAULT '{}',
    related_checklist_item_ids UUID[] DEFAULT '{}',
    related_drl_item_ids UUID[] DEFAULT '{}',
    related_contract_ids UUID[] DEFAULT '{}',
    
    -- Assignment
    reported_by UUID REFERENCES users(id),
    assignee_id UUID REFERENCES users(id),
    
    -- Dates
    due_date DATE,
    resolved_at TIMESTAMP,
    
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_deal_issues_deal ON deal_issues(deal_id);
CREATE INDEX idx_deal_issues_severity ON deal_issues(severity);
CREATE INDEX idx_deal_issues_status ON deal_issues(status);
CREATE INDEX idx_deal_issues_category ON deal_issues(category);

    version INTEGER DEFAULT 1,
    
    assignee_id UUID REFERENCES users(id),
    
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_closing_docs_deal ON closing_documents(deal_id);
CREATE INDEX idx_closing_docs_status ON closing_documents(status);

CREATE TABLE post_closing_obligations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    deal_id UUID REFERENCES deals(id) ON DELETE CASCADE,
    
    obligation_type VARCHAR(100), -- 'earnout', 'holdback_release', 'indemnity_period', 'covenant', 'filing', 'integration_milestone'
    
    description TEXT NOT NULL,
    
    due_date DATE,
    reminder_date DATE,
    
    status VARCHAR(50) DEFAULT 'pending', -- 'pending', 'in_progress', 'complete', 'expired'
    
    responsible_party VARCHAR(100),
    assignee_id UUID REFERENCES users(id),
    
    amount DECIMAL(20, 2),
    
    completed_at TIMESTAMP,
    notes TEXT,
    
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_post_closing_deal ON post_closing_obligations(deal_id);
CREATE INDEX idx_post_closing_due ON post_closing_obligations(due_date);

-- =====================================================
-- TRANSACTION REPORTS
-- =====================================================

CREATE TABLE deal_reports (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    deal_id UUID REFERENCES deals(id) ON DELETE CASCADE,
    
    report_type VARCHAR(100), -- 'executive_summary', 'full_dd_report', 'issue_list', 'closing_checklist', 'bring_down'
    
    title VARCHAR(500),
    
    -- Content
    content JSONB DEFAULT '{}',
    generated_content TEXT, -- AI-generated content
    
    -- Generation
    template_id UUID,
    generation_settings JSONB DEFAULT '{}',
    
    -- Status
    status VARCHAR(50) DEFAULT 'draft', -- 'draft', 'in_progress', 'review', 'final'
    
    -- Output
    output_document_id UUID,
    output_format VARCHAR(50), -- 'pdf', 'docx', 'html'
    
    generated_by UUID REFERENCES users(id),
    generated_at TIMESTAMP,
    
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_deal_reports_deal ON deal_reports(deal_id);
```

### Issue Tracker with Risk Matrix

```tsx
// components/business-transactions/issues/RiskMatrix.tsx

import React from 'react';
import { useIssues } from '@/hooks/business-transactions/useIssues';

interface RiskMatrixProps {
  dealId: string;
}

export function RiskMatrix({ dealId }: RiskMatrixProps) {
  const { issues } = useIssues(dealId);

  const categories = ['legal', 'financial', 'operational', 'regulatory', 'reputational', 'integration'];
  const severities = ['critical', 'high', 'medium', 'low'];

  const getIssueCount = (category: string, severity: string) => {
    return issues.filter(
      i => i.category === category && i.severity === severity && i.status !== 'closed'
    ).length;
  };

  const severityColors = {
    critical: 'bg-red-600',
    high: 'bg-orange-500',
    medium: 'bg-yellow-400',
    low: 'bg-green-400',
  };

  return (
    <div className="bg-white rounded-lg border p-6">
      <h3 className="text-lg font-semibold mb-4">Risk Heat Map</h3>
      
      <div className="overflow-x-auto">
        <table className="w-full">
          <thead>
            <tr>
              <th className="text-left p-2">Category</th>
              {severities.map(s => (
                <th key={s} className="text-center p-2 capitalize">{s}</th>
              ))}
            </tr>
          </thead>
          <tbody>
            {categories.map(category => (
              <tr key={category} className="border-t">
                <td className="p-2 capitalize font-medium">{category}</td>
                {severities.map(severity => {
                  const count = getIssueCount(category, severity);
                  return (
                    <td key={severity} className="p-2">
                      <div className="flex justify-center">
                        {count > 0 ? (
                          <div className={`
                            w-10 h-10 rounded-lg flex items-center justify-center
                            text-white font-bold ${severityColors[severity]}
                          `}>
                            {count}
                          </div>
                        ) : (
                          <div className="w-10 h-10 rounded-lg bg-gray-100" />
                        )}
                      </div>
                    </td>
                  );
                })}
              </tr>
            ))}
          </tbody>
        </table>
      </div>

      {/* Summary Stats */}
      <div className="mt-6 grid grid-cols-4 gap-4 pt-4 border-t">
        <div className="text-center">
          <p className="text-2xl font-bold text-red-600">
            {issues.filter(i => i.severity === 'critical' && i.status !== 'closed').length}
          </p>
          <p className="text-sm text-gray-500">Critical Issues</p>
        </div>
        <div className="text-center">
          <p className="text-2xl font-bold text-orange-500">
            {issues.filter(i => i.severity === 'high' && i.status !== 'closed').length}
          </p>
          <p className="text-sm text-gray-500">High Issues</p>
        </div>
        <div className="text-center">
          <p className="text-2xl font-bold text-amber-600">
            {issues.filter(i => i.is_deal_breaker).length}
          </p>
          <p className="text-sm text-gray-500">Deal Breakers</p>
        </div>
        <div className="text-center">
          <p className="text-2xl font-bold text-green-600">
            {issues.filter(i => i.status === 'resolved').length}
          </p>
          <p className="text-sm text-gray-500">Resolved</p>
        </div>
      </div>
    </div>
  );
}
```
