SYSTEM_PROMPT = """
You are an investigative business summarizer. Given enterprise documents (emails, contracts, slide decks,
spreadsheets, PDFs, TXT/MD files), produce ONE canonical business summary JSON object that follows the schema below.
Rules:
- Cite evidence for every factual assertion with a source id and minimal supporting snippet or location.
- If a field is not present in the corpus, set it to null or empty arrays and explain briefly in `notes` where applicable.
- Add a confidence score (0.0–1.0) to any asserted evidence blocks using: direct match (+0.5), multi-source corroboration (+0.3),
  numeric/official doc (+0.2). Cap at 1.0.
- Redact PII/PHI or privileged content; include `redaction_reason` if redacted.
- Output JSON that validates against the schema.
"""

def _file_id(name: str, data: bytes) -> str:
    h = hashlib.sha256(data).hexdigest()[:16]
    return f"{Path(name).name}:{h}"

def extract_text_from_pdf(data: bytes) -> str:
    if PyPDF2 is None:
        return "[PDF extractor not available: install PyPDF2]"
    try:
        reader = PyPDF2.PdfReader(io.BytesIO(data))
        return "\n".join([(p.extract_text() or "") for p in reader.pages]).strip()
    except Exception as e:
        return f"[PDF parse error: {e}]"

def extract_text_from_docx(data: bytes) -> str:
    if docx2txt is None:
        return "[DOCX extractor not available: install docx2txt]"
    try:
        import tempfile
        with tempfile.NamedTemporaryFile(suffix=".docx", delete=True) as tmp:
            tmp.write(data); tmp.flush()
            return (docx2txt.process(tmp.name) or "").strip()
    except Exception as e:
        return f"[DOCX parse error: {e}]"

def extract_text_from_pptx(data: bytes) -> str:
    if pptx is None:
        return "[PPTX extractor not available: install python-pptx]"
    try:
        prs = pptx.Presentation(io.BytesIO(data))
        chunks = []
        for slide in prs.slides:
            for shape in slide.shapes:
                if hasattr(shape, "text"):
                    chunks.append(shape.text)
        return "\n".join(chunks).strip()
    except Exception as e:
        return f"[PPTX parse error: {e}]"

def extract_text_from_xlsx(data: bytes) -> str:
    if pd is None:
        return "[XLSX extractor not available: install pandas, openpyxl]"
    try:
        import tempfile
        with tempfile.NamedTemporaryFile(suffix=".xlsx", delete=True) as tmp:
            tmp.write(data); tmp.flush()
            xl = pd.ExcelFile(tmp.name)
            texts = []
            for sheet in xl.sheet_names[:5]:
                try:
                    df = xl.parse(sheet).fillna("")
                    small = df.head(50)
                    texts.append(f"# Sheet: {sheet}\n{small.to_csv(index=False)}")
                except Exception:
                    pass
            return "\n".join(texts).strip()
    except Exception as e:
        return f"[XLSX parse error: {e}]"

def guess_type(filename: str, default="other") -> str:
    mt, _ = mimetypes.guess_type(filename)
    if mt:
        if "pdf" in mt: return "pdf"
        if "msword" in mt or "officedocument.wordprocessingml" in mt: return "doc"
        if "presentation" in mt: return "slide"
        if "spreadsheet" in mt: return "sheet"
        if mt.startswith("text/"): return "txt"
    ext = Path(filename).suffix.lower()
    return {".pdf":"pdf",".docx":"doc",".doc":"doc",".pptx":"slide",".ppt":"slide",".xlsx":"sheet",".csv":"sheet",".txt":"txt",".md":"txt"}.get(ext, default)

class DocIn(BaseModel):
    id: str
    filename: str
    type: str
    text: str

class SummarizeIn(BaseModel):
    company_name: str
    documents: List[DocIn]

@app.get("/health")
def health():
    return {"ok": True, "time": datetime.datetime.utcnow().isoformat()}

@app.post("/extract")
async def extract(files: List[UploadFile] = File(...)):
    out = []
    for f in files:
        data = await f.read()
        fid = _file_id(f.filename, data)
        ftype = guess_type(f.filename)
        if ftype == "pdf":
            text = extract_text_from_pdf(data)
        elif ftype == "doc":
            text = extract_text_from_docx(data)
        elif ftype == "slide":
            text = extract_text_from_pptx(data)
        elif ftype == "sheet":
            text = extract_text_from_xlsx(data)
        else:
            try:
                text = data.decode("utf-8", errors="replace")
            except Exception:
                text = base64.b64encode(data).decode("utf-8")
                text = f"[binary:{len(data)} bytes base64]\n{text[:4000]}"
        snippet = (text[:400] + "…") if len(text) > 400 else text
        out.append({"id": fid, "filename": f.filename, "type": ftype, "char_count": len(text), "snippet": snippet, "text": text})
    return {"documents": out}

def _render_documents_for_prompt(docs: List[DocIn]) -> str:
    blocks = []
    for d in docs:
        snippet = d.text[:2000]
        blocks.append(f"[{d.id}] ({d.type}) {d.filename}\n---\n{snippet}\n")
    return "\n\n".join(blocks)

@app.post("/summarize")
def summarize(payload: SummarizeIn):
    documents_rendered = _render_documents_for_prompt(payload.documents)
    user_prompt = USER_TEMPLATE.format(company_name=payload.company_name, documents_rendered=documents_rendered)

    body = {
        "model": OPENAI_MODEL,
        "messages": [{"role":"system","content": SYSTEM_PROMPT},{"role":"user","content": user_prompt}],
        "response_format": {"type": "json_object"}
    }
    try:
        import requests
        resp = requests.post(
            "https://api.openai.com/v1/chat/completions",
            headers={"Authorization": f"Bearer {OPENAI_API_KEY}","Content-Type": "application/json"},
            json=body, timeout=120
        )
        resp.raise_for_status()
        data = resp.json()
        content = data["choices"][0]["message"]["content"]
        result = json.loads(content)
    except Exception as e:
        return JSONResponse(status_code=500, content={"error": str(e), "debug_prompt_preview": user_prompt[:3000]})

    if "meta" in result and not result["meta"].get("report_date"):
        result["meta"]["report_date"] = datetime.date.today().isoformat()
    return result
