# Sentinel PE Due Diligence Platform
## Technical Specification for Replit Implementation

---

## Project Overview

**Product Name:** Sentinel Deal Intelligence  
**Target User:** Private Equity firms conducting M&A due diligence  
**Core Value Proposition:** AI-powered due diligence platform that provides institutional memory across deals, real-time intelligence during management calls, and automated pattern matching from historical transactions.

**Design Aesthetic:** Refined, institutional finance aesthetic. Navy blue (#0A1628) primary with warm gold (#C9A962) accents. Typography: Playfair Display for headings, DM Sans for body. Dark mode default. Subtle gradients, crisp data visualization, premium feel appropriate for PE managing directors.

---

## Tech Stack

```
Frontend: React 18 + TypeScript + Vite
Styling: Tailwind CSS + Framer Motion
State: Zustand
Backend: Node.js + Express + TypeScript
Database: PostgreSQL + Prisma ORM
AI/ML: Google Gemini API (via proxy)
Real-time: Socket.io for live transcription
File Processing: pdf-parse, mammoth (docx), xlsx
Search: PostgreSQL full-text search (upgrade path: Pinecone)
Auth: JWT + bcrypt
```

```prisma
// ============ DILIGENCE QUESTIONS ============

model DiligenceQuestion {
  id              String    @id @default(uuid())
  dealId          String
  workstreamId    String?
  question        String    @db.Text
  status          QuestionStatus @default(OPEN)
  priority        Priority  @default(MEDIUM)
  
  // Answer tracking
  answer          String?   @db.Text
  answeredAt      DateTime?
  source          String?   // "Management Call 3/15", "Data Room Doc X"
  
  // Follow-up
  followUpNeeded  Boolean   @default(false)
  followUpNote    String?
  
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt
  
  deal            Deal      @relation(fields: [dealId], references: [id], onDelete: Cascade)
  workstream      Workstream? @relation(fields: [workstreamId], references: [id])
  documents       QuestionDocument[]
  callReferences  CallQuestionReference[]
  
  @@index([dealId, status])
}

enum QuestionStatus {
  OPEN
  PENDING_RESPONSE
  ANSWERED
  CLOSED
  NA
}

enum Priority {
  CRITICAL
  HIGH
  MEDIUM
  LOW
}

model QuestionDocument {
  id              String    @id @default(uuid())

  documentId      String

  question        DiligenceQuestion @relation(fields: [questionId], references: [id], onDelete: Cascade)

// ============ CALLS & TRANSCRIPTION ============

model Call {
  id              String    @id @default(uuid())
  dealId          String
  title           String    // "Management Presentation", "CFO Deep Dive"
  callType        CallType
  scheduledAt     DateTime
  startedAt       DateTime?
  endedAt         DateTime?
  duration        Int?      // minutes
  
  // Recording & Transcription
  recordingUrl    String?

  transcriptProcessed Json? // Structured with speakers, timestamps
  summary         String?   @db.Text
  keyPoints       Json?     // AI-extracted key points
  
  // Pre-call prep
  suggestedQuestions Json?  // AI-generated based on context
  briefingDoc     String?   @db.Text
  
  // Post-call
  followUps       Json?     // Generated follow-up items
  
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt
  
  deal            Deal      @relation(fields: [dealId], references: [id], onDelete: Cascade)
  participants    CallParticipant[]
  documentRefs    CallDocumentReference[]
  questionRefs    CallQuestionReference[]
  
  @@index([dealId, scheduledAt])

model Comment {
  id              String    @id @default(uuid())
  userId          String
  content         String    @db.Text
  
  // Polymorphic reference
  entityType      String    // "DEAL", "DOCUMENT", "RISK_FLAG", "QUESTION"
  entityId        String
  
  parentId        String?   // For threaded comments
  
  createdAt       DateTime  @default(now())

  replies         Comment[] @relation("CommentThread")
  riskFlag        RiskFlag? @relation(fields: [entityId], references: [id], map: "comment_risk_flag")
  
  @@index([entityType, entityId])
```

```typescript
import { useEffect, useState, useCallback } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { useParams } from 'react-router-dom';
import { useWebSocket } from '../hooks/useWebSocket';
import { useCall } from '../hooks/useCalls';

interface TranscriptSegment {
  id: string;
  speaker: string;
  text: string;
  timestamp: number;
  confidence: number;
}

interface DocumentSuggestion {
  id: string;
  documentId: string;
  filename: string;
  relevance: number;
  context: string;
  timestamp: number;
}

interface Inconsistency {
  id: string;
  type: string;
  description: string;
  source1: string;
  source2: string;
  timestamp: number;
}

export function LiveCall() {
  const { callId } = useParams();
  const { call } = useCall(callId);
  const [transcript, setTranscript] = useState<TranscriptSegment[]>([]);
  const [suggestions, setSuggestions] = useState<DocumentSuggestion[]>([]);
  const [inconsistencies, setInconsistencies] = useState<Inconsistency[]>([]);
  const [isRecording, setIsRecording] = useState(false);

  const { sendMessage, lastMessage, connectionStatus } = useWebSocket(
    `/ws/call/${callId}`
  );

  useEffect(() => {
    if (!lastMessage) return;

    const data = JSON.parse(lastMessage.data);

    switch (data.type) {
      case 'transcription':
        setTranscript(prev => [...prev, data.segment]);
        break;
      case 'document_suggestion':
        setSuggestions(prev => [data.suggestion, ...prev].slice(0, 5));
        break;
      case 'inconsistency_detected':
        setInconsistencies(prev => [...prev, data.inconsistency]);
        break;
    }
  }, [lastMessage]);

  const handleStartRecording = useCallback(() => {
    sendMessage(JSON.stringify({ type: 'start_recording' }));
    setIsRecording(true);
  }, [sendMessage]);

  return (
    <div className="min-h-screen bg-slate-950 flex">
      {/* Main Transcript Area */}
      <div className="flex-1 flex flex-col">
        {/* Call Header */}
        <header className="bg-slate-900/80 backdrop-blur-sm border-b border-slate-800 p-4">
          <div className="flex items-center justify-between">

              </div>
              <button
                onClick={isRecording ? () => setIsRecording(false) : handleStartRecording}
                className={`px-4 py-2 rounded-lg font-medium transition-colors
                  ${isRecording 
                    ? 'bg-red-500 hover:bg-red-600 text-white' 
                    : 'bg-amber-500 hover:bg-amber-600 text-slate-900'}`}
              >
                {isRecording ? 'Stop' : 'Start Recording'}
              </button>
            </div>
          </div>
        </header>

        {/* Transcript */}
        <div className="flex-1 overflow-y-auto p-6 space-y-4">
          <AnimatePresence>
            {transcript.map((segment) => (
              <motion.div
                key={segment.id}
                initial={{ opacity: 0, y: 20 }}
                animate={{ opacity: 1, y: 0 }}
                className="flex gap-4"
              >
                <div className="w-24 flex-shrink-0">
                  <span className="text-sm font-medium text-amber-400">{segment.speaker}</span>
```

### 3. Pattern Matching Service

```typescript
// server/services/patternMatching.ts
import { prisma } from '../db';
import { generateEmbedding, findSimilarEmbeddings } from './embeddings';
import { analyzeWithGemini } from './gemini';

interface PatternMatchResult {
  type: string;
  title: string;
  description: string;
  confidence: number;
  historicalDealId?: string;
  historicalContext?: string;
  outcome?: string;
  lesson?: string;
  recommendation?: string;
}

export async function findPatterns(dealId: string): Promise<PatternMatchResult[]> {
  const deal = await prisma.deal.findUnique({
    where: { id: dealId },
    include: {
      documents: true,
      riskFlags: true,
      workstreams: { include: { questions: true } },
    },
  });

  if (!deal) throw new Error('Deal not found');

  const patterns: PatternMatchResult[] = [];

  // 1. Customer Concentration Analysis
  const customerConcentration = await analyzeCustomerConcentration(deal);
  if (customerConcentration) patterns.push(customerConcentration);

  // 2. Key Person Risk
  const keyPersonRisk = await analyzeKeyPersonRisk(deal);
  if (keyPersonRisk) patterns.push(keyPersonRisk);

  // 3. Contract Terms Patterns
  const contractPatterns = await analyzeContractPatterns(deal);
  patterns.push(...contractPatterns);

  // 4. Financial Trend Patterns
  const financialPatterns = await analyzeFinancialPatterns(deal);
  patterns.push(...financialPatterns);

  // 5. Sector-Specific Patterns
  const sectorPatterns = await findSectorPatterns(deal);
  patterns.push(...sectorPatterns);

  // Store patterns in database
  for (const pattern of patterns) {
    await prisma.patternMatch.create({
      data: {
        dealId,
        matchType: pattern.type as any,
        title: pattern.title,
        description: pattern.description,
        confidence: pattern.confidence,
        historicalDealId: pattern.historicalDealId,
        historicalContext: pattern.historicalContext,
        outcome: pattern.outcome,
        lesson: pattern.lesson,
        recommendation: pattern.recommendation,
      },
    });
  }

  return patterns;
}

async function analyzeCustomerConcentration(deal: any): Promise<PatternMatchResult | null> {
  // Find financial documents
  const financialDocs = deal.documents.filter(
    (d: any) => d.category === 'FINANCIAL_STATEMENTS' || d.category === 'QUALITY_OF_EARNINGS'
  );

  if (financialDocs.length === 0) return null;

  // Extract customer concentration data using AI
  const analysisPrompt = `
    Analyze these financial documents for customer concentration risk.
    Extract: top customer revenue %, top 5 customers revenue %, any concentration trends.
    Documents: ${financialDocs.map((d: any) => d.extractedText).join('\n\n')}
  `;

  const analysis = await analyzeWithGemini(analysisPrompt);
  
  // Find historical deals with similar concentration
  const historicalDeals = await prisma.portfolioCompany.findMany({
    where: { firmId: deal.firmId },
  });

  // Match against historical patterns
  const similarDeals = historicalDeals.filter((h: any) => {
    const findings = h.diligenceFindings as any;
    return findings?.customerConcentration > 30; // Example threshold
  });

  if (similarDeals.length === 0) return null;

  return {
    type: 'CUSTOMER_CONCENTRATION',
    title: 'Customer Concentration Risk Identified',
    description: analysis.summary,
    confidence: analysis.concentration > 40 ? 0.9 : 0.7,
    historicalDealId: similarDeals[0].id,
    historicalContext: `${similarDeals[0].name} had ${similarDeals[0].diligenceFindings?.customerConcentration}% concentration`,
    outcome: similarDeals[0].integrationLessons?.concentrationOutcome,
    lesson: 'Customer concentration above 30% has historically required pricing protection',
    recommendation: 'Consider reps & warranties insurance, earnout structure, or price adjustment',
  };
}

async function findSectorPatterns(deal: any): Promise<PatternMatchResult[]> {
  // Generate embedding for deal description
  const dealEmbedding = await generateEmbedding(
    `${deal.sector} ${deal.subsector} ${deal.targetDescription}`
  );

  // Find similar historical deals by embedding similarity
  const similarDeals = await prisma.$queryRaw`
    SELECT 
      pc.id,
      pc.name,
      pc.sector,
      pc."diligenceFindings",
      pc."integrationLessons",
      1 - (pc.embedding <=> ${dealEmbedding}::vector) as similarity
    FROM "PortfolioCompany" pc
    WHERE pc."firmId" = ${deal.firmId}
      AND pc.embedding IS NOT NULL
    ORDER BY pc.embedding <=> ${dealEmbedding}::vector
    LIMIT 5
  `;

  return similarDeals.map((sd: any) => ({
    type: 'SECTOR_SPECIFIC',
    title: `Similar to ${sd.name}`,
    description: `This deal shares characteristics with your prior ${sd.sector} investment`,
    confidence: sd.similarity,
    historicalDealId: sd.id,
    historicalContext: JSON.stringify(sd.diligenceFindings),
    lesson: sd.integrationLessons?.keyLesson,
    recommendation: sd.integrationLessons?.recommendation,
  }));
}
```

### 4. Document Processing Pipeline

```typescript
// server/services/documentProcessor.ts
import { prisma } from '../db';
import { extractTextFromPDF, extractTextFromDocx, extractTextFromXlsx } from './extractors';
import { generateEmbedding } from './embeddings';
import { analyzeWithGemini } from './gemini';

interface ProcessingResult {
  extractedText: string;
  summary: string;
  keyEntities: Record<string, any>;
  embedding: number[];
  suggestedCategory?: string;
}

export async function processDocument(documentId: string): Promise<ProcessingResult> {
  const document = await prisma.document.findUnique({
    where: { id: documentId },
  });

  if (!document) throw new Error('Document not found');

  // Update status to processing
  await prisma.document.update({
    where: { id: documentId },
    data: { status: 'PROCESSING' },
  });

  try {
    // 1. Extract text based on file type
    let extractedText = '';
    switch (document.mimeType) {
      case 'application/pdf':
        extractedText = await extractTextFromPDF(document.storagePath);
        break;
      case 'application/vnd.openxmlformats-officedocument.wordprocessingml.document':
        extractedText = await extractTextFromDocx(document.storagePath);
        break;
      case 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet':
        extractedText = await extractTextFromXlsx(document.storagePath);
        break;
      default:
        extractedText = await extractTextFromPDF(document.storagePath);
    }

    // 2. Generate AI summary and extract entities
    const analysisPrompt = `
      Analyze this legal/financial document and provide:
      1. A concise summary (2-3 sentences)
      2. Key entities (parties, dates, amounts, obligations)
      3. Document type classification
      4. Any red flags or notable terms
      
      Document text:
      ${extractedText.slice(0, 50000)} // Limit for context window
    `;

    const analysis = await analyzeWithGemini(analysisPrompt);

    // 3. Generate embedding for semantic search
    const embedding = await generateEmbedding(
      `${document.filename} ${analysis.summary} ${extractedText.slice(0, 5000)}`
    );

    // 4. Update document with results
    await prisma.document.update({
      where: { id: documentId },
      data: {
        status: 'COMPLETED',
        extractedText,
        summary: analysis.summary,
        keyEntities: analysis.entities,
        embedding: Buffer.from(new Float32Array(embedding).buffer),
        processedAt: new Date(),
      },
    });

    // 5. Check for automatic risk flagging
    if (analysis.redFlags && analysis.redFlags.length > 0) {
      for (const flag of analysis.redFlags) {
        await prisma.riskFlag.create({
          data: {
            dealId: document.dealId,
            title: flag.title,
            description: flag.description,
            category: mapToRiskCategory(flag.category),
            severity: mapToSeverity(flag.severity),
            sourceType: 'DOCUMENT',
            sourceId: documentId,
            sourceContext: flag.context,
            flaggedById: 'SYSTEM',
          },
        });
      }
    }

    return {
      extractedText,
      summary: analysis.summary,
      keyEntities: analysis.entities,
      embedding,
      suggestedCategory: analysis.documentType,
    };

  } catch (error) {
    await prisma.document.update({
      where: { id: documentId },
      data: { status: 'FAILED' },
    });
    throw error;
  }
}

// Batch processor for data room sync
export async function processDataRoomBatch(dealId: string, documents: any[]) {
  const queue = documents.map((doc) => ({
    dealId,
    ...doc,
    priority: categorizePriority(doc.filename, doc.path),
  }));

  // Sort by priority (financials, contracts, then others)
  queue.sort((a, b) => a.priority - b.priority);

  // Process in parallel batches of 5
  const batchSize = 5;
  for (let i = 0; i < queue.length; i += batchSize) {
    const batch = queue.slice(i, i + batchSize);
    await Promise.all(batch.map((doc) => processDocument(doc.id)));
  }
}

function categorizePriority(filename: string, path: string): number {
  const lower = (filename + path).toLowerCase();
  if (lower.includes('financ') || lower.includes('qoe') || lower.includes('quality of earnings')) return 1;
  if (lower.includes('contract') || lower.includes('agreement')) return 2;
  if (lower.includes('cim') || lower.includes('management')) return 3;
  return 4;
}
```
