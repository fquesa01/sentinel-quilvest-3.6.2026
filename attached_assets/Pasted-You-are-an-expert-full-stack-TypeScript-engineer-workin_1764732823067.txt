export type RiskLevel = "low" | "medium" | "high";

export interface TranscriptSegment {
  id: string;
  interviewId: string;
  speaker: "AI_INTERVIEWER" | "SUBJECT";
  startSec: number;
  endSec: number;
  text: string;
  sentiment: "positive" | "neutral" | "negative";
  stressScore: number; // 0-1
  riskLevel: RiskLevel;
  contradictions?: string[]; // ids of other segments or notes
}

export interface QuestionScore {
  questionId: string;
  label: "good" | "okay" | "poor";
  credibility: number;   // 0-1
  riskLevel: RiskLevel;
}

export interface Question {
  id: string;
  prompt: string;
  category: "fact" | "behavior" | "compliance" | "timeline" | "other";
}

export interface Scorecard {
  interviewScore: number;       // 0-100
  credibilityScore: number;     // 0-100
  complianceRiskScore: number;  // 0-100 (higher = riskier)
  documentConflictIndex: number;// 0-100
  questionScores: QuestionScore[];
}

export interface SummaryInsight {
  id: string;
  title: string;
  body: string;
  tag: "admission" | "denial" | "risk" | "timeline" | "policy" | "other";
}

export interface Note {
  id: string;
  interviewId: string;
  author: string;
  body: string;
  createdAt: string;
  isPrivileged: boolean;
  linkedSegmentId?: string;
}

export interface EvidenceLink {
  id: string;
  interviewId: string;
  label: string;
  type: "email" | "document" | "chat" | "other";
  url: string;
  relatedSegments: string[]; // TranscriptSegment ids
}

export interface Interview {
  id: string;
  subjectName: string;
  subjectEmail?: string;

  createdAt: string;
  completedAt?: string;
  language: string;
  riskLevel: RiskLevel;
  credibilityIndex: number; // 0-100
  stressIndex: number;      // 0-100
  evasionProbability: number;// 0-100
  // aggregates:
  summary: SummaryInsight[];
