# Privileged Research Feature Implementation

## Overview

Add a new feature called **"Privileged Research"** to the Sentinel Counsel LLP application. This feature provides users with a Claude-like AI chat interface for conducting legal research and general inquiries under the protection of attorney-client privilege.

The feature button should be placed in the sidebar navigation **directly under the "Ambient Intelligence" button**, with the following properties:
- Icon: A shield with a sparkle or brain icon (suggesting protected AI)
- Label: "Privileged Research"
- Route: `/privileged-research`

---

## Feature Requirements

### Core Functionality

1. **LLM Pass-through Interface**: A clean, Claude-inspired chat interface that allows users to communicate with an LLM of the organization's choosing
2. **Case Context Mode**: Users can optionally select a case to provide context, enabling the AI to access case-specific documents via RAG
3. **General Research Mode**: Users can ask any question without case context for general legal research or any topic
4. **Privilege Protection Banner**: Prominent, always-visible indication that communications are protected under attorney-client privilege
5. **Session History**: Maintain conversation history organized by case association
6. **Multi-Model Support**: Architecture to support different LLM providers (Claude, GPT-4, Gemini)

### Privacy & Security Positioning

- All data remains within the customer's Azure deployment
- No data transmitted to vendor servers
- Communications occur within the attorney-client relationship context
- Audit logging for compliance purposes

## Component Structure

```tsx
import React, { useState, useEffect, useRef } from 'react';
import { useQuery, useMutation } from '@tanstack/react-query';
import { Shield, Sparkles, Send, Plus, FolderOpen, Bot, FileText, Search, Scale, GitCompare, Lightbulb, Lock, MoreVertical, X } from 'lucide-react';

interface PrivilegedSession {
  id: number;
  title: string;
  caseId: number | null;
  caseName: string | null;
  modelProvider: string;
  modelId: string;
  createdAt: string;
  lastActivityAt: string;
}

interface PrivilegedMessage {
  id: number;
  sessionId: number;
  role: 'user' | 'assistant' | 'system';
  content: string;
  referencedDocuments: string[];
  createdAt: string;
}

interface Case {
  id: number;
  name: string;
  caseNumber: string;
}

interface ModelOption {
  provider: string;
  modelId: string;
  displayName: string;
}

export default function PrivilegedResearch() {
  const [activeSessionId, setActiveSessionId] = useState<number | null>(null);
  const [selectedCaseId, setSelectedCaseId] = useState<number | null>(null);
  const [selectedModel, setSelectedModel] = useState<ModelOption | null>(null);
  const [inputMessage, setInputMessage] = useState('');
  const [isStreaming, setIsStreaming] = useState(false);
  const messagesEndRef = useRef<HTMLDivElement>(null);
  const inputRef = useRef<HTMLTextAreaElement>(null);

  // Fetch available cases
  const { data: cases } = useQuery<Case[]>({
    queryKey: ['/api/cases'],
  });

  // Fetch available models
  const { data: models } = useQuery<ModelOption[]>({
    queryKey: ['/api/privileged-research/models'],
  });

  // Fetch recent sessions
  const { data: sessions, refetch: refetchSessions } = useQuery<PrivilegedSession[]>({
    queryKey: ['/api/privileged-research/sessions'],
  });

  // Fetch messages for active session
  const { data: messages, refetch: refetchMessages } = useQuery<PrivilegedMessage[]>({
    queryKey: ['/api/privileged-research/sessions', activeSessionId, 'messages'],
    enabled: !!activeSessionId,
  });

  // Set default model when models load
  useEffect(() => {
    if (models && models.length > 0 && !selectedModel) {
      setSelectedModel(models[0]);
    }
  }, [models]);

    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });

      // Create new session if needed
      if (!sessionId) {
        const response = await fetch('/api/privileged-research/sessions', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            caseId: selectedCaseId,
            modelProvider: selectedModel?.provider || 'anthropic',
            modelId: selectedModel?.modelId || 'claude-3-5-sonnet-20241022',
          }),
        });
        const newSession = await response.json();
        sessionId = newSession.id;
        setActiveSessionId(sessionId);
        refetchSessions();
      }

      // Send message and handle streaming response
      const response = await fetch(`/api/privileged-research/sessions/${sessionId}/messages`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ content: messageContent }),
      });

      if (!response.ok) throw new Error('Failed to send message');

      // Handle streaming response
      const reader = response.body?.getReader();
      const decoder = new TextDecoder();

      if (reader) {
        while (true) {
          const { done, value } = await reader.read();
          if (done) break;
          const chunk = decoder.decode(value);
          // Process streamed chunks - update UI accordingly
          // This would update a streaming message state
        }
      }

      refetchSessions();
    } catch (error) {

    } finally {
      setIsStreaming(false);
    }

// Utility functions
function formatRelativeTime(dateString: string): string {
  const date = new Date(dateString);
  const now = new Date();
  const diffMs = now.getTime() - date.getTime();
  const diffMins = Math.floor(diffMs / 60000);
  const diffHours = Math.floor(diffMs / 3600000);
  const diffDays = Math.floor(diffMs / 86400000);

  if (diffMins < 1) return 'Just now';
  if (diffMins < 60) return `${diffMins} min ago`;
  if (diffHours < 24) return `${diffHours} hours ago`;
  if (diffDays === 1) return 'Yesterday';
  return `${diffDays} days ago`;
}

function formatTime(dateString: string): string {
```

```typescript
    // Set up streaming response
```

### LLM Service

**File: `server/services/llmService.ts`**

```typescript
import Anthropic from '@anthropic-ai/sdk';
import OpenAI from 'openai';
import { GoogleGenerativeAI } from '@google/generative-ai';

interface LLMMessage {
  role: 'system' | 'user' | 'assistant';
  content: string;
}

export function getLLMClient(provider: string, organizationId: number): any {
  // In production, fetch API keys from secure storage based on organizationId
  switch (provider) {
    case 'anthropic':
      return new Anthropic({
        apiKey: process.env.ANTHROPIC_API_KEY,
      });
    case 'openai':
      return new OpenAI({
        apiKey: process.env.OPENAI_API_KEY,
      });
    case 'google':
      return new GoogleGenerativeAI(process.env.GOOGLE_AI_API_KEY || '');
    default:
      throw new Error(`Unknown provider: ${provider}`);
  }
}

export async function streamLLMResponse(
  client: any,
  modelId: string,
  messages: LLMMessage[],
  onChunk: (chunk: string) => void,
  onDocuments?: (docs: string[]) => void
): Promise<void> {
  // Determine provider from client type
  if (client instanceof Anthropic) {
    await streamAnthropic(client, modelId, messages, onChunk);
  } else if (client instanceof OpenAI) {
    await streamOpenAI(client, modelId, messages, onChunk);
  } else {
    await streamGoogle(client, modelId, messages, onChunk);
  }
}

async function streamAnthropic(
  client: Anthropic,
  modelId: string,
  messages: LLMMessage[],
  onChunk: (chunk: string) => void
): Promise<void> {
  const systemMessage = messages.find(m => m.role === 'system')?.content || '';
  const chatMessages = messages
    .filter(m => m.role !== 'system')
    .map(m => ({ role: m.role as 'user' | 'assistant', content: m.content }));

  const stream = await client.messages.stream({
    model: modelId,
    max_tokens: 4096,
    system: systemMessage,
    messages: chatMessages,
  });

  for await (const event of stream) {
    if (event.type === 'content_block_delta' && event.delta.type === 'text_delta') {
      onChunk(event.delta.text);
    }
  }
}

async function streamOpenAI(
  client: OpenAI,
  modelId: string,
  messages: LLMMessage[],
  onChunk: (chunk: string) => void
): Promise<void> {
  const stream = await client.chat.completions.create({
    model: modelId,
    messages: messages.map(m => ({ role: m.role, content: m.content })),
    stream: true,
    max_tokens: 4096,
  });

  for await (const chunk of stream) {
    const content = chunk.choices[0]?.delta?.content;
    if (content) {
      onChunk(content);
    }
  }
}

async function streamGoogle(
  client: GoogleGenerativeAI,
  modelId: string,
  messages: LLMMessage[],
  onChunk: (chunk: string) => void
): Promise<void> {
  const model = client.getGenerativeModel({ model: modelId });
  
  // Convert messages to Gemini format
  const systemInstruction = messages.find(m => m.role === 'system')?.content;
  const chatHistory = messages
    .filter(m => m.role !== 'system')
    .map(m => ({
      role: m.role === 'assistant' ? 'model' : 'user',
      parts: [{ text: m.content }],
    }));

  const chat = model.startChat({
    history: chatHistory.slice(0, -1),
    systemInstruction,
  });

  const lastMessage = chatHistory[chatHistory.length - 1];
  const result = await chat.sendMessageStream(lastMessage.parts[0].text);

  for await (const chunk of result.stream) {
    const text = chunk.text();
    if (text) {
      onChunk(text);
    }
  }
}
```
